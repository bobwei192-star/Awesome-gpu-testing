### 二、实践操作

#### 2.1 完整编译安装流程

```bash
#!/bin/bash
# build_igt.sh - IGT 编译安装脚本
# 参考：IGT 官方文档

set -e

IGT_DIR="${1:-$HOME/igt-gpu-tools}"
INSTALL_PREFIX="${2:-/usr/local}"

echo "=== IGT GPU Tools 编译安装 ==="
echo "源码目录: $IGT_DIR"
echo "安装前缀: $INSTALL_PREFIX"
echo ""

# ═══════════════════════════════════════════════════════════════
# 步骤 1: 安装依赖
# ═══════════════════════════════════════════════════════════════

install_dependencies() {
    echo "【步骤 1】安装编译依赖"
    
    sudo apt-get update
    
    # 必需依赖
    sudo apt-get install -y \
        build-essential \
        meson \
        ninja-build \
        pkg-config \
        libdrm-dev \
        libpciaccess-dev
    
    # 可选依赖（推荐安装）
    sudo apt-get install -y \
        libkmod-dev \
        libprocps-dev \
        libunwind-dev \
        libdw-dev \
        libcairo2-dev \
        libudev-dev \
        libwayland-dev \
        wayland-protocols \
        libx11-dev \
        libxext-dev \
        libxrandr-dev \
        libegl1-mesa-dev \
        libglib2.0-dev \
        python3-docutils \
        valgrind
    
    echo "依赖安装完成"
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 步骤 2: 配置构建
# ═══════════════════════════════════════════════════════════════

configure_build() {
    echo "【步骤 2】配置构建"
    cd "$IGT_DIR"
    
    # 清理旧构建
    rm -rf build
    
    # 配置 meson
    meson setup build \
        --prefix="$INSTALL_PREFIX" \
        --buildtype=release \
        -Ddocs=disabled \
        -Dtests=true \
        -Drunner=true \
        -Dchamelium=disabled \
        -Dvalgrind=disabled
    
    echo "配置完成"
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 步骤 3: 编译
# ═══════════════════════════════════════════════════════════════

compile() {
    echo "【步骤 3】编译"
    cd "$IGT_DIR"
    
    ninja -C build
    
    echo "编译完成"
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 步骤 4: 安装
# ═══════════════════════════════════════════════════════════════

install_igt() {
    echo "【步骤 4】安装"
    cd "$IGT_DIR"
    
    sudo ninja -C build install
    sudo ldconfig
    
    echo "安装完成"
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 步骤 5: 验证安装
# ═══════════════════════════════════════════════════════════════

verify_installation() {
    echo "【步骤 5】验证安装"
    
    # 检查关键工具
    echo "检查安装的工具:"
    for tool in igt-runner modetest kms-steal-crtc; do
        if command -v "$tool" &> /dev/null; then
            echo "  ✓ $tool: $(which $tool)"
        else
            echo "  ✗ $tool: 未找到"
        fi
    done
    
    echo ""
    
    # 检查测试程序
    echo "检查测试程序:"
    if [ -d "$IGT_DIR/build/tests" ]; then
        TEST_COUNT=$(find "$IGT_DIR/build/tests" -maxdepth 1 -type f -executable | wc -l)
        echo "  可执行测试数量: $TEST_COUNT"
    fi
    
    echo ""
    
    # 运行简单测试验证
    echo "运行验证测试 (gem_basic):"
    if [ -f "$IGT_DIR/build/tests/gem_basic" ]; then
        sudo "$IGT_DIR/build/tests/gem_basic" --help 2>/dev/null || true
    fi
    
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 主流程
# ═══════════════════════════════════════════════════════════════

case "${1:-all}" in
    deps)
        install_dependencies
        ;;
    config)
        configure_build
        ;;
    compile)
        compile
        ;;
    install)
        install_igt
        ;;
    verify)
        verify_installation
        ;;
    all)
        install_dependencies
        configure_build
        compile
        install_igt
        verify_installation
        ;;
    *)
        echo "用法: $0 {deps|config|compile|install|verify|all}"
        exit 1
        ;;
esac

echo "=== IGT 编译安装完成 ==="
```

#### 2.2 运行环境配置

```bash
#!/bin/bash
# setup_igt_env.sh - IGT 运行环境配置

set -e

echo "=== IGT 运行环境配置 ==="
echo ""

# 1. 检查 DRM 设备
echo "【1】检查 DRM 设备"
if [ -d /sys/class/drm ]; then
    echo "DRM 设备:"
    ls -la /sys/class/drm/ | grep "card" || true
else
    echo "警告: 未找到 DRM 设备"
fi
echo ""

# 2. 检查权限
echo "【2】检查权限"
if [ -c /dev/dri/card0 ]; then
    echo "card0 权限: $(ls -la /dev/dri/card0)"
    
    # 检查当前用户是否有权限
    if [ -r /dev/dri/card0 ] && [ -w /dev/dri/card0 ]; then
        echo "当前用户有 DRM 设备访问权限"
    else
        echo "当前用户无 DRM 设备访问权限"
        echo "建议: 将用户加入 video 组: sudo usermod -a -G video $USER"
    fi
fi
echo ""

# 3. 检查内核模块
echo "【3】检查内核模块"
lsmod | grep -E "amdgpu|i915|radeon" || echo "未加载 GPU 驱动模块"
echo ""

# 4. 检查 debugfs
echo "【4】检查 debugfs"
if mountpoint -q /sys/kernel/debug; then
    echo "debugfs 已挂载"
    
    # 检查 DRM debug 信息
    if [ -d /sys/kernel/debug/dri ]; then
        echo "DRM debug 信息:"
        ls /sys/kernel/debug/dri/ | head -10
    fi
else
    echo "debugfs 未挂载"
    echo "挂载命令: sudo mount -t debugfs none /sys/kernel/debug"
fi
echo ""

# 5. 环境变量
echo "【5】建议的环境变量"
echo "export IGT_TEST_ROOT=/usr/local/libexec/igt-gpu-tools"
echo "export PATH=\$PATH:/usr/local/libexec/igt-gpu-tools"
echo ""

echo "=== 环境配置完成 ==="
```

## 今日小结

- IGT 编译需要 meson、ninja、libdrm 等依赖
- 可选依赖（cairo、wayland、x11 等）启用更多测试功能
- 运行时需要有 DRM 设备访问权限（video 组）
- debugfs 挂载是部分测试的必要条件
- 编译问题通常与缺少开发包或权限配置有关
