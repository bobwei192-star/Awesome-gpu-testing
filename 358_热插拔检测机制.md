# Day 358: 热插拔检测（Hotplug Detection）机制

## 1. HPD 概述

热插拔检测 (Hotplug Detection, HPD) 是显示子系统中最基础也最关键的功能之一。当用户连接或断开显示器时，系统必须能够及时检测到这一事件，并相应地初始化或释放显示资源。在 AMDGPU DC 子系统中，HPD 机制涵盖了从硬件中断到上层状态管理的完整链路。

### 1.1 HPD 的基本概念

```c
// HPD 事件类型
enum hpd_event_type {
    HPD_EVENT_NONE           = 0,    // 无事件
    HPD_EVENT_CONNECT        = 1,    // 显示器连接 (Long HPD)
    HPD_EVENT_DISCONNECT     = 2,    // 显示器断开 (Long HPD)
    HPD_EVENT_SHORT_PULSE    = 3,    // 短脉冲 (DP 状态变化)
    HPD_EVENT_MST_MSG        = 4,    // MST 侧边消息
    HPD_EVENT_LINK_LOSS      = 5,    // 链路丢失
};

// HPD 引脚编号 (最多支持 6 个)
#define DC_MAX_HPD_PINS      6

// HPD 中断源定义

    DC_IRQ_SOURCE_HPD1       = 0,    // 物理 HPD 引脚 1 (Long HPD)
    DC_IRQ_SOURCE_HPD2       = 1,    // 物理 HPD 引脚 2
    DC_IRQ_SOURCE_HPD3       = 2,    // 物理 HPD 引脚 3
    DC_IRQ_SOURCE_HPD4       = 3,    // 物理 HPD 引脚 4
    DC_IRQ_SOURCE_HPD5       = 4,    // 物理 HPD 引脚 5
    DC_IRQ_SOURCE_HPD6       = 5,    // 物理 HPD 引脚 6

    DC_IRQ_SOURCE_HPD1RX     = 6,    // DP RX 中断 (Short HPD)
    DC_IRQ_SOURCE_HPD2RX     = 7,
    DC_IRQ_SOURCE_HPD3RX     = 8,
    DC_IRQ_SOURCE_HPD4RX     = 9,
    DC_IRQ_SOURCE_HPD5RX     = 10,
    DC_IRQ_SOURCE_HPD6RX     = 11,

```

### 1.2 HPD 硬件原理

```
HPD 信号电气特性:

Long HPD (连接/断开):
  +5V ─────────────────────────
       │                       │
       │  HPD High (>2.25V)    │
       │  = 显示器已连接        │
       │                       │
  0V ──┘                       └─────
       │                       │
       │  HPD Low (<0.8V)      │
       │  = 显示器已断开        │
       │                       │

Short HPD (DP 状态变化):
  +5V ──┐   ┌───┐   ┌───┐   ┌─
       │   │   │   │   │   │
       │   │   │   │   │   │
       │   │   │   │   │   │
  0V  ─┘   └───┘   └───┘   └─
       <2ms><2ms>  <2ms>
       └──── 短脉冲序列 ────┘
       DP 协议要求:
       - 脉冲宽度: 0.25ms - 2ms
       - 脉冲间隔: 0.25ms - 2ms
       - 总序列时长: 2ms - 7ms
```

## 2. 核心数据结构和枚举

### 2.1 hpd_irq_data

```c
// DP HPD RX 中断数据 (映射到 DPCD 200h-205h)
union hpd_irq_data {
    struct {
        // DPCD 200h: SINK_COUNT
        uint8_t sink_cnt;                // 连接的显示器数量 (MST)
        uint8_t reserved1;

        // DPCD 201h: DEVICE_SERVICE_IRQ_VECTOR
        uint8_t device_service_irq;
#define DEVICE_SERVICE_IRQ_VECTOR_AUTOMATED_TEST     (1 << 0)
#define DEVICE_SERVICE_IRQ_VECTOR_UP_REQ_MSG_RDY     (1 << 1)
#define DEVICE_SERVICE_IRQ_VECTOR_DOWN_REP_MSG_RDY   (1 << 2)
#define DEVICE_SERVICE_IRQ_VECTOR_DOWNSTRM_PORT_CHG  (1 << 3)

        // DPCD 202h-203h: LANE0_1_STATUS / LANE2_3_STATUS
        uint8_t lane01_status;
        uint8_t lane23_status;
#define LANE_CR_DONE       (1 << 0)  // Clock Recovery 完成
#define LANE_CHN_EQ_DONE   (1 << 1)  // Channel Equalization 完成
#define LANE_SYMBOL_LOCK   (1 << 2)  // Symbol Lock 完成
#define LANE_STATUS_ALIGN  (1 << 3)  // Lane Align 完成

        // DPCD 204h-205h: LANE0_1_ALIGN / 保留
        uint8_t lane01_align;
#define LANE_ALIGN_CDONE    (1 << 0)
#define LANE_ALIGN_CE_DONE  (1 << 1)
        uint8_t reserved2;
    } bits;

    uint8_t raw[6];  // 原始 6 字节 (200h-205h)
};

// HPD IRQ 数据解析工具
static void dump_hpd_irq_data(union hpd_irq_data *data)
{
    DRM_DEBUG("HPD IRQ Data:\n");
    DRM_DEBUG("  Sink Count: %u\n",
              data->bits.sink_cnt);
    DRM_DEBUG("  Device Service IRQ: 0x%02X\n",
              data->bits.device_service_irq);
    if (data->bits.device_service_irq &
        DEVICE_SERVICE_IRQ_VECTOR_AUTOMATED_TEST)
        DRM_DEBUG("    -> Automated Test Request\n");
    if (data->bits.device_service_irq &
        DEVICE_SERVICE_IRQ_VECTOR_UP_REQ_MSG_RDY)
        DRM_DEBUG("    -> MST Up Request Ready\n");
    if (data->bits.device_service_irq &
        DEVICE_SERVICE_IRQ_VECTOR_DOWN_REP_MSG_RDY)
        DRM_DEBUG("    -> MST Down Reply Ready\n");
    if (data->bits.device_service_irq &
        DEVICE_SERVICE_IRQ_VECTOR_DOWNSTRM_PORT_CHG)
        DRM_DEBUG("    -> Downstream Port Changed\n");
    DRM_DEBUG("  Lane01 Status: 0x%02X\n",
              data->bits.lane01_status);
    DRM_DEBUG("  Lane23 Status: 0x%02X\n",
              data->bits.lane23_status);
}
```

## 3. Long HPD 处理流程

Long HPD 表示显示器的物理连接或断开事件。

### 3.1 Long HPD 中断处理

```c
// Long HPD 中断处理入口
static int handle_hpd_irq(struct amdgpu_device *adev,
                           struct amdgpu_irq_src *source,
                           struct amdgpu_iv_entry *entry)
{
    struct dmub_srv *srv = adev->dmub_srv;
    uint32_t hpd_inst = entry->src_id;

    DRM_DEBUG("Long HPD received: instance=%u\n", hpd_inst);

    // 1. 去抖处理 (硬件或软件去抖)
    if (!dmub_hpd_debounce(srv, hpd_inst))
        return 0;  // 去抖失败，忽略

    // 2. 确定 HPD 状态 (连接/断开)
    bool connected = dmub_read_hpd_state(srv, hpd_inst);

    // 3. 查找对应的链路
    struct dc_link *link = dc_get_link_at_hpd_inst(adev->dm, hpd_inst);
    if (!link) {
        DRM_ERROR("No link found for HPD instance %u\n", hpd_inst);
        return -EINVAL;
    }

    // 4. 更新链路状态
    if (connected) {
        DRM_DEBUG("Display connected on link %d\n",
                  link->link_id);
        link->status = DC_LINK_STATUS_PLUGGED;

        // 调度检测工作
        schedule_work(&link->detect_work);
    } else {
        DRM_DEBUG("Display disconnected from link %d\n",
                  link->link_id);
        link->status = DC_LINK_STATUS_UNPLUGGED;

        // 通知上层断开事件
        drm_kms_helper_connector_hotplug_event(
            link->connector->dev);
    }

    return 0;
}

// HPD 去抖处理
static bool dmub_hpd_debounce(struct dmub_srv *srv,
                               uint32_t hpd_inst)
{
    uint32_t stable_count = 0;
    bool last_state, current_state;

    // 读取当前 HPD 状态
    last_state = dmub_read_hpd_state(srv, hpd_inst);

    // 在去抖窗口内多次采样
    for (int i = 0; i < HPD_DEBOUNCE_SAMPLES; i++) {
        mdelay(HPD_DEBOUNCE_INTERVAL_MS);

        current_state = dmub_read_hpd_state(srv, hpd_inst);

        if (current_state == last_state)
            stable_count++;
        else {
            stable_count = 0;
            last_state = current_state;
        }

        if (stable_count >= HPD_DEBOUNCE_STABLE_THRESHOLD)
            return true;
    }

    // 去抖失败，HPD 信号不稳定
    DRM_WARN("HPD debounce failed for instance %u\n", hpd_inst);
    return false;
}
```

### 3.2 显示检测流程

```c
// 完整的显示检测流程
void dc_link_detect(struct dc_link *link,
                     enum dc_detect_reason reason)
{
    struct dc_sink *sink = NULL;
    struct dc_sink_init_data init_data = {0};

    DRM_DEBUG("Detecting link %d (reason=%d)\n",
              link->link_id, reason);

    // 1. 读取显示器的 DPCD 或 HDMI EDID
    bool sink_found = detect_link_and_local_sink(link, reason);
    if (!sink_found) {
        DRM_DEBUG("No sink found on link %d\n", link->link_id);
        goto out;
    }

    // 2. 创建或更新 sink
    init_data.link = link;
    sink = dc_sink_create(&init_data);
    if (!sink) {
        DRM_ERROR("Failed to create sink for link %d\n",
                  link->link_id);
        goto out;
    }

    // 3. 读取 EDID 获取显示能力
    if (link->connector_signal & SIGNAL_TYPE_DISPLAY_PORT) {
        // DP: 读取 DPCD 和 EDID
        dp_read_dpcd(link, DPCD_ADDR_LINK_BW_SET, 1);
        dp_read_edid(link, sink);
    } else {
        // HDMI/DVI: 通过 DDC 读取 EDID
        hdmi_read_edid(link, sink);
    }

    // 4. 配置链路参数 (lane count, link rate)
    if (link->connector_signal & SIGNAL_TYPE_DISPLAY_PORT) {
        dp_decide_link_settings(link, sink);
    }

    // 5. 分配显示资源
    bool acquired = dc_acquire_resource_for_link(link);
    if (!acquired) {
        DRM_ERROR("Failed to acquire resources for link %d\n",
                  link->link_id);
        dc_sink_release(sink);
        goto out;
    }

    // 6. 将 sink 关联到 link
    dc_sink_retain(sink);
    link->local_sink = sink;

    // 7. 通知 DRM 子系统
    drm_kms_helper_connector_hotplug_event(
        link->connector->dev);

    DRM_DEBUG("Link %d detection complete\n", link->link_id);
    return;

out:
    // 清理
    if (sink)
        dc_sink_release(sink);
    link->status = DC_LINK_STATUS_UNPLUGGED;
}

// 检测连接器上的 sink 设备
static bool detect_link_and_local_sink(struct dc_link *link,
                                        enum dc_detect_reason reason)
{
    // 对于物理端口，通过 GPIO/DMUB 读取 HPD 状态
    if (link->endpoint_type == DISPLAY_ENDPOINT_PHY) {
        return gpio_hpd_get_state(link->hpd_pin);
    }

    // 对于 USB4 DPIA，通过 DMUB 查询
    if (link->endpoint_type == DISPLAY_ENDPOINT_USB4_DPIA) {
        return dpia_query_hpd_status(link);
    }

    return false;
}
```

## 4. Short HPD 处理流程

Short HPD 表示 DP 链路状态发生变化，需要通过 DPCD 读取详细原因。

### 4.1 Short HPD 中断处理

```c
// Short HPD (DP RX 中断) 处理入口
int dc_link_handle_hpd_rx_irq(struct dc_link *link,
                                union hpd_irq_data *irq_data,
                                bool *pending_hpd)
{
    int ret = 0;
    *pending_hpd = false;

    DRM_DEBUG("Handling HPD RX IRQ on link %d\n",
              link->link_id);

    // 1. 读取 HPD IRQ 数据 (DPCD 200h-205h)
    dp_read_hpd_rx_irq_data(link, irq_data);
    dump_hpd_irq_data(irq_data);

    // 2. 检查 sink count 变化 (MST 连接/断开)
    if (irq_data->bits.sink_cnt > 0) {
        ret = handle_sink_count_change(link, irq_data);
        if (ret < 0)
            goto out;
    }

    // 3. 检查设备服务 IRQ
    uint8_t service_irq = irq_data->bits.device_service_irq;

    if (service_irq & DEVICE_SERVICE_IRQ_VECTOR_AUTOMATED_TEST) {
        // 自动化测试请求 (兼容性测试)
        ret = handle_automated_test_request(link);
    }

    if (service_irq & DEVICE_SERVICE_IRQ_VECTOR_UP_REQ_MSG_RDY) {
        // MST 上行请求消息就绪
        ret = handle_mst_up_req(link);
    }

    if (service_irq & DEVICE_SERVICE_IRQ_VECTOR_DOWN_REP_MSG_RDY) {
        // MST 下行回复消息就绪
        ret = handle_mst_down_rep(link);
    }

    if (service_irq & DEVICE_SERVICE_IRQ_VECTOR_DOWNSTRM_PORT_CHG) {
        // 下游端口状态变化 (MST 分支)
        ret = handle_downstream_port_change(link);
    }

    // 4. 检查链路状态
    if (is_link_status_changed(irq_data)) {
        ret = handle_link_status_change(link, irq_data);
    }

out:
    // 5. 清除 IRQ 向量 (DPCD 201h 写 1 清零)
    dp_write_device_service_irq(link, service_irq);

    return ret;
}

// 读取 DP HPD RX IRQ 数据
static bool dp_read_hpd_rx_irq_data(struct dc_link *link,
                                     union hpd_irq_data *irq_data)
{
    // 从 DPCD 200h 地址读取 6 字节
    uint32_t result = dpcd_read(link,
                                DPCD_ADDR_SINK_COUNT,
                                irq_data->raw,
                                sizeof(irq_data->raw));

    if (result != sizeof(irq_data->raw)) {
        DRM_ERROR("Failed to read HPD IRQ data on link %d\n",
                  link->link_id);
        return false;
    }

    return true;
}

// 检查链路状态是否变化
static bool is_link_status_changed(union hpd_irq_data *irq_data)
{
    // 检查 Lane 状态寄存器
    uint8_t lane_status = irq_data->bits.lane01_status |
                          irq_data->bits.lane23_status;

    // 任何 Lane 的 CR/CE/Symbol Lock 失败表示链路问题
    return (lane_status & (LANE_CR_DONE | LANE_CHN_EQ_DONE |
                           LANE_SYMBOL_LOCK)) !=
           (LANE_CR_DONE | LANE_CHN_EQ_DONE | LANE_SYMBOL_LOCK);
}
```

### 4.2 链路状态变化处理

```c
// 处理链路状态变化 (Lane 失败等情况)
static int handle_link_status_change(struct dc_link *link,
                                      union hpd_irq_data *irq_data)
{
    DRM_WARN("Link status changed on link %d\n", link->link_id);

    // 1. 读取详细的链路状态
    uint8_t lane_status[4];
    dpcd_read(link, DPCD_ADDR_LANE0_1_STATUS,
              lane_status, sizeof(lane_status));

    // 2. 记录失败的 Lane
    char failed_lanes[16] = {0};
    for (int i = 0; i < link->cur_link_settings.lane_count; i++) {
        uint8_t lane_stat = (i < 2) ?
            irq_data->bits.lane01_status :
            irq_data->bits.lane23_status;
        uint8_t lane_shift = (i % 2) * 4;

        if (!(lane_stat & (LANE_CR_DONE << lane_shift))) {
            char lane_info[16];
            snprintf(lane_info, sizeof(lane_info), "L%d(CR) ", i);
            strcat(failed_lanes, lane_info);
        }
        if (!(lane_stat & (LANE_CHN_EQ_DONE << lane_shift))) {
            char lane_info[16];
            snprintf(lane_info, sizeof(lane_info), "L%d(CE) ", i);
            strcat(failed_lanes, lane_info);
        }
    }

    DRM_WARN("Failed lanes: %s\n", failed_lanes);

    // 3. 尝试链路恢复
    // 先降低链路参数重试，最后才触发完整重检测
    bool retrained = dp_retrain_link(link,
                                      &link->cur_link_settings,
                                      irq_data);
    if (!retrained) {
        DRM_WARN("Link retrain failed, scheduling full detect\n");
        schedule_work(&link->detect_work);
        return -EIO;
    }

    return 0;
}
```

## 5. DPIA (USB4) HPD 处理

USB4 DPIA (DisplayPort In Adapter) 的 HPD 处理与物理端口不同，需要通过 DMUB 通信。

### 5.1 DPIA HPD 查询

```c
// DPIA HPD 状态查询 (通过 DMCUB)
bool dpia_query_hpd_status(struct dc_link *link)
{
    struct dmub_srv *srv = link->dmub_srv;
    struct dmub_msg msg = {0};
    struct dmub_cmd_query_hpd_state *cmd =
        (struct dmub_cmd_query_hpd_state *)msg.data;
    bool connected = false;

    // 1. 构造 DMUB 命令
    msg.header.type = DMUB_MSG_TYPE_CMD;
    msg.header.length = sizeof(struct dmub_cmd_query_hpd_state);
    msg.header.seq_num = srv->next_seq_num++;
    msg.header.flags = DMUB_MSG_FLAG_RESP_REQ;

    cmd->header.type = DMUB_CMD_QUERY_HPD_STATE;
    cmd->dpia_inst = link->dpia_inst;
    cmd->query_type = DPIA_QUERY_HPD_STATE;

    // 2. 发送命令到 DMCUB
    int ret = dmub_send_command_sync(srv, &msg, 100);
    if (ret) {
        DRM_ERROR("Failed to query DPIA HPD status: %d\n", ret);
        return false;
    }

    // 3. 从响应中读取 HPD 状态
    // DMCUB 通过 notify_queue 返回结果
    struct dmub_msg response;
    ret = dmub_receive_notification(srv, &response);
    if (ret > 0) {
        struct dmub_cmd_query_hpd_state *resp =
            (struct dmub_cmd_query_hpd_state *)response.data;
        connected = resp->hpd_high;
        link->dpia_hpd_status = connected;

        DRM_DEBUG("DPIA %u HPD: %s\n",
                  link->dpia_inst,
                  connected ? "high" : "low");
    }

    return connected;
}

// DPIA HPD 中断处理
void dpia_handle_hpd_irq(struct amdgpu_device *adev,
                          uint32_t dpia_inst)
{
    // DPIA 的 HPD 中断由 DMCUB 代为处理
    // 驱动收到 DMCUB 通知后，查询具体状态

    struct dc_link *link = dc_get_link_at_dpia_inst(adev->dm,
                                                     dpia_inst);
    if (!link) {
        DRM_ERROR("No link for DPIA %u\n", dpia_inst);
        return;
    }

    // 通过 DMCUB 查询 HPD 状态
    bool connected = dpia_query_hpd_status(link);

    if (connected) {
        DRM_DEBUG("DPIA %u: display connected\n", dpia_inst);
        link->status = DC_LINK_STATUS_PLUGGED;
        schedule_work(&link->detect_work);
    } else {
        DRM_DEBUG("DPIA %u: display disconnected\n", dpia_inst);
        link->status = DC_LINK_STATUS_UNPLUGGED;
        drm_kms_helper_connector_hotplug_event(
            link->connector->dev);
    }
}

// DPIA 链路参数配置
struct dpia_link_config {
    // USB4 DPIA 特有的链路参数
    uint32_t dpia_inst;
    uint32_t cable_id;              // 线缆 ID (USB4 标识)
    uint32_t dp_tunnel_throughput;  // DP 隧道带宽 (Gbps)
    uint32_t usb4_link_rate;        // USB4 链路速率

    // 映射到物理 DP 参数
    uint8_t  dp_lane_count;         // DP Lane 数量
```

## 6. 异步工作队列与去重

为了避免 HPD 事件处理冲突，DC 使用异步工作队列并实现了去重机制。

### 6.1 Offload Workqueue

```c
// HPD Offload Workqueue 结构
struct hpd_offload_work {
    struct work_struct work;            // 工作项
    struct dc_link *link;               // 关联链路
    enum dc_detect_reason reason;       // 检测原因
    union hpd_irq_data irq_data;        // HPD IRQ 数据

    // 去重标志
    bool is_handling_link_loss;         // 正在处理链路丢失
    bool is_handling_mst_msg_rdy_event; // 正在处理 MST 消息

    // 完成通知
    struct completion *completion;      // 同步完成通知
};

// 初始化 HPD offload workqueue
static struct workqueue_struct *hpd_workqueue;

int dc_hpd_workqueue_init(struct amdgpu_device *adev)
{
    // 创建专用的 HPD 工作队列
    hpd_workqueue = alloc_ordered_workqueue("amdgpu-dc-hpd",
                                              WQ_HIGHPRI, 0);
    if (!hpd_workqueue) {
        DRM_ERROR("Failed to create HPD workqueue\n");
        return -ENOMEM;
    }

    return 0;
}

// 调度 HPD 检测工作
void dc_hpd_schedule_detect(struct dc_link *link,
                             enum dc_detect_reason reason)
{
    struct hpd_offload_work *work;

    // 检查去重
    if (link->hpd_offload_work &&
        work_pending(&link->hpd_offload_work->work)) {
        DRM_DEBUG("HPD detect already pending for link %d\n",
                  link->link_id);
        return;
    }

    // 分配工作项
    work = kzalloc(sizeof(*work), GFP_KERNEL);
    if (!work)
        return;

    INIT_WORK(&work->work, dc_hpd_detect_work_handler);
    work->link = link;
    work->reason = reason;

    link->hpd_offload_work = work;

    // 提交到工作队列
    queue_work(hpd_workqueue, &work->work);
}

// HPD 检测工作处理函数
static void dc_hpd_detect_work_handler(struct work_struct *w)
{
    struct hpd_offload_work *work =
        container_of(w, struct hpd_offload_work, work);
    struct dc_link *link = work->link;

    DRM_DEBUG("Processing HPD detect for link %d (reason=%d)\n",
              link->link_id, work->reason);

    // 执行实际的链路检测
    dc_link_detect(link, work->reason);

    // 清理
    link->hpd_offload_work = NULL;
    kfree(work);
}
```

### 6.2 去重机制详解

```c
// HPD 去重状态
struct hpd_dedup_state {
    // 上次 HPD 事件的时间戳
    unsigned long last_hpd_jiffies;

    // 上次 HPD 事件的类型
    enum hpd_event_type last_event;

    // 事件计数 (用于检测异常)
    uint32_t event_count;
    uint32_t same_event_count;

    // 去重窗口 (毫秒)
    uint32_t dedup_window_ms;

    // 忽略窗口 (去抖后的稳定期)
    uint32_t ignore_window_ms;
};

// 初始化去重状态
void hpd_dedup_init(struct hpd_dedup_state *state)
{
    state->last_hpd_jiffies = 0;
    state->last_event = HPD_EVENT_NONE;
    state->event_count = 0;
    state->same_event_count = 0;
    state->dedup_window_ms = 100;    // 100ms 去重窗口
    state->ignore_window_ms = 500;   // 500ms 忽略窗口
}

// 检查 HPD 事件是否需要去重
bool hpd_dedup_check(struct hpd_dedup_state *state,
                      enum hpd_event_type event)
{
    unsigned long now = jiffies;
    unsigned long elapsed_ms;

    elapsed_ms = jiffies_to_msecs(now - state->last_hpd_jiffies);

    state->event_count++;

    // 如果距离上次事件太近 (去重窗口内)
    if (elapsed_ms < state->dedup_window_ms) {
        // 如果事件类型相同，丢弃
        if (event == state->last_event) {
            state->same_event_count++;
            DRM_DEBUG("HPD dedup: dropping %s event "
                      "(elapsed=%lu ms, same_count=%u)\n",
                      event == HPD_EVENT_CONNECT ? "CONNECT" :
                      event == HPD_EVENT_DISCONNECT ? "DISCONNECT" :
                      "SHORT_PULSE",
                      elapsed_ms, state->same_event_count);
            return false;  // 去重，丢弃
        }
    }

    // 更新状态
    state->last_hpd_jiffies = now;
    state->last_event = event;

    // 如果同一事件连续出现过多，可能是 HPD 故障
    if (state->same_event_count > 50) {
        DRM_WARN("Excessive HPD events detected (%u in a row)\n",
                 state->same_event_count);
    }

    return true;  // 通过去重，继续处理
}

// 链路丢失去重
bool hpd_is_handling_link_loss(struct dc_link *link)
{
    if (link->hpd_offload_work &&
        link->hpd_offload_work->is_handling_link_loss) {
        DRM_DEBUG("Link loss handling already in progress "
                  "for link %d\n", link->link_id);
        return true;
    }
    return false;
}

// MST 消息去重
bool hpd_is_handling_mst_msg(struct dc_link *link)
{
    if (link->hpd_offload_work &&
        link->hpd_offload_work->is_handling_mst_msg_rdy_event) {
        DRM_DEBUG("MST message handling already in progress "
                  "for link %d\n", link->link_id);
        return true;
    }
    return false;
}
```

## 7. DM IRQ 框架

AMDGPU 的 IRQ 框架将硬件中断路由到 DC 的 HPD 处理器。

### 7.1 IRQ 注册与管理

```c
// DM IRQ 处理器数据结构
struct amdgpu_dm_irq_handler_data {
    struct list_head list;           // 链表节点
    void (*handler)(void *);        // 中断处理函数
    void *handler_arg;               // 处理函数参数
    struct dmub_srv *dmub_srv;       // 关联的 DMUB
    uint32_t irq_source;             // IRQ 源
    bool is_high_tab;                // 是否高优先级表
};

// IRQ 处理器表 (高低优先级)
struct amdgpu_dm_irq_handler_table {
    // 高优先级处理表 (立即处理)
    struct list_head handlers_high_tab[DC_IRQ_SOURCES_NUMBER];

    // 低优先级处理表 (延迟处理)
    struct list_head handlers_low_tab[DC_IRQ_SOURCES_NUMBER];

    // 锁
    spinlock_t handler_lock;
};

// 注册 HPD 中断处理器
int amdgpu_dm_irq_register_interrupt(
    struct amdgpu_device *adev,
    struct amdgpu_dm_irq_handler_data *handler)
{
    struct amdgpu_dm_irq_handler_table *table =
        adev->dm_irq_handler_table;
    unsigned long flags;

    if (handler->irq_source >= DC_IRQ_SOURCES_NUMBER) {
        DRM_ERROR("Invalid IRQ source: %u\n",
                  handler->irq_source);
        return -EINVAL;
    }

    spin_lock_irqsave(&table->handler_lock, flags);

    // 根据优先级添加到对应链表
    if (handler->is_high_tab) {
        list_add_tail(&handler->list,
                      &table->handlers_high_tab[handler->irq_source]);
    } else {
        list_add_tail(&handler->list,
                      &table->handlers_low_tab[handler->irq_source]);
    }

    spin_unlock_irqrestore(&table->handler_lock, flags);

    DRM_DEBUG("Registered IRQ handler for source %u (%s)\n",
              handler->irq_source,
              handler->is_high_tab ? "high" : "low");

    return 0;
}

// 注销 HPD 中断处理器
int amdgpu_dm_irq_unregister_interrupt(
    struct amdgpu_device *adev,
    struct amdgpu_dm_irq_handler_data *handler)
{
    struct amdgpu_dm_irq_handler_table *table =
        adev->dm_irq_handler_table;
    unsigned long flags;

    spin_lock_irqsave(&table->handler_lock, flags);
    list_del(&handler->list);
    spin_unlock_irqrestore(&table->handler_lock, flags);

    return 0;
}

// HPD 中断分发
void amdgpu_dm_irq_dispatch(struct amdgpu_device *adev,
                             uint32_t irq_source)
{
    struct amdgpu_dm_irq_handler_table *table =
        adev->dm_irq_handler_table;
    struct amdgpu_dm_irq_handler_data *handler;
    unsigned long flags;

    if (irq_source >= DC_IRQ_SOURCES_NUMBER)
        return;

    // 先处理高优先级
    spin_lock_irqsave(&table->handler_lock, flags);
    list_for_each_entry(handler,
                        &table->handlers_high_tab[irq_source],
                        list) {
        handler->handler(handler->handler_arg);
    }
    spin_unlock_irqrestore(&table->handler_lock, flags);

    // 再处理低优先级
    spin_lock_irqsave(&table->handler_lock, flags);
    list_for_each_entry(handler,
                        &table->handlers_low_tab[irq_source],
                        list) {
        handler->handler(handler->handler_arg);
    }
    spin_unlock_irqrestore(&table->handler_lock, flags);
}
```

### 7.2 中断处理完整流程

```c
// HPD 中断处理完整流程 (从硬件中断到 DRM 事件)

// Step 1: GPU 硬件中断 → amdgpu_irq_handler
static int amdgpu_irq_handler(struct amdgpu_device *adev,
                               struct amdgpu_irq_src *source,
                               struct amdgpu_iv_entry *entry)
{
    // 将硬件中断映射到 DC IRQ 源
    uint32_t dc_irq_source;

    if (entry->src_id < 6) {
        // Long HPD
        dc_irq_source = DC_IRQ_SOURCE_HPD1 + entry->src_id;
    } else {
        // Short HPD (DP RX)
        dc_irq_source = DC_IRQ_SOURCE_HPD1RX +
                        (entry->src_id - 6);
    }

    // 分发到 DM IRQ 框架
    amdgpu_dm_irq_dispatch(adev, dc_irq_source);

    return 0;
}

// Step 2: DM IRQ 分发 → HPD 处理器
void dm_hpd_irq_handler(void *arg)
{
    struct amdgpu_device *adev = (struct amdgpu_device *)arg;
    uint32_t hpd_inst = adev->current_hpd_instance;

    bool connected = dmub_read_hpd_state(adev->dmub_srv,
                                          hpd_inst);
    struct dc_link *link = dc_get_link_at_hpd_inst(adev->dm,
                                                    hpd_inst);

    if (link && connected) {
        // Step 3: 调度检测工作
        dc_hpd_schedule_detect(link, DETECT_REASON_HPD);
    } else if (link && !connected) {
        // 断开事件
        link->status = DC_LINK_STATUS_UNPLUGGED;
        // Step 4: 通知 DRM
        drm_kms_helper_connector_hotplug_event(
            link->connector->dev);
    }
}

// Step 3: 检测工作 → dc_link_detect
// (参见 3.2 节 dc_link_detect 实现)

// Step 4: DRM 事件 → 用户空间
// drm_kms_helper_connector_hotplug_event 会:
// 1. 更新 connector 状态
// 2. 发送 uevent (通过 sysfs)
// 3. 触发用户空间重新配置 (xrandr --auto)
// 4. 内核中的 DRM 客户端 (modeset) 重检测
```

```c
// 强制触发 HPD 事件 (测试用)

                                    size_t size, loff_t *pos)
{
    struct amdgpu_device *adev = file_inode(f)->i_private;
    uint32_t link_id;
```

```c
// HPD 常见错误分类
enum hpd_common_issue {
    HPD_ISSUE_NONE,                    // 无问题
    HPD_ISSUE_HPD_STUCK_LOW,           // HPD 卡在低电平
    HPD_ISSUE_HPD_STUCK_HIGH,          // HPD 卡在高电平
    HPD_ISSUE_HPD_GLITCH,              // HPD 信号毛刺
    HPD_ISSUE_HPD_STORM,               // HPD 风暴 (频繁 IRQ)
    HPD_ISSUE_DEBOUNCE_FAIL,           // 去抖失败
    HPD_ISSUE_DPCD_READ_FAIL,          // DPCD 读取失败
    HPD_ISSUE_MST_MSG_TIMEOUT,         // MST 消息超时
    HPD_ISSUE_DPIA_COMM_FAIL,          // DPIA 通信失败
};

// HPD 问题诊断函数
enum hpd_common_issue diagnose_hpd_issue(struct dc_link *link)
{
    // 1. 检查 HPD 引脚电平
    uint32_t hpd_state = dmub_read_hpd_state_reg(link);
    uint32_t hpd_pin = link->hpd_pin;

    DRM_INFO("Diagnosing HPD issue on link %d (pin %u):\n",
             link->link_id, hpd_pin);

    // 2. 读取 GPIO 寄存器
    uint32_t gpio_reg = REG_READ(gpio_get_reg(hpd_pin));
    DRM_INFO("  GPIO reg: 0x%08X\n", gpio_reg);

    // 3. 读取中断状态
    uint32_t irq_stat = REG_READ(irq_get_stat_reg(hpd_pin));
    DRM_INFO("  IRQ stat: 0x%08X\n", irq_stat);

    // 4. 检查去抖状态
    uint32_t debounce_stat = REG_READ(DC_HPD_DEBOUNCE_CTRL(hpd_pin));
    DRM_INFO("  Debounce: 0x%08X\n", debounce_stat);

    // 5. 分析问题
    if ((gpio_reg & HPD_GPIO_MASK) == 0)
        return HPD_ISSUE_HPD_STUCK_LOW;
    if ((irq_stat & HPD_IRQ_PENDING) && hpd_state)
        return HPD_ISSUE_HPD_STUCK_HIGH;
    if (link->hpd_dedup->same_event_count > 100)
        return HPD_ISSUE_HPD_STORM;

    return HPD_ISSUE_NONE;
}

// 问题修复建议
const char *hpd_issue_suggestion(enum hpd_common_issue issue)
{
    switch (issue) {
    case HPD_ISSUE_HPD_STUCK_LOW:
        return "Check cable connection. If issue persists, "
               "try: echo 1 > /sys/kernel/debug/dri/0/amdgpu_hpd/reset";
    case HPD_ISSUE_HPD_STUCK_HIGH:
        return "HPD signal stuck high. Check for short circuit. "
               "Try re-plugging the display.";
    case HPD_ISSUE_HPD_GLITCH:
        return "HPD signal unstable. Check cable shielding. "
               "Increase debounce time in driver parameters.";
    case HPD_ISSUE_HPD_STORM:
        return "HPD storm detected. System may become unresponsive. "
               "Check: dmesg | grep 'hpd.*storm'. "
               "Disable and re-enable the connector.";
    case HPD_ISSUE_DEBOUNCE_FAIL:
        return "Debounce failed. Signal too unstable. "
               "Try different cable or port.";
    case HPD_ISSUE_DPCD_READ_FAIL:
        return "DPCD communication failed. "
               "Check if display is powered on. "
               "Try: cat /sys/kernel/debug/dri/0/amdgpu_dpcd";
    default:
        return "No specific issue detected.";
    }
}
```

## 11. 总结

本章深入介绍了 AMDGPU DC 子系统的热插拔检测 (HPD) 机制：

1. **HPD 硬件原理**：Long HPD 表示物理连接/断开，通过电平变化检测；Short HPD 表示 DP 链路状态变化，通过脉冲序列触发。

2. **核心数据结构**：`union hpd_irq_data` 映射 DPCD 200h-205h，包含 sink count、设备服务 IRQ 向量、Lane 状态等信息。

3. **Long HPD 处理**：从硬件中断到去抖、状态更新、链路检测、资源分配，最后通过 DRM uevent 通知用户空间。

4. **Short HPD 处理**：读取 DPCD 分析具体原因，处理链路状态变化、MST 消息、自动化测试请求等子事件。

5. **DPIA HPD**：USB4 端口的 HPD 通过 DMCUB 通信实现，使用 `DMUB_CMD_QUERY_HPD_STATE` 命令查询状态。

6. **异步工作队列**：使用专用的 HPD workqueue 处理检测任务，实现去重机制防止事件风暴。

7. **DM IRQ 框架**：高低优先级处理器链表，支持灵活的中断分发和注册/注销。

8. **MST HPD**：处理上行请求和下行回复消息，涉及复杂的侧边消息协议。

作为测试开发工程师，HPD 是验证显示子系统稳定性的关键测试点。通过本章提供的调试接口和自动化测试工具，可以系统性地测试 HPD 检测延迟、去重效果、风暴处理能力等关键指标。
