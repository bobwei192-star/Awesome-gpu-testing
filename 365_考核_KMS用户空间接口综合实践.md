#### 一、选择题（每题 2 分，共 20 分）

**1. 关于 DRM Atomic API，以下哪个说法是正确的？**

A. `drmModeAtomicCommit` 的 `DRM_MODE_ATOMIC_TEST_ONLY` 标志会实际修改硬件状态
B. Atomic 提交不允许在单个请求中包含多个 CRTC 的配置
C. `drmModeAtomicAddProperty` 在提交失败后需要重新创建请求对象
D. `DRM_MODE_ATOMIC_NONBLOCK` 标志要求提交必须同步等待完成

**2. 关于 DRM connector 热插拔检测，以下哪个描述是正确的？**

A. `drmModeGetConnector` 返回 `DRM_MODE_CONNECTED` 时一定包含有效 EDID
B. 用户空间可以通过轮询 `drmModeGetConnector` 来检测连接状态变化
C. DP MST 拓扑变化不会产生 `DRM_EVENT_CONNECT_CHANGE` 事件
D. Short HPD 只用于热插拔检测，不会触发链路训训练

**3. 关于 dumb buffer，以下哪个说法是错误的？**

A. dumb buffer 始终是连续物理内存
B. `DRM_IOCTL_MODE_MAP_DUMB` 用于获取 dumb buffer 的偏移量
C. dumb buffer 可以通过 `mmap` 映射到用户空间直接访问
D. dumb buffer 适合用于测试场景和小分辨率 framebuffer

**4. 关于 DRM plane 的 `IN_FORMATS` 属性，以下哪个描述是正确的？**

A. `IN_FOMRATS` 是一个 Range 类型的属性
B. `IN_FORMATS` 包含该 plane 支持的所有 pixel format 和 modifier 组合
C. `IN_FORMATS` 仅对 overlay plane 有效
D. `IN_FORMATS` 的值可以通过 `drmModeSetPlane` 直接修改

**5. 关于 DRM mode setting 的时序参数，以下哪个说法是正确的？**

A. `htotal` 等于 `hdisplay` + `hback_porch` + `hfront_porch`
B. `clock` 字段的单位是 MHz
C. `vrefresh` 的计算公式是 `clock / (htotal * vtotal)`
D. `flags` 字段中 `DRM_MODE_FLAG_INTERLACE` 表示逐行扫描

**6. 关于 libdrm 的事件处理，以下哪个描述是正确的？**

A. `drmHandleEvent` 只能处理 page flip 事件
B. page flip 事件处理函数中可以直接调用 `drmModeAtomicCommit`
C. `drmWaitVBlank` 的 `request.type` 可以为 `DRM_VBLANK_RELATIVE` 表示绝对帧号
D. page flip 事件通过 `DRM_IOCTL_MODE_PAGE_FLIP` 请求产生

**7. 关于 DRM 的 CRTC 概念，以下哪个描述是正确的？**

A. 一个 CRTC 可以同时驱动多个 encoder
B. CRTC 负责将 framebuffer 数据转换为显示时序信号
C. CRTC 的 `mode_valid` 回调在 Atomic check 阶段不会被调用
D. 每个 CRTC 都包含一个独立的 hardware cursor

**8. 关于 DRM leasing，以下哪个说法是正确的？**

A. lease 可以授予任意的 DRM 文件描述符
B. lease 持有者可以访问所有 CRTC 资源
C. lease 的创建需要调用者拥有 DRM master 权限
D. lease 一旦创建就无法撤销，必须等待持有者关闭 FD

**9. 关于 GEM buffer 和 dumb buffer 的区别，以下哪个描述是正确的？**

A. dumb buffer 是 GEM buffer 的一种特例
B. GEM buffer 始终比 dumb buffer 性能更好
C. dumb buffer 不支持 `mmap`，只能通过 `drmModeMapDumbBuffer` 访问
D. GEM buffer 只能通过 `drmIoctl` 创建，不支持 `drmModeAddFB`

**10. 关于 modetest 工具，以下哪个说法是正确的？**

A. modetest 的 `-a` 参数使用 Legacy API 进行模式设置
B. modetest 的 `-v` 参数指定页面翻转的帧数
C. modetest 的 `-s` 参数格式为 `<connector_id>@<crtc_id>:<width>x<height>`
D. modetest 不支持 Atomic API 的 TEST_ONLY 验证

#### 16. Bug 定位与修复（15 分）

以下是一段存在多个问题的 KMS 测试程序代码。请找出至少 5 个错误，说明每个错误的行号和原因，并给出修复建议。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <fcntl.h>
#include <unistd.h>
#include <errno.h>
#include <xf86drm.h>
#include <xf86drmMode.h>

#include <drm_fourcc.h>

int main(int argc, char *argv[])
{
    const char *device = "/dev/dri/card0";
    int fd = open(device, O_RDWR);
    if (fd < 0) {
        perror("open");
        return -1;
    }

    // 获取资源
    drmModeRes *res = drmModeGetResources(fd);
    if (!res) {
        fprintf(stderr, "GetResources failed\n");
        goto out;
    }

    // 找到第一个连接的 connector
    drmModeConnector *conn = NULL;
    int conn_id = 0;
    for (int i = 0; i < res->count_connectors; i++) {
        conn = drmModeGetConnector(fd, res->connectors[i]);
        if (conn && conn->connection == DRM_MODE_CONNECTED) {
            conn_id = res->connectors[i];
            break;
        }
    }

    if (!conn) {
        fprintf(stderr, "No connected connector\n");
        goto out_res;
    }

    // 使用第一个模式
    drmModeModeInfo *mode = &conn->modes[0];
    printf("Using mode: %dx%d@%dHz\n",
           mode->hdisplay, mode->vdisplay, mode->vrefresh);

    // 创建 dumb buffer
    struct drm_mode_create_dumb create = {
        .width = mode->hdisplay,
        .height = mode->vdisplay,
        .bpp = 32
    };

    int ret = ioctl(fd, DRM_IOCTL_MODE_CREATE_DUMB, &create);
    if (ret) {
        perror("CREATE_DUMB");
        goto out_conn;
    }

    // 添加 framebuffer
    uint32_t handles[4] = {create.handle};
    uint32_t pitches[4] = {create.pitch};
    uint32_t offsets[4] = {0};
    uint32_t fb_id;

    ret = drmModeAddFB2(fd, mode->hdisplay, mode->vdisplay,
                         DRM_FORMAT_XRGB8888,
                         handles, pitches, offsets,
                         &fb_id, 0);
    if (ret) {
        perror("AddFB2");
        goto out_conn;
    }

    // 设置模式（Legacy API）
    ret = drmModeSetCrtc(fd, res->crtcs[0], fb_id, 0, 0,
                          &conn_id, 1, mode);
    if (ret) {
        perror("SetCrtc");
        goto out_fb;
    }

    printf("Mode set successfully\n");
    sleep(5);

    // page flip 测试
    struct drm_mode_crtc_page_flip flip = {
        .crtc_id = res->crtcs[0],
        .fb_id = fb_id,
        .flags = DRM_MODE_PAGE_FLIP_EVENT,
        .user_data = 0
    };

    ret = ioctl(fd, DRM_IOCTL_MODE_PAGE_FLIP, &flip);
    if (ret) {
        perror("PAGE_FLIP");
        goto out_fb;
    }

    // 等待 page flip 事件
    fd_set fds;
    FD_ZERO(&fds);
    FD_SET(fd, &fds);
    struct timeval timeout = {5, 0};
    
    ret = select(fd + 1, &fds, NULL, NULL, &timeout);
    if (ret > 0) {
        drmEventContext ev = {
            .version = DRM_EVENT_CONTEXT_VERSION,
            .page_flip_handler = NULL
        };
        drmHandleEvent(fd, &ev);
    }

out_fb:
    drmModeRmFB(fd, fb_id);
out_conn:
    drmModeFreeConnector(conn);
out_res:
    drmModeFreeResources(res);
out:
    close(fd);
    return 0;
}
```

static int dm_plane_atomic_check(struct drm_plane *plane,
                                  struct drm_atomic_state *state)
{
    struct drm_plane_state *new_plane_state =
        drm_atomic_get_new_plane_state(state, plane);
    struct drm_crtc_state *new_crtc_state = NULL;
    struct dm_plane_state *new_dm_state = 
        to_dm_plane_state(new_plane_state);
    int ret;

    if (!new_plane_state->crtc)
        return 0;

    new_crtc_state = drm_atomic_get_new_crtc_state(
        state, new_plane_state->crtc);

    if (!new_crtc_state->enable)
        return 0;

    ret = dm_plane_helper_check_state(new_plane_state,
                                       new_crtc_state);
    if (ret)
        return ret;

    if (new_plane_state->fb) {
        if (new_plane_state->fb->modifier) {
            new_dm_state->tiling_flags =
                amdgpu_fb_get_tiling_flags(new_plane_state->fb);
        }

        new_dm_state->tmz_surface =
            amdgpu_fb_is_tmz_surface(new_plane_state->fb);
    }

    return 0;
}

static void dm_plane_atomic_update(struct drm_plane *plane,
                                    struct drm_atomic_state *state)
{
    struct drm_plane_state *old_state = 
        drm_atomic_get_old_plane_state(state, plane);
    struct drm_plane_state *new_state = 
        drm_atomic_get_new_plane_state(state, plane);
    struct dm_plane_state *new_dm_state = 
        to_dm_plane_state(new_state);

    if (!new_state->crtc || !new_state->fb)
        goto cleanup;

    if (!new_state->visible)
        goto cleanup;

    /* 提交更新到 DC */
    dc_update_planes(new_dm_state->dc, &new_dm_state->dc_state, 1);

    return;

cleanup:
    dc_remove_plane_from_context(new_dm_state->dc,
                                  new_dm_state->dc_state);
}

#### 第一部分参考答案

**选择题**：
1. C — AtomicAddProperty 在提交失败后请求对象不可复用
2. B — 用户空间可以轮询 `drmModeGetConnector` 检测状态变化
3. A — dumb buffer 不一定是连续物理内存，但满足连续虚拟地址
4. B — `IN_FORMATS` 包含 pixel format 和 modifier 的组合信息
5. C — vrefresh = clock / (htotal * vtotal)
6. D — page flip 事件通过 `DRM_IOCTL_MODE_PAGE_FLIP` 产生
7. B — CRTC 负责将 framebuffer 数据转换为显示时序
8. C — lease 创建需要调用者拥有 DRM master 权限
9. A — dumb buffer 是 GEM buffer 的特例
10. B — `-v` 指定 page flip 的帧数

**简答题要点**：
11. TEST_ONLY：仅验证配置有效性，不修改硬件状态；ALLOW_MODESET：允许修改显示时序。典型流程：AtomicAlloc → AddProperty（FB_ID/CRTC_ID/MODE_ID/ACTIVE）→ TEST_ONLY 验证 → AtomicCommit（ALLOW_MODESET）
12. FB_ID 指定 framebuffer，CRTC_ID 指定显示控制器，MODE_ID 指定时序（需创建 blob），ACTIVE 使能/禁用输出。依赖关系：FB_ID 依赖 CRTC_ID 和 MODE_ID，ACTIVE=1 时需要 MODE_ID
13. Legacy：drmModeSetCrtc（单次设置）+ drmModePageFlip（翻转）+ drmHandleEvent（事件处理）；Atomic：一个 drmModeAtomicCommit 完成全部配置
14. EDID 通过 blob 属性访问：drmModeGetPropertyBlob → 解析 EDID 结构体 → 获取 Detailed Timing Descriptor 中的分辨率/刷新率/色彩深度
15. drmHandleEvent 通过 event.type 分发：DRM_EVENT_FLIP_COMPLETE → page_flip_handler，DRM_EVENT_CONNECT_CHANGE → connector_change_handler。poll 保证有事件可读，避免阻塞在 read 系统调用
