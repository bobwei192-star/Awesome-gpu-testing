```c
static int mst_allocate_vc_payload(struct dc_link *link,
                                     struct mst_stream *stream)
{
    uint32_t requested_pbn;
    uint32_t allocated_pbn;
    uint32_t vcpi;
    int ret;

    // 步骤 A: 计算所需带宽
    requested_pbn = calc_pbn_for_mode(
        stream->mode.timing,
        stream->mode.bpp,
        stream->dsc_enabled);

    // 步骤 B: 查询可用带宽
    uint32_t avail_pbn = get_available_pbn(link);

    if (requested_pbn > avail_pbn) {
        DRM_ERROR("Insufficient bandwidth\n");
        return -ENOSPC;
    }

    // 步骤 C: 分配 VCPI
    ret = dpcd_set_vcpi(link, stream->stream_id,
                         requested_pbn, &vcpi);
    if (ret) {
        DRM_ERROR("VCPI allocation failed\n");
        return ret;
    }

    stream->vcpi = vcpi;
    stream->allocated_pbn = requested_pbn;

    // 步骤 D: 发送 ACT (Allocation Change Trigger)
    ret = dpcd_send_act(link);
    if (ret) {
        DRM_ERROR("ACT failed\n");
        dpcd_clear_vcpi(link, vcpi);
        return ret;
    }

    // 步骤 E: 验证分配
    uint32_t actual_pbn;
    dpcd_read_vcpi_table(link, vcpi, &actual_pbn);

    if (actual_pbn != requested_pbn) {
        DRM_WARN("VCPI mismatch: req=%u actual=%u\n",
                 requested_pbn, actual_pbn);
    }

    // 步骤 F: 配置 DIG 发送器
    configure_dig_for_mst(link, stream);

    return 0;
}
```

### 题目 2: DSC 配置协商 (8 分)

```c
static int dsc_negotiate_config(struct dc_sink *sink,
                                 struct dsc_config *config)
{
    uint8_t dsc_caps[16];
    int ret;

    // 1. 读取 sink DSC 能力
    ret = dpcd_read(sink->link,
                     DP_DSC_SUPPORT, dsc_caps, 16);
    if (ret)
        return ret;

    // 2. 检查 DSC 版本
    if ((dsc_caps[0] & 0xF0) < 0x10) {
        DRM_DEBUG("Sink DSC version < 1.0, disabling DSC\n");
        config->dsc_enable = false;
        return 0;
    }

    // 3. 解析能力
    struct dsc_sink_caps {
        uint8_t version;                  // [0]: DSC version
        uint8_t slice_caps1;              // [1]: Slice capabilities
        uint8_t line_buf_depth;           // [2]: Line buffer depth
        uint8_t block_pred;               // [3]: Block prediction
        uint8_t max_bpp;                  // [4]: Max bits per pixel * 16
        uint8_t color_format;             // [5]: Color format support
        uint8_t decoder_color_depth;      // [6]: Color depth support
    } *caps = (struct dsc_sink_caps *)dsc_caps;

    // 4. 配置切片参数
    if (caps->slice_caps1 & 0x01)
        config->slice_count = 1;
    if (caps->slice_caps1 & 0x02)
        config->slice_count = 2;
    if (caps->slice_caps1 & 0x04)
        config->slice_count = 4;

    config->slice_width =
        config->timing.h_addressable.h_total /
        config->slice_count;

    // 5. 配置压缩比
    uint8_t max_bpp_x16 = caps->max_bpp;
    config->bpp = min(config->target_bpp,
                      max_bpp_x16 / 16);

    // 6. 配置颜色深度
    config->bits_per_component = 8;
    if (caps->decoder_color_depth & 0x01)
        config->bits_per_component = 8;
    if (caps->decoder_color_depth & 0x02)
        config->bits_per_component = 10;
    if (caps->decoder_color_depth & 0x04)
        config->bits_per_component = 12;

    DRM_DEBUG("DSC config: slices=%u width=%u bpp=%u.%u "
              "bpc=%u\n",
              config->slice_count,
              config->slice_width,
              config->bpp / 16, config->bpp % 16,
              config->bits_per_component);

    return 0;
}
```

### 题目 3: HPD 中断处理 (8 分)

```c
// 热插拔中断处理 (简化版)
static void hpd_irq_handler(struct dc_link *link)
{
    uint8_t dpcd_val;
    bool link_status_changed = false;
    bool downstream_irq = false;

    // 1. 读取 DPCD 中断状态
    dpcd_read(link, DPCD_SINK_COUNT_ESI,
              &dpcd_val, 1);

    if (dpcd_val & DPCD_SINK_COUNT_ESI_HPD) {
        // Long HPD: 连接/断开
        link_status_changed = true;
    }

    if (dpcd_val & DPCD_LANE0_1_STATUS_ESI) {
        // Short HPD: 链路状态变更
        downstream_irq = true;
    }

    // 2. 处理 Long HPD
    if (link_status_changed) {
        if (dpcd_val & DPCD_SINK_COUNT_NEW) {
            // 新设备连接
            dc_link_detect(link, DETECT_REASON_HPD);
            schedule_mst_topology_probe(link);
        } else {
            // 设备断开
            dc_link_disconnect(link);
            invalidate_mst_topology(link);
        }
    }

    // 3. 处理 Short HPD (链路中断)
    if (downstream_irq) {
        uint8_t lane_status;
        dpcd_read(link, DPCD_LANE0_1_STATUS,
                  &lane_status, 1);

        if (lane_status & DPCD_LANE_STATUS_CR) {
            // 时钟恢复失败 → 链路重训练
            dp_retrain_link(link);
        }

        if (lane_status & DPCD_LANE_STATUS_CE) {
            // 符号锁定失败 → 通道均衡
            dp_adjust_drive_settings(link);
        }

        // 检查 MST 下行设备中断
        handle_mst_downstream_irq(link);
    }
}
```

```c
static int audio_stream_enable(struct audio_stream *stream,
                                struct audio_format *fmt)
{
    struct dc_link *link = stream->link;
    int ret;

    // 1. 格式协商
    ret = negotiate_hw_format(stream, fmt);
    if (ret)
        return ret;

    // 2. 分配 DMA 缓冲区
    ret = alloc_audio_dma_buf(stream);
    if (ret)
        return ret;

    // 3. 配置 HDA 流
    ret = hda_stream_set_format(stream, &stream->hw_format);
    if (ret)
        goto err_free_dma;

    // 4. 设置通道映射
    struct channel_map map = calc_channel_map(fmt->channels);
    ret = hda_stream_set_channel_map(stream, &map);
    if (ret)
        goto err_free_dma;

    // 5. 配置显示引擎
    if (link->connector_signal & SIGNAL_TYPE_HDMI) {
        // HDMI: Audio InfoFrame + N/CTS
        ret = hdmi_config_audio_infoframe(link, stream);
        if (ret)
            goto err_free_dma;

        ret = hdmi_set_n_cts(link, stream);
        if (ret)
            goto err_free_dma;
    } else {
        // DP: Audio SDP
        ret = dp_config_audio_sdp(link, stream);
        if (ret)
            goto err_free_dma;
    }

    // 6. 启动 HDA 流
    ret = hda_stream_start(stream);
    if (ret)
        goto err_free_dma;

    stream->state = AUDIO_STREAM_RUNNING;
    return 0;

err_free_dma:
    free_audio_dma_buf(stream);
    return ret;
}
```

## 附: 参考答案与解析

### 第一部分答案

**选择题:**
1. B — VCPI 用于标识 MST 流的虚拟通道带宽分配。
2. C — Repeater 是 MST 标准中不存在的类型；MST 节点为 Source / Branch / Sink。
3. B — DSC 基于 DPCM (差分脉冲编码调制) + 索引颜色历史 (Indexed Color History)。
4. B — FHD (1920x1080) 典型 slice width 为 1080/2 = 540 pixels (2 slices)。
5. C — Short HPD (IRQ_HPD) 通常由 MST 拓扑变化或链路状态变化触发。
6. C — 入口为 `dm_irq_handler()` → 调用 `dc_link_handle_hpd_irq()`。
7. B — HDMI 使用 N/CTS (N 值 + Cycle Time Stamp) 从 TMDS 时钟恢复音频时钟。
8. B — DP 使用 Audio SDP (Secondary Data Packet) 封装音频数据。
9. A — DPCD 0x00100~0x001FF 是 MST 全局控制寄存器区域，包括 MSTM_CTRL。
10. A — DSC 切片完全独立解码 (独立刷新)，便于并行处理。

**填空题:**
11. 树形 (层级) / 6
12. 计算 PBN → 分配 VCPI → 发送 ACT
13. 预测器 (Predictor) / 量化器 (Quantizer) / 熵编码器 (Entropy Coder)
14. target_bpp (bits per pixel) / 6~48 bpp (每像素 6~48 比特)
15. High (HPD high) / 100 μs (100μs = long HPD, <100μs = short HPD)
16. 轮询 (Polling) / 检测已连接的设备
17. 0x84, 0x01, 0x0A
18. BDL (Buffer Descriptor List)
19. slice_width (切片宽度) / slot_count (时隙数)
20. DCN (Display Core Next) → OPTC (Timing Controller) → DSC (编码) → DIO (输出) → AFMT (音频)

### 第二部分参考解析

**题目 1:**
1. 返回 `-ENOSPC`。解决方法: 启用 DSC 降低带宽需求、降低显示分辨率/刷新率、减少 MST 流数量、使用更高带宽的链路 (如 HBR3)。
2. `dpcd_clear_vcpi()` 清理已分配的 VCPI 条目，释放带宽给其他流使用。回滚是必须的，否则可能导致带宽泄漏或其他流分配失败。
3. VCPI 验证失败不返回错误是因为某些显示器在 VCPI 表中报告的值可能有细微舍入误差。使用 WARN 而不是 ERROR，配合动态调整机制 (`adjust_pbn()`) 可以增加兼容性。
4. `configure_dig_for_mst()` 需要配置: DIG 的 slot 分配 (start slot + slot count)、VC Payload ID 映射、MST 模式使能、时序参数与主链路参数的关联。

**题目 2:**
1. 代码使用 `if` 而不是 `else if`，导致前面的赋值被后面的覆盖。最终 `caps->slice_caps1 & 0x04` 总是生效，所以 slice_count 总是 4。应改为 `if-else if` 链或使用 `switch`。
2. `max_bpp_x16 / 16` 使用整数除法导致截断。应使用: `config->bpp = min(config->target_bpp * 16, max_bpp_x16)` 并以 1/16 bpp 精度存储。
3. Bug: 第 52 行无条件将 `bits_per_component = 8`，后面即使检测到 10/12bit 支持也被覆盖。修复: 移除第 52 行的默认赋值，或将其设为 `else` 分支。
4. 恢复流程:
   ```
   DSC 编码失败
       ↓
   停止当前显示流
       ↓
   关闭 DSC (bpp 回退到原生)
       ↓  
   重新配置时序 (无压缩)
       ↓
   验证链路带宽是否足够
     ├─ 足够 → 直接输出未压缩流
     └─ 不足 → 降低分辨率/刷新率
   ```

**题目 3:**
1. 通过 DPCD 0x00200 (SINK_COUNT_ESI) 的 `HPD` 位区分: bit 0 变化 + bit 1 (SINK_COUNT_NEW) = Long HPD; bit 2 (LANE0_1_STATUS_ESI) = Short HPD。脉冲宽度不可靠因为 MST 分支器可能转发/转换 HPD 信号。
2. `DETECT_REASON_HPD` 触发完整的重新探测: 读取 EDID、重新协商链路参数、重建 MST 拓扑。如果已知是同一设备 (Short HPD)，可以使用 `DETECT_REASON_IRQ` 跳过 EDID 重读。
3. 检查: Lane status (0x00202~0x00203) 全部为 0 (CR/CE/ES 完成)、Interlane align done、Link training success (0x00102 bit 0 = 1)。
4. 清理操作: 释放该分支器下所有 sink 的 VCPI、清除 DPCD down-stream port 状态、更新 MST topology 缓存、通知 DRM 框架移除 connector、恢复可用带宽计数。

**题目 4:**
1. 顺序不可调换。HDA 流格式配置是基础 (决定 DMA 缓冲区大小和 BDL 设置)，显示引擎配置依赖最终的 `hw_format`。如果先配显示引擎再配 HDA，可能出现格式不匹配或异步错误。
2. 应加入更多清理: `hda_stream_reset(stream)` 重置 HDA 控制器状态、清除已配置的 InfoFrame/SDP (`clear_display_audio`)、恢复 AFMT 模块默认状态。
3. 恢复流程: `hda_stream_stop()` 停止 DMA → `free_audio_dma_buf()` 释放缓冲区 → `hda_stream_reset()` 重置控制器 → 清除 InfoFrame/SDP → 标记流状态为 IDLE → 通知 ALSA 上层音频设备移除。

### 第三部分参考方案

**场景 1:**
1. 带宽计算:
   - 4K@60Hz: 3840×2160×60×24×1.2 ≈ 14.93 Gbps
   - 2K@60Hz: 2560×1440×60×24×1.2 ≈ 6.37 Gbps
   - 总计: ~21.3 Gbps → HBR3 32.4 Gbps (有效约 25.92 Gbps) → 理论上够用，但加上 overhead 可能不足
   - 注意: 实际 PBN 计算还需考虑 MTPH、HBlank 等参数

2. 解决方案:
   - 启用 DSC 压缩降低带宽需求
   - 降低其中一个显示器的刷新率 (如 4K@30Hz + 2K@60Hz)
   - 调整 timing 减少 blanking overhead
   - 使用独立的 MST 链路 (如第二个 DP 口) 而非菊花链

3. 启用 DSC 后: 4K@60Hz 14.93/2.25 ≈ 6.64 Gbps，加 2K 6.37 Gbps = 13.01 Gbps，远低于 25.92 Gbps 有效带宽，可行。

**场景 2:**
1. 原因: 切片边界不匹配，下半部分花屏说明第 2/3/4 个切片的解码参考数据出错。可能因为 slice_width 不满足 DSC 规范对齐要求，导致硬件解码器在切片边界处状态不同步。
2. DSC 规范要求 slice_width 必须是切片宽度粒度 (Slice Width Granularity) 的整数倍。DSC 1.2 对于 10bpc 要求 slice_width 是 2 pixels 的整数倍。5120/4=1280，1280 是 2 的倍数，满足基本要求。但用户空间的验证可能还需要检查 slice_height 是否正确: 2880/4=720 行。
3. 修复方案: 检查 slice_height 是否符合 `pic_height / slice_count` 的整除要求。如果 2880 不能被 4 整除，需要调整 slice_count 为 2 或 8。验证: 使用 `modetest` 和 CRC 校验对比压缩前后的无损性。

**场景 3:**
1. ELD 分析: 显示器支持 PCM 编码、16-bit 位深、32/44.1/48kHz 采样率、2 声道。驱动正确识别了基本能力，但 ELD 显示无 24bit 和多声道支持。这不是异常，DELL U2723QE 确实只支持 2ch 16bit。
2. 排查步骤:
   - 检查 ALSA 软件混音器: `alsamixer` 确认 HDMI 输出未静音、音量正常
   - 检查 PulseAudio: `pactl list sinks` 确认 HDMI sink 存在且未设为静音
   - 检查音频路由: `pavucontrol` 确认播放应用程序路由到正确的 HDMI 输出
   - 检查 HDMI 音频驱动: `dmesg | grep -i audio` 确认 AFMT 模块已配置
   - 强制测试: `speaker-test -D hw:0,3 -c 2` (跳过 PulseAudio 直接测试)
   - 检查 ELD 有效性: `cat /proc/asound/card0/eld#0` 确认 ELD 有效

**场景 4:**
1. 可能原因:
   - 信号完整性: DP 线缆屏蔽不良或接触不良
   - 电源干扰: 显示器的电源模块产生噪声耦合到 HPD 信号线
   - 显示器固件 Bug: 特定型号的 HPD 电平不稳定
   - DP 接收器故障: 显示器的 DP 端口 HPD 驱动电路问题
   - 静电放电 (ESD): 环境静电影响 HPD 电平

2. 去抖动方案:
   ```
   状态机:
   [STABLE] ──HPD low > 2ms──→ [DEBOUNCE]
              ──HPD low > 100ms─→ [DISCONNECTED]
   
   [DEBOUNCE] ──HPD high──→ [STABLE] (去抖动成功)
               ──定时器超时 50ms──→ [DISCONNECTED]
   
   [DISCONNECTED] ──HPD high > 100ms──→ [STABLE] (重新连接)
   
   实现: 
   - HPD 中断触发后启动 50ms 去抖动定时器
   - 定时器到期前忽略后续 HPD 中断
   - 50ms 后重新采样 HPD 电平确认状态
   - 连续 3 次采样一致才触发状态变更
   ```

### 第四部分参考要点

**论述题 1:**
1. 数据流图:
   ```
   应用 (DRM/ALSA)
       ↓
   ┌───────────── DC 调度层 ─────────────┐
   │  Context (MST流) → 管道 (Pipe)       │
   │       ↓                    ↓          │
   │   ┌─────────┐       ┌──────────┐      │
   │   │ DSC Enc │       │ Audio AFMT│     │
   │   └────┬────┘       └────┬─────┘      │
   │        ↓                 ↓            │
   │   ┌─────────┐       ┌──────────┐      │
   │   │ DIO/DIG │←──────│ Slot Mux │      │
   │   └────┬────┘       └──────────┘      │
   │        ↓                              │
   │   DP MST Encoder (VC Payload Mux)     │
   └────────┼──────────────────────────────┘
            ↓
       物理层 (DP TX)
   ```

2. DSC-MST 耦合:
   - DSC 的 slice_width 必须与 MST 的 slot 分配协调
   - DSC 压缩后的带宽直接影响每个 VC Payload 的 PBN 需求
   - MST 拓扑变化时需要重新协商 DSC 参数 (如增加显示器时可能需提高压缩比)
   - DSC 解码器延迟必须纳入 MST 时序预算

3. DP 2.0 UHBR 支持需修改:
   - 链路层: DPCD 地址映射扩展 (UHBR 寄存器区域)
   - 物理层: UHBR 训练序列和电气参数
   - DSC: 支持更高压缩比 (bpp 范围扩展)
   - Audio: 时钟恢复算法适配更高链路速率
   - MST: 更大的 PBN 池管理

**论述题 2:**
1. 测试用例:
   - MST: (1) 4 台显示器菊花链 (2) 2 层分支器拓扑 (3) 热插拔时动态带宽重分配
   - DSC: (1) 所有 slice count 配置 (2) 压缩比扫描 6~48bpp (3) 压缩/解压缩 CRC 一致性
   - HPD: (1) 连续 1000 次 HPD 脉冲稳定性 (2) 去抖动时间参数验证 (3) MST HPD 转发延迟
   - Audio: (1) 全采样率扫描 (2) 多声道映射验证 (3) 热插拔音频恢复

2. 组合测试矩阵 (8 种):
   ```
   | MST | DSC | Audio | 场景                        |
   |-----|-----|-------|-----------------------------|
   |  0  |  0  |   0   | 单显示器, 无压缩, 无音频     |
   |  0  |  0  |   1   | 单显示器, 无压缩, PCM 音频   |
   |  0  |  1  |   0   | 单显示器, DSC 压缩, 无音频   |
   |  0  |  1  |   1   | 单显示器, DSC 压缩, 音频     |
   |  1  |  0  |   0   | MST 多流, 无压缩, 无音频     |
   |  1  |  0  |   1   | MST 多流, 无压缩, 多路音频   |
   |  1  |  1  |   0   | MST 多流, DSC 压缩, 无音频   |
   |  1  |  1  |   1   | MST 多流, DSC 压缩, 多路音频 |
   ```

3. 自动化测试难点与应对:
   - 硬件依赖: 需要多种显示器和分支器 → 构建硬件矩阵库 + 矩阵切换器
   - 时序不确定性: MST/DSC 时序敏感 → 使用 CRT (Clock Recovery Timing) 分析仪
   - 音频验证: 无法自动检测声音 → 使用 HDMI Audio Analyzer + 频谱分析
   - 问题复现: 特定环境条件 → CI/CD 中建立标准化环境配置管理
   - 日志关联: 大量并行测试 → 统一日志标识符 (connection ID + test case ID)
