# Day 403: GPU驱动测试特殊性——硬件依赖 / 并发 / 非确定性

## 1. 概述

GPU驱动测试与普通软件测试有本质区别。GPU驱动需要直接操作硬件，面临并发执行模型、非确定性行为等多重挑战。本章深入分析GPU驱动测试的三个核心特殊性，帮助测试工程师理解并应对这些挑战。

### 1.1 为什么GPU驱动测试特殊？

```
普通软件测试                     GPU驱动测试
──────────────────────          ──────────────────────
纯逻辑验证                      硬件交互验证
确定性行为                      非确定性行为
单线程/有限并发                  大规模并行(数千shader核心)
标准输入输出                     专用硬件接口(MMIO/PCIe)
用户模式运行                     内核模式运行
简单的状态管理                   复杂的电源状态管理
标准调试工具(GDB)                有限调试手段(kgdb/示波器)
```

### 1.2 三种特殊性的相互关系

```
                           ┌──────────────────┐
```

### 2.4 真实案例：硬件差异导致的测试失败

**案例背景**：一个KUnit测试在GFX9 (Vega)上通过，但在GFX10 (RDNA 1)上失败。

```c
/*
 * 原始测试代码
 */
static void test_amdgpu_bo_compute_alignment(struct kunit *test)
{
    struct amdgpu_bo bo = {0};
    
    /* 错误假设：所有GPU使用相同的对齐要求 */
    bo.tiling_flags = AMDGPU_TILING_SET(VALUE, 4);
    
    /* 这个计算在GFX9和GFX10上结果不同！ */
    uint32_t alignment = amdgpu_bo_compute_alignment(&bo);
    
    KUNIT_EXPECT_EQ(test, alignment, 256); /* GFX9上是256，但GFX10需要512 */
}
```

**修复方案**：

```c
/*
 * 修复后的测试 - 参数化不同硬件版本
 */
static void test_amdgpu_bo_compute_alignment_gfx9(struct kunit *test)
{
    struct amdgpu_bo bo = {0};
    bo.tiling_flags = AMDGPU_TILING_SET(VALUE, 4);
    
    /* 设置模拟的GFX IP版本 */
    struct amdgpu_device *adev = get_mock_adev();
    amdgpu_mock_set_ip_version(adev, IP_VERSION_GFX9);
    
    uint32_t alignment = amdgpu_bo_compute_alignment(&bo);
    KUNIT_EXPECT_EQ(test, alignment, 256);
}

static void test_amdgpu_bo_compute_alignment_gfx10(struct kunit *test)
{
    struct amdgpu_bo bo = {0};
    bo.tiling_flags = AMDGPU_TILING_SET(VALUE, 4);
    
    struct amdgpu_device *adev = get_mock_adev();
    amdgpu_mock_set_ip_version(adev, IP_VERSION_GFX10);
    
    uint32_t alignment = amdgpu_bo_compute_alignment(&bo);
    KUNIT_EXPECT_EQ(test, alignment, 512); /* GFX10的对齐要求不同 */
}
```

### 3.2 多进程并发测试

```c
/*
 * 多进程并发命令提交测试
 * 模拟多个进程同时使用GPU的场景
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>
#include <sys/ioctl.h>
#include <fcntl.h>
#include <errno.h>
#include <time.h>

#define NUM_CHILDREN 8
#define ITERATIONS_PER_CHILD 100

/*
 * 子进程执行的GPU操作
 */
static int child_gpu_work(int child_id, int iterations)
{
    char path[64];
    snprintf(path, sizeof(path), "/dev/dri/card%d", child_id % 4);
    
    int fd = open(path, O_RDWR);
    if (fd < 0) {
        perror("open");
        return -1;
    }
    
    for (int i = 0; i < iterations; i++) {
        /* 分配BO */
        struct drm_amdgpu_gem_create create = {
            .bo_size = 4096 * (1 + (child_id * i) % 64),
            .alignment = 4096,
            .domain_flags = AMDGPU_GEM_DOMAIN_VRAM,
        };
        int r = ioctl(fd, DRM_IOCTL_AMDGPU_GEM_CREATE, &create);
        if (r) {
            fprintf(stderr, "Child %d: BO create failed at iter %d: %d\n",
                    child_id, i, errno);
            close(fd);
            return -1;
        }
        
        /* 提交命令 */
        struct drm_amdgpu_cs_submit submit = {0};
        /* ... 填充命令提交参数 ... */
        r = ioctl(fd, DRM_IOCTL_AMDGPU_CS, &submit);
        if (r && errno != ETIMEDOUT) {
            fprintf(stderr, "Child %d: CS failed at iter %d: %d\n",
                    child_id, i, errno);
        }
    }
    
    close(fd);
    printf("Child %d: completed %d iterations\n", child_id, iterations);
    return 0;
}

/*
 * 并发测试主函数
 */
static int test_multi_process_concurrency(void)
{
    pid_t children[NUM_CHILDREN];
    int status;
    int failures = 0;
    
    printf("多进程并发测试: %d个进程, 每个%d次迭代\n",
           NUM_CHILDREN, ITERATIONS_PER_CHILD);
    
    for (int i = 0; i < NUM_CHILDREN; i++) {
        children[i] = fork();
        if (children[i] == 0) {
            /* 子进程 */
            exit(child_gpu_work(i, ITERATIONS_PER_CHILD));
        }
    }
    
    /* 父进程等待所有子进程 */
    for (int i = 0; i < NUM_CHILDREN; i++) {
        waitpid(children[i], &status, 0);
        if (WIFEXITED(status) && WEXITSTATUS(status) != 0) {
            failures++;
            printf("Child %d failed with status %d\n",
                   i, WEXITSTATUS(status));
        }
    }
    
    printf("测试结果: %d/%d 子进程成功\n",
           NUM_CHILDREN - failures, NUM_CHILDREN);
    
    return failures == 0 ? 0 : 1;
}
```

```c
 * 验证GFX/SDMA/VCN引擎同时工作的稳定性

/*
 * 引擎并发工作负载定义
 */
struct engine_workload concurrent_workloads[] = {
    {AMDGPU_HW_IP_GFX,  0, "GFX 3D渲染",       60},
    {AMDGPU_HW_IP_SDMA, 0, "SDMA DMA传输",      60},
    {AMDGPU_HW_IP_VCN,  0, "VCN 视频解码",      60},
    {AMDGPU_HW_IP_VCN,  1, "VCN 视频编码",      60},
    {AMDGPU_HW_IP_JPEG, 0, "JPEG 图像解码",     60},
};

/*
 * 多引擎并发测试框架
 */
static int test_concurrent_engines(void)
{
    pthread_t threads[5];
    struct thread_args {
        int engine_id;
        int fd;
        bool failed;
    } args[5];
    
    int fd = open("/dev/dri/card0", O_RDWR);
    
    printf("多引擎并发测试启动\n");
    printf("==================\n\n");
    
    /* 启动各引擎线程 */
    for (int i = 0; i < 5; i++) {
        args[i].engine_id = i;
        args[i].fd = fd;
        args[i].failed = false;
        
        struct engine_workload *wl = &concurrent_workloads[i];
        printf("启动 %s (%ds)\n", wl->name, wl->duration_seconds);
        
        pthread_create(&threads[i], NULL, engine_thread_func, &args[i]);
    }
    
    /* 等待所有线程完成 */
    for (int i = 0; i < 5; i++) {
        pthread_join(threads[i], NULL);
        if (args[i].failed) {
            printf("FAIL: %s 引擎测试失败\n",
                   concurrent_workloads[i].name);
        }
    }
    
    close(fd);
    
    /* 验证测试结果 */

    } else {
```

```c
/*
 * 非确定性测试的统计方法
 */

struct statistical_test_config {
    int min_runs;            /* 最少运行次数 */
    int max_runs;            /* 最多运行次数 */
    double target_confidence; /* 目标置信度 (0.95 = 95%) */
    double acceptable_failure_rate; /* 允许的失败率 */
    bool stop_on_first_failure;     /* 遇到失败立即停止 */
};

/*
 * 统计测试结果
 */
struct statistical_test_result {
    int total_runs;
    int passes;
    int failures;
    double pass_rate;
    double mean_execution_time_ms;
    double std_dev_execution_time_ms;
    bool passed;
    const char *conclusion;
};

/*
 * 执行统计测试
 */
static struct statistical_test_result run_statistical_test(
    int (*test_func)(void),
    struct statistical_test_config *config)
{
    struct statistical_test_result result = {0};
    double *exec_times = malloc(config->max_runs * sizeof(double));
    
    for (int i = 0; i < config->max_runs; i++) {
        struct timespec start, end;
        clock_gettime(CLOCK_MONOTONIC, &start);
        
        int r = test_func();
        
        clock_gettime(CLOCK_MONOTONIC, &end);
        exec_times[i] = (end.tv_sec - start.tv_sec) * 1000.0 +
                       (end.tv_nsec - start.tv_nsec) / 1000000.0;
        
        if (r == 0) {
            result.passes++;
        } else {
            result.failures++;
            if (config->stop_on_first_failure)
                break;
        }
        
        result.total_runs++;
        
        /* 检查是否已达到足够的统计置信度 */
        if (result.total_runs >= config->min_runs) {
            double pass_rate = (double)result.passes / result.total_runs;
            if (pass_rate >= (1.0 - config->acceptable_failure_rate))
                break;
        }
    }
    
    /* 计算统计指标 */
    result.pass_rate = (double)result.passes / result.total_runs;
    
    double sum = 0;
    for (int i = 0; i < result.total_runs; i++)
        sum += exec_times[i];
    result.mean_execution_time_ms = sum / result.total_runs;
    
    double sq_sum = 0;
    for (int i = 0; i < result.total_runs; i++) {
        double diff = exec_times[i] - result.mean_execution_time_ms;
        sq_sum += diff * diff;
    }
    result.std_dev_execution_time_ms = sqrt(sq_sum / result.total_runs);
    
    /* 判断是否通过 */
    result.passed = (result.pass_rate >= (1.0 - config->acceptable_failure_rate));
    
    if (result.passed) {
        result.conclusion = "PASS - 统计测试通过";
    } else {
        result.conclusion = "FAIL - 失败率超过阈值";
    }
    
    free(exec_times);
    return result;
}
```

#### 案例2: GPU Hang的非确定性复现

GPU Hang是典型的非确定性问题，相同的测试用例可能有时触发、有时不触发。

```
GPU Hang 特征分析:
  - 触发概率: 可能只有0.1% - 5%
  - 触发条件: 需要特定的时序重叠
  - 复现难度: 通常需要数小时到数天的连续测试
  - 定位难度: 无法在模拟环境中复现

应对策略:
  1. 压力放大 - 增加并发度、减少超时、加速时序
  2. 统计检测 - 长时间运行后分析dmesg中的错误模式
  3. 硬件计数器 - 使用GPU的性能计数器检测异常状态
  4. 自动复现 - 记录导致hang的操作序列并自动重放
```

### 4.5 非确定性测试框架设计

```python
#!/usr/bin/env python3
"""
GPU驱动非确定性测试框架
用于量化和管理测试中的非确定性行为
"""

import subprocess
import time
import json
import statistics
import sys
from dataclasses import dataclass
from typing import List, Callable
from enum import Enum

class TestResult(Enum):
    PASS = "PASS"
    FAIL = "FAIL"
    FLAKY = "FLAKY"  # 非确定性的

@dataclass
class FlakyTestConfig:
    """非确定性测试配置"""
    name: str
    min_runs: int = 30
    max_runs: int = 100
    acceptable_fail_rate: float = 0.01  # 1%失败率可接受
    confidence_level: float = 0.95      # 95%置信度
    timeout_seconds: int = 60
    cooldown_seconds: int = 1

@dataclass
class FlakyTestResult:
    """非确定性测试结果"""
    name: str
    total_runs: int
    passes: int
    failures: int
    fail_rate: float
    execution_times: List[float]
    mean_time: float
    std_time: float
    verdict: TestResult

class FlakyTestRunner:
    """非确定性测试运行器"""
    
    def __init__(self, config: FlakyTestConfig):
        self.config = config
        self.results = []
    
    def run_test(self, test_func: Callable) -> FlakyTestResult:
        """运行统计测试"""
        passes = 0
        failures = 0
        execution_times = []
        
        print(f"\n运行非确定性测试: {self.config.name}")
        print(f"  最小运行: {self.config.min_runs}, "
              f"最大运行: {self.config.max_runs}")
        
        for i in range(self.config.max_runs):
            start = time.time()
            
            try:
                r = test_func()
                elapsed = time.time() - start
                
                if r == 0:
                    passes += 1
                else:
                    failures += 1
                
                execution_times.append(elapsed)
                
            except subprocess.TimeoutExpired:
                failures += 1
                execution_times.append(self.config.timeout_seconds)
            
            # 进度显示
            if (i + 1) % 10 == 0:
                current_fail_rate = failures / (i + 1)
                print(f"  [{i+1}/{self.config.max_runs}] "
                      f"通过: {passes}, 失败: {failures}, "
                      f"失败率: {current_fail_rate:.4f}")
            
            # 检查是否可以提前结束
            if i + 1 >= self.config.min_runs:
                current_fail_rate = failures / (i + 1)
                if current_fail_rate <= self.config.acceptable_fail_rate:
                    break
            
            time.sleep(self.config.cooldown_seconds)
        
        total = passes + failures
        fail_rate = failures / total if total > 0 else 1.0
        
        # 判定结果
        if fail_rate == 0:
            verdict = TestResult.PASS
        elif fail_rate <= self.config.acceptable_fail_rate:
            verdict = TestResult.FLAKY
        else:
            verdict = TestResult.FAIL
        
        result = FlakyTestResult(
            name=self.config.name,
            total_runs=total,
            passes=passes,
            failures=failures,
            fail_rate=fail_rate,
            execution_times=execution_times,
            mean_time=statistics.mean(execution_times) if execution_times else 0,
            std_time=statistics.stdev(execution_times) if len(execution_times) > 1 else 0,
            verdict=verdict
        )
        
        self.results.append(result)
        return result
    
    def report(self):
        """生成测试报告"""
        report = {
            "test_suite": "GPU驱动非确定性测试",
            "timestamp": time.strftime("%Y-%m-%d %H:%M:%S"),
            "results": []
        }
        
        for r in self.results:
            result_entry = {
                "name": r.name,
                "total_runs": r.total_runs,
                "passes": r.passes,
                "failures": r.failures,
                "fail_rate": r.fail_rate,
                "mean_time_ms": round(r.mean_time * 1000, 2),
                "std_time_ms": round(r.std_time * 1000, 2),
                "verdict": r.verdict.value
            }
            report["results"].append(result_entry)
        
        with open(f"flaky_test_report_{int(time.time())}.json", "w") as f:
            json.dump(report, f, indent=2)
        
        print("\n" + "=" * 60)
        print("非确定性测试报告")
        print("=" * 60)
        for r in self.results:
            status = "✅" if r.verdict == TestResult.PASS else \
                     "⚠️" if r.verdict == TestResult.FLAKY else "❌"
            print(f"{status} {r.name}: {r.verdict.value} "
                  f"(运行{r.total_runs}次, "
                  f"失败率{r.fail_rate:.4f})")
        
        print(f"\n报告保存至: flaky_test_report_{int(time.time())}.json")

# 使用示例
if __name__ == "__main__":
    def example_flaky_test():
        """模拟一个非确定性测试"""
        import random
        # 90%概率通过
        return 0 if random.random() < 0.9 else 1
    
    config = FlakyTestConfig(
        name="GFXOFF唤醒延迟测试",
        min_runs=30,
        max_runs=100,
        acceptable_fail_rate=0.05
    )
    
    runner = FlakyTestRunner(config)
    result = runner.run_test(example_flaky_test)
    runner.report()
```

## 6. 综合案例：AMDGPU并发+非确定性Bug诊断

### 6.1 Bug场景描述

以下是一个真实的AMDGPU驱动Bug，同时涉及并发和非确定性的挑战：

```
Bug #118724: amdgpu: 多进程Vulkan应用导致随机GPU hang
链接: https://gitlab.freedesktop.org/drm/amd/-/issues/118724

现象:
  - 两个Vulkan应用同时运行时，随机出现GPU hang
  - hang概率约2-5%
  - 单一应用运行时不会出现
  - 特定场景: 一个应用做compute，另一个做graphics

复现条件:
  - RDNA 2 (Navi 21) GPU
  - Linux 5.15+ 内核
  - 两个Vulkan应用使用不同的queue family
  - 系统内存压力 > 80%

根因:
  - 两个进程的VMID分配冲突
  - amdgpu_vmid_grab() 函数在并发时没有正确处理vmid重用
  - 导致两个进程使用了相同的VMID
  - GPU执行时page table混乱导致hang
```

### 6.2 测试策略设计

```c
/*
 * Bug复现测试设计
 * 目标: 可靠复现并发VMID冲突
 */

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <pthread.h>
#include <sys/wait.h>
#include <sys/ioctl.h>
#include <fcntl.h>
#include <errno.h>
#include <time.h>

#define NUM_PROCESSES 2
#define CONCURRENT_SUBMISSIONS 1000
#define VMID_STRESS_ITERATIONS 50

/*
 * 测试1: 并发VMID占用
 * 模拟两个进程同时请求VMID的场景
 */
static int test_concurrent_vmid_grab(void)
{
    int fd = open("/dev/dri/card0", O_RDWR);
    
    /* 强制频繁VMID切换 */
    for (int i = 0; i < VMID_STRESS_ITERATIONS; i++) {
        /* 提交大量命令使驱动频繁分配VMID */
        for (int j = 0; j < CONCURRENT_SUBMISSIONS; j++) {
            struct drm_amdgpu_cs_submit submit = {0};
            
            /* 随机切换不同引擎以增加VMID使用 */
            int engine = rand() % 3;
            switch (engine) {
            case 0:
                submit.ip_type = AMDGPU_HW_IP_GFX;
                break;
            case 1:
                submit.ip_type = AMDGPU_HW_IP_SDMA;
                break;
            case 2:
                submit.ip_type = AMDGPU_HW_IP_COMPUTE;
                break;
            }
            
            int r = ioctl(fd, DRM_IOCTL_AMDGPU_CS, &submit);
            if (r && errno != ETIMEDOUT) {
                fprintf(stderr, "CS提交失败: %d (iter %d/%d)\n",
                        errno, i, j);
            }
        }
        
        /* 主动flush VMID缓存 */
        system("echo 1 > /sys/kernel/debug/dri/0/amdgpu_vm_flush");
    }
    
    close(fd);
    return 0;
}

/*
 * 测试2: 多进程VMID压力测试
 */
static int test_multi_process_vmid_stress(void)
{
    pid_t pids[NUM_PROCESSES];
    int status;
    int failures = 0;
    
    printf("多进程VMID压力测试: %d进程, %d提交/进程\n",
           NUM_PROCESSES, CONCURRENT_SUBMISSIONS);
    
    srand(time(NULL));
    
    for (int i = 0; i < NUM_PROCESSES; i++) {
        pids[i] = fork();
        if (pids[i] == 0) {
            /* 子进程: 执行VMID压力测试 */
            srand(time(NULL) ^ (i << 16));
            exit(test_concurrent_vmid_grab());
        }
    }
    
    /* 父进程等待 */
    for (int i = 0; i < NUM_PROCESSES; i++) {
        waitpid(pids[i], &status, 0);
        if (WIFEXITED(status) && WEXITSTATUS(status)) {
            failures++;
        }
    }
    
    return failures;
}

/*
 * 测试3: VMID分配正确性验证
 * 验证不同进程使用不同VMID
 */
static int test_vmid_allocation_correctness(void)
{
    int fd = open("/dev/dri/card0", O_RDWR);
    
    /* 从debugfs读取VMID分配信息 */
    FILE *fp = fopen("/sys/kernel/debug/dri/0/amdgpu_vm_info", "r");
    if (!fp) {
        printf("无法读取VMID信息\n");
        return -1;
    }
    
    char line[256];
    int vmid_assignments[16] = {0}; /* VMID 0-15 */
    int vmid_count = 0;
    
    while (fgets(line, sizeof(line), fp)) {
        if (strstr(line, "VMID")) {
            int vmid, pid;
            sscanf(line, "VMID %d: pid %d", &vmid, &pid);
            if (pid > 0) {
                vmid_assignments[vmid_count++] = vmid;
            }
        }
    }
    fclose(fp);
    
    /* 检查是否有重复VMID分配给不同进程 */
    for (int i = 0; i < vmid_count; i++) {
        for (int j = i + 1; j < vmid_count; j++) {
            if (vmid_assignments[i] == vmid_assignments[j]) {
                printf("FAIL: VMID %d 被多个进程使用!\n",
                       vmid_assignments[i]);
                close(fd);
                return 1;
            }
        }
    }
    
    printf("PASS: VMID分配正确, %d个不同VMID在使用\n", vmid_count);
    close(fd);
    return 0;
}

/*
 * 测试4: 统计测试 - 确定非确定性失败的频率
 */
static int test_vmid_race_statistical(void)
{
    int total_runs = 0;
    int failures = 0;
    
    for (int i = 0; i < 30; i++) {
        int r = test_multi_process_vmid_stress();
        total_runs++;
        if (r) failures++;
        
        printf("  Run %d/30: %s\n", i + 1, r ? "FAIL" : "PASS");
    }
    
    printf("\n统计结果: %d/%d 失败 (%.1f%%)\n",
           failures, total_runs,
           (float)failures / total_runs * 100);
    
    if (failures > 0) {
        printf("结论: Bug已复现 - VMID竞争条件存在\n");
    } else {
        printf("结论: 未检测到VMID竞争\n");
    }
    
    return failures > 0 ? 1 : 0;
}
```

### 6.3 修复验证

```c
/*
 * 修复后的VMID分配逻辑（简化）
 * 原来的bug: vmid重用前没有检查是否被其他进程使用
 */

/*
 * 修复前:
 */
static int amdgpu_vmid_grab_buggy(struct amdgpu_vmid *vmid,
                                  struct amdgpu_ring *ring)
{
    /* BUG: 没有检查vmid是否被其他进程使用 */
    if (vmid->owner != current->mm) {
        /* 直接分配，可能导致冲突 */
        vmid->owner = current->mm;
    }
    return 0;
}

/*
 * 修复后:
 */
static int amdgpu_vmid_grab_fixed(struct amdgpu_vmid *vmid,
                                  struct amdgpu_ring *ring)
{
    spin_lock(&vmid->lock);
    
    /* 检查vmid是否被其他进程使用 */
    if (vmid->owner != NULL && vmid->owner != current->mm) {
        /* 需要等待vmid释放 */
        int r = amdgpu_vmid_wait_free(vmid, ring);
        if (r) {
            spin_unlock(&vmid->lock);
            return r;
        }
    }
    
    /* 现在安全地分配vmid */
    vmid->owner = current->mm;
    vmid->last_used = jiffies;
    
    spin_unlock(&vmid->lock);
    return 0;
}

/*
 * 修复验证测试
 */
static int test_vmid_fix_verification(void)
{
    int total_passes = 0;
    int total_tests = 0;
    
    /* 运行所有VMID测试 */
    printf("VMID修复验证测试\n");
    printf("================\n\n");
    
    total_tests++;
    if (test_vmid_allocation_correctness() == 0) {
        printf("VMID分配正确性: PASS\n");
        total_passes++;
    }
    
    /* 并发测试运行多次 */
    int conc_passes = 0;
    for (int i = 0; i < 10; i++) {
        if (test_multi_process_vmid_stress() == 0) {
            conc_passes++;
        }
    }
    printf("并发VMID测试: %d/10 通过\n", conc_passes);
    
    total_tests++;
    if (conc_passes >= 9) {
        printf("  结论: PASS (通过率 >= 90%%)\n");
        total_passes++;
    }
    
    printf("\n总体结果: %d/%d 测试通过\n", total_passes, total_tests);
    return total_passes == total_tests ? 0 : 1;
}
```

## 7. 总结与实践指南

### 7.1 核心要点

1. **硬件依赖**是GPU驱动测试的根本挑战，需要分层测试策略
2. **并发问题**需要通过专门的并发测试工具(KCSAN/Lockdep)检测
3. **非确定性**要求统计测试方法而非简单的通过/失败判断
4. **模拟+真实硬件**的混合测试策略是最经济的方案
5. **调试手段**包括dmesg/Umr/kgdb等多种工具的组合使用

### 7.3 推荐实践流程

```
GPU驱动测试开发流程:

1. 需求分析
   ├── 确认硬件依赖类型和程度
   ├── 识别可能的并发风险点
   └── 评估非确定性行为的概率

2. 测试设计
   ├── 单元测试: KUnit + 硬件模拟
   ├── 集成测试: IGT + 真实硬件
   ├── 并发测试: KCSAN + Lockdep
   └── 统计测试: 多轮执行 + 统计分析

3. 测试执行
   ├── 无硬件CI: KUnit/QEMU
   ├── 硬件CI: IGT + 压力测试
   └── 长期测试: 24h+ 稳定性监控

4. 结果分析
   ├── 确定性失败: 定位并修复
   ├── 非确定性失败: 统计分析和风险评估
```
