### A.1 选择题详细解析

**单选题 1（答案 C）**：
SCLK 只控制 GPU 核心（计算单元、渲染后端等）的运行频率。MCLK 是独立的显存时钟域。A 选项的错误在于虽然 SCLK 控制 CU，但也控制其他核心单元。B 选项错误，SCLK 和 MCLK 可以独立控制。D 选项错误，SCLK 的最大值受多个因素限制，包括温度、功耗、VID 等。

**单选题 2（答案 B）**：
当 SMU 检测到功耗超过 PPT 限制时，首先触发降频（Power Throttling）——降低 SCLK 频率以减少动态功耗。这是最快速有效的响应。如果降频后功耗仍超过限制，才会进一步降低电压或触发更严格的保护机制。

**单选题 3（答案 C）**：
V/F 曲线是非线性的。频率与电压的关系遵循半导体物理特性：随着频率增加，所需电压的增长速率也会增加（超线性关系）。C 选项说"频率加倍时电压也需要加倍"是完全错误的。

**单选题 4（答案 C）**：
sysfs hwmon 接口中的功耗值单位是微瓦（µW），因此 250W = 250,000,000 µW。power1_cap 是设置功耗限制的正确属性名。

**单选题 5（答案 B）**：
`dmesg | grep -i amdgpu` 是最有效的内核日志查看方法。A 选项可能没有实时日志。C 选项的 /proc/gpu_freq_debug 不存在。D 选项虽然有 debugfs 接口，但不是专门用于日志查看的。

**多选题 6（答案 ABC）**：
温度、PPT 限制、硅片体质都会影响 GPU 能达到的最高稳定频率。操作系统类型通常不影响。显存容量与核心频率无直接关系（影响的是 MCLK）。

**多选题 7（答案 ABCD）**：
rocm-smi 的 all 选项可以显示全部信息，包括电压、温度、功耗等。rocm-smi 不需要 root 权限即可运行（但某些操作如设置功耗限制需要）。

```c
// drivers/gpu/drm/amd/amdgpu/amdgpu_pm.c
// sysfs 属性定义
static DEVICE_ATTR(pp_od_clk_voltage, S_IRUGO | S_IWUSR,
                   amdgpu_get_pp_od_clk_voltage,
                   amdgpu_set_pp_od_clk_voltage);

static ssize_t amdgpu_set_pp_od_clk_voltage(struct device *dev,
    struct device_attribute *attr, const char *buf, size_t count)
{
    struct drm_device *ddev = dev_get_drvdata(dev);
    struct amdgpu_device *adev = drm_to_adev(ddev);
    int ret;

    // 参数验证
    if (!adev->pm.dpm_enabled || !adev->pm.funcs->set_pp_od_clk_voltage)
        return -EOPNOTSUPP;

    // 锁定互斥防止并发
    mutex_lock(&adev->pm.mutex);
    ret = amdgpu_dpm_set_pp_od_clk_voltage(adev, buf, count);
    mutex_unlock(&adev->pm.mutex);

    return ret < 0 ? ret : count;
}
```

### B.2 SMU 消息处理流程

```c
// drivers/gpu/drm/amd/pm/swsmu/smu_cmn.c
int smu_cmn_send_smc_msg_with_param(struct smu_context *smu,
                                    uint32_t msg, uint32_t param)
{
    struct amdgpu_device *adev = smu->adev;
    uint32_t reg_val;
    int ret;

    // 步骤1：等待 SMU 空闲
    ret = smu_cmn_wait_for_response(smu);
    if (ret)
        return ret;

    // 步骤2：写入消息参数到寄存器
    WREG32(adev->smu.mailbox_param_offset, param);

    // 步骤3：写入消息ID到触发寄存器
    WREG32(adev->smu.mailbox_offset, msg);

    // 步骤4：发送门铃信号触发 SMU 处理
    WREG32(adev->smu.mailbox_trigger_offset, 1);

    // 步骤5：等待 SMU 完成处理
    ret = smu_cmn_wait_for_response(smu);
    if (ret) {
        dev_err(adev->dev, "SMU message %u failed with %d\n", msg, ret);
        return ret;
    }

    // 步骤6：读取返回状态
    reg_val = RREG32(adev->smu.mailbox_return_offset);
    if (reg_val) {
        dev_dbg(adev->dev, "SMU message %u returned %u\n", msg, reg_val);
    }

    return 0;
}
```

**SMU 消息寄存器映射（RDNA 3）**：

```
0x03A00000 + 0x10C | MP0_SMN_C2PMSG_82 | 参数邮箱（写入参数）
0x03A00000 + 0x104 | MP0_SMN_C2PMSG_80 | 门铃寄存器（写入1触发）
0x03A00000 + 0x108 | MP0_SMN_C2PMSG_81 | 返回邮箱（读取返回码）
```

### B.4 并发保护机制分析

```c
// drivers/gpu/drm/amd/pm/amdgpu_dpm.c

// 全局 DPM 互斥锁
// 保护所有 DPM 相关操作，防止并发访问导致 SMU 状态不一致
static DEFINE_MUTEX(amdgpu_dpm_mutex);

int amdgpu_dpm_set_pp_od_clk_voltage(struct amdgpu_device *adev,
                                      const char *buf, size_t count)
{
    int ret;

    // adev->pm.mutex 在调用层已经被持有
    // 这里确保不会出现递归锁

    if (!adev->pm.funcs->set_pp_od_clk_voltage)
        return -EOPNOTSUPP;

    // 调用 SMU 层的具体实现
    ret = adev->pm.funcs->set_pp_od_clk_voltage(adev, buf, count);

    return ret;
}
```

**竞争条件分析**：
1. **多进程并发写入**：`adev->pm.mutex` 确保一次只有一个进程执行频率/电压调整
2. **中断上下文**：SMU 响应中断在 threaded IRQ 中处理，不会与 sysfs 操作的进程上下文冲突
3. **硬件级保护**：SMU 固件内部也有队列机制，确保一次只处理一个消息

```python
        with open(power_path) as f:
            return int(f.read().strip()) / 1_000_000

    def read_temp_c(self) -> float:
        temp_path = os.path.join(self.hwmon_path, "temp1_input")
        with open(temp_path) as f:
            return int(f.read().strip()) / 1000

    def run_stability_test(self, duration_s: int = 30) -> Tuple[float, float, float, float]:
        """运行稳定性测试，返回 (avg_power, max_power, avg_temp, max_temp)"""
        powers, temps = [], []

        proc = subprocess.Popen(
            self.benchmark_cmd.split(),
            stdout=subprocess.DEVNULL,
            stderr=subprocess.DEVNULL
        )

        for _ in range(duration_s):
            powers.append(self.read_power_w())
            temps.append(self.read_temp_c())
            time.sleep(1)

        proc.terminate()
        proc.wait()

        return (
            sum(powers) / len(powers),
            max(powers),
            sum(temps) / len(temps),
            max(temps)
        )

    def scan(self, freq_levels: List[int], voltage_offsets: List[int],
             ppt_limits: List[int], test_duration: int = 30):
        """执行全空间扫描"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        csv_file = f"efficiency_scan_{timestamp}.csv"

        with open(csv_file, 'w', newline='') as f:
            writer = csv.writer(f)
            writer.writerow([
                "freq_level", "voltage_mv", "ppt_w",
                "fps", "avg_power_w", "max_power_w",
                "avg_temp_c", "max_temp_c", "efficiency"
            ])

        total_points = len(freq_levels) * len(voltage_offsets) * len(ppt_limits)
        current = 0

        for freq_lvl in freq_levels:
            for volt_off in voltage_offsets:
                for ppt_w in ppt_limits:
                    current += 1
                    tp = TestPoint(freq_lvl, volt_off, ppt_w)
                    print(f"[{current}/{total_points}] 测试: {tp}")

                    # 应用设置
                    self.set_freq_level(freq_lvl)
                    time.sleep(2)
                    self.set_voltage_offset(volt_off)
                    time.sleep(2)
                    self.set_ppt_limit(ppt_w)
                    time.sleep(3)

                    # 运行测试
                    print(f"  正在测试 {test_duration}s...")
                    avg_p, max_p, avg_t, max_t = self.run_stability_test(test_duration)

                    # 性能数据（来自 glmark2 的 FPS 输出）
                    fps = 0.0  # 实际使用时应解析负载输出

                    # 计算能效
                    efficiency = fps / max(avg_p, 1)
                    result = TestResult(tp, fps, avg_p, max_p, avg_t, max_t, efficiency)
                    self.results.append(result)

                    # 写入 CSV
                    with open(csv_file, 'a', newline='') as f:
                        writer = csv.writer(f)
                        writer.writerow([
                            freq_lvl, volt_off, ppt_w,
                            fps, avg_p, max_p, avg_t, max_t, efficiency
                        ])

                    # 冷却
                    print(f"  冷却中...")
                    time.sleep(10)

        print(f"\n扫描完成！结果已保存至: {csv_file}")
        return self.analyze_results()

    def analyze_results(self):
        """分析扫描结果，找出最优能效点"""
        if not self.results:
            print("没有测试数据")
            return

        best_efficiency = max(self.results, key=lambda r: r.efficiency)
        best_performance = max(self.results, key=lambda r: r.fps)

        print("=" * 60)
        print("扫描结果分析")
        print("=" * 60)
        print(f"最优能效: {best_efficiency.test_point} -> "
              f"{best_efficiency.efficiency:.2f} FPS/W")
        print(f"最高性能: {best_performance.test_point} -> "
              f"{best_performance.fps:.1f} FPS")
        print("=" * 60)
        return best_efficiency, best_performance
```

### C.2 项目二：GPU 稳定性压力测试框架

```bash
#!/bin/bash
# gpu_stress_test_framework.sh
# GPU 稳定性压力测试框架 - 验证电压调节后的系统稳定性

set -e

GPU_ID=${1:-0}
TEST_DURATION=${2:-3600}  # 默认 1 小时
LOG_DIR="stress_test_$(date +%Y%m%d_%H%M%S)"
PASS_THRESHOLD=0.99  # 99% 通过率

mkdir -p "$LOG_DIR"
echo "GPU 稳定性压力测试"
echo "===================="
echo "GPU ID: $GPU_ID"
echo "测试时长: ${TEST_DURATION}s"
echo "日志目录: $LOG_DIR"

# 阶段 1：GPU 信息记录
echo ""
echo "阶段 1/5：记录 GPU 信息"
{
    echo "=== GPU Information ==="
    rocm-smi --showallinfo
    echo ""
    echo "=== V/F Curve ==="
    cat "/sys/class/drm/card${GPU_ID}/device/pp_od_clk_voltage"
    echo ""
    echo "=== Power Limits ==="
    HWMON=$(ls /sys/class/drm/card${GPU_ID}/device/hwmon/)
    cat "/sys/class/drm/card${GPU_ID}/device/hwmon/$HWMON/power1_cap"
    cat "/sys/class/drm/card${GPU_ID}/device/hwmon/$HWMON/power1_input"
} > "$LOG_DIR/gpu_info.txt"

# 阶段 2：3D 渲染稳定性测试
echo ""
echo "阶段 2/5：3D 渲染稳定性测试 (glmark2)"
{
    echo "=== glmark2 Stability Test ==="
    timeout $TEST_DURATION glmark2 --fullscreen --run-forever 2>&1
} > "$LOG_DIR/glmark2_test.log" 2>&1 &
GLMARK_PID=$!

# 阶段 3：计算负载测试（OpenCL/HIP）
echo ""
echo "阶段 3/5：计算稳定性测试 (clpeak)"
{
    echo "=== clpeak Compute Test ==="
    timeout $TEST_DURATION clpeak --all 2>&1
} > "$LOG_DIR/clpeak_test.log" 2>&1 &
CLPEAK_PID=$!

# 阶段 4：并发负载测试
echo ""
echo "阶段 4/5：并发负载测试"
{
    echo "=== Concurrent Load Test ==="
    # 同时运行渲染和计算负载
    timeout $((TEST_DURATION/2)) \
        bash -c "while true; do glmark2 --fullscreen --off-screen --duration 60; done" &
    timeout $((TEST_DURATION/2)) \
        bash -c "while true; do stress-ng --vm 4 --vm-bytes 80% 2>/dev/null; done" &
    wait
} > "$LOG_DIR/concurrent_test.log" 2>&1 &
CONCURRENT_PID=$!

# 阶段 5：实时监控和错误检测
echo ""
echo "阶段 5/5：实时监控和错误检测"
MONITOR_LOG="$LOG_DIR/monitor.csv"
echo "timestamp,temp_c,power_w,freq_mhz,errors" > "$MONITOR_LOG"

ERROR_COUNT=0
HWMON=$(ls /sys/class/drm/card${GPU_ID}/device/hwmon/)

for i in $(seq 1 $((TEST_DURATION / 10))); do
    timestamp=$(date +%Y%m%d_%H%M%S)
    temp=$(cat "/sys/class/drm/card${GPU_ID}/device/hwmon/$HWMON/temp1_input" 2>/dev/null || echo "0")
    power=$(cat "/sys/class/drm/card${GPU_ID}/device/hwmon/$HWMON/power1_input" 2>/dev/null || echo "0")
    freq=$(cat "/sys/class/drm/card${GPU_ID}/device/pp_dpm_sclk" 2>/dev/null | grep "*" | awk '{print $2}' || echo "0")
    
    # 检查 dmesg 是否有新错误
    new_errors=$(dmesg | grep -i "amdgpu.*error\|amdgpu.*fail\|gpu.*reset" | tail -1 2>/dev/null || echo "")
    if [ -n "$new_errors" ]; then
        ERROR_COUNT=$((ERROR_COUNT + 1))
        echo "$new_errors" >> "$LOG_DIR/dmesg_errors.log"
    fi
    
    echo "$timestamp,$((temp/1000)),$((power/1000000)),$freq,$ERROR_COUNT" >> "$MONITOR_LOG"
    
    # 进度显示
    progress=$((i * 10))
    if [ $((progress % 60)) -eq 0 ] || [ $i -eq 1 ]; then
        echo "  进度: ${progress}s / ${TEST_DURATION}s, 错误数: $ERROR_COUNT, 温度: $((temp/1000))°C"
    fi
    
    sleep 10
done

# 等待所有测试完成
wait $GLMARK_PID 2>/dev/null || true
wait $CLPEAK_PID 2>/dev/null || true
wait $CONCURRENT_PID 2>/dev/null || true

# 结果分析
echo ""
echo "===================="
echo "测试结果汇总"
echo "===================="
echo "总错误数: $ERROR_COUNT"
echo "测试时长: ${TEST_DURATION}s"

if [ "$ERROR_COUNT" -eq 0 ]; then
    PASS_RATE=1.0
else
    PASS_RATE=$(echo "scale=2; 1 - $ERROR_COUNT / ($TEST_DURATION / 10)" | bc)
fi

echo "通过率: $(echo "scale=1; $PASS_RATE * 100" | bc)%"

if (( $(echo "$PASS_RATE >= $PASS_THRESHOLD" | bc -l) )); then
    echo "结果: 通过 ✅"
else
    echo "结果: 失败 ❌ (需要 ≥ $(echo "$PASS_THRESHOLD * 100" | bc)%)"
fi

echo "详细日志: $LOG_DIR"
```

## 附录D：故障排查指南

### D.1 频率调整失败

| 症状 | 可能原因 | 排查方法 |
|------|---------|---------|
| 写入 pp_od_clk_voltage 返回 EINVAL | 格式错误 | 检查组是否以 "c" 提交 |
| 频率不变化 | 未设置手动模式 | 检查 power_dpm_force_performance_level |
| 设置后自动恢复 | SMU 固件干预 | 检查 dmesg 中 thermal/power throttle 消息 |
| 系统卡死 | 电压过低导致不稳定 | 增加电压偏移或恢复默认 |
| OverDrive 不可用 | 内核参数未开启 | 检查 amdgpu.ppfeaturemask 是否包含 OverDrive |

### D.2 功耗限制设置失败

| 症状 | 可能原因 | 排查方法 |
|------|---------|---------|
| 写入 power1_cap 返回 EINVAL | 超出范围 | 检查 power1_cap_min 和 power1_cap_max |
| 限制不生效 | 微瓦单位换算错误 | 确认写入值 = 瓦特 × 1,000,000 |
| 限制被自动覆盖 | SMU 安全策略 | dmesg 查看 SMU 消息 |
| 多 GPU 设置混淆 | 多个 hwmon 入口 | 确认操作的 cardX 对应的 GPU |

### D.3 电压调节故障

| 症状 | 可能原因 | 排查方法 |
|------|---------|---------|
| Undervolt 后性能下降 | 降频补偿 | 检查实际运行频率是否低于设定值 |
| 屏幕闪烁/花屏 | 电压过低 | 逐步增加电压直到稳定 |
| 应用程序崩溃 | GPU 计算错误 | 运行 memtest 或 clpeak 验证 |
| 系统重启 | VRM 欠压保护 | 检查主板 BIOS 设置 |

### D.4 监控数据异常

| 症状 | 可能原因 | 排查方法 |
|------|---------|---------|
| 功耗读数为 0 | 驱动未加载 | `lsmod | grep amdgpu` |
| 温度读数不变 | 传感器故障 | 检查 hwmon temp*_input 文件 |
| 频率读数异常 | DPM 状态错误 | `cat power_dpm_force_performance_level` |
| 数据更新延迟 | 采样周期限制 | 确认驱动程序采样频率 |

### F.2 GPU 功耗模型数学公式

**动态功耗**：
```
P_dynamic = C_eff × V² × f

其中：
  C_eff = 有效开关电容（与工艺、架构相关）
  V     = 核心电压 (V)
  f     = 运行频率 (Hz)
```

**静态功耗（漏电流）**：
```
P_static = V × I_leak
I_leak = I_sub + I_gate + I_btbt

I_sub = μ × C_ox × (W/L) × V_T² × exp((V_GS - V_TH) / (n × V_T))
  × (1 - exp(-V_DS / V_T))

其中：
  I_sub  = 亚阈值漏电流（温度敏感）
  I_gate = 栅极漏电流
  I_btbt = 带带隧穿漏电流
  V_T    = 热电压 (kT/q)
  V_TH   = 阈值电压
```

**总功耗**：
```
P_total = P_dynamic + P_static
```

### I.5 Python自动化PM测试框架

```python
#!/usr/bin/env python3
"""
amdgpu_pm_test_framework.py
AMDGPU电源管理自动化测试框架
支持频率/电压/功耗限制的批量测试与数据采集
"""

import subprocess
import time
import csv
import os
import sys
import json
from dataclasses import dataclass, field
from typing import List, Dict, Optional
from datetime import datetime


@dataclass
class PMTestConfig:
    """电源管理测试配置"""
    ppt_watts: List[int] = field(default_factory=lambda: [150, 200, 250])
    sclk_mhz: List[int] = field(default_factory=lambda: [1500, 1800, 2100])
    voltages_mv: List[int] = field(default_factory=lambda: [950, 1000, 1050])
    workload_duration: int = 30
    cooldown_seconds: int = 15
    output_dir: str = "./pm_test_results"
    card_index: int = 0


class AMDGPUPowerTester:
    """AMDGPU电源管理功能测试器"""

    def __init__(self, config: PMTestConfig):
        self.config = config
        self.results: List[Dict] = []
        self.hwmon_path = self._find_hwmon_path()
        self.pp_od_path = f"/sys/class/drm/card{config.card_index}/device/pp_od_clk_voltage"

    def _find_hwmon_path(self) -> Optional[str]:
        """查找hwmon sysfs路径"""
        base = f"/sys/class/drm/card{self.config.card_index}/device/hwmon"
        if not os.path.exists(base):
            return None
        entries = os.listdir(base)
        for e in entries:
            if e.startswith("hwmon"):
                return os.path.join(base, e)
        return None

    def _read_sysfs(self, path: str) -> Optional[str]:
        """安全地读取sysfs文件"""
        try:
            with open(path, 'r') as f:
                return f.read().strip()
        except (IOError, OSError) as e:
            print(f"  [ERROR] 读取失败 {path}: {e}")
            return None

    def _write_sysfs(self, path: str, value: str) -> bool:
        """安全地写入sysfs文件"""
        try:
            with open(path, 'w') as f:
                f.write(value)
            return True
        except (IOError, OSError) as e:
            print(f"  [ERROR] 写入失败 {path}={value}: {e}")
            return False

    def set_ppt_limit(self, watts: int) -> bool:
        """设置功耗限制（微瓦）"""
        if not self.hwmon_path:
            return False
        path = os.path.join(self.hwmon_path, "power1_cap")
        return self._write_sysfs(path, str(watts * 1000000))

    def get_current_power(self) -> Optional[float]:
        """获取当前功耗（瓦）"""
        if not self.hwmon_path:
            return None
        path = os.path.join(self.hwmon_path, "power1_input")
        val = self._read_sysfs(path)
        if val:
            return int(val) / 1000000.0
        return None

    def get_current_temp(self) -> Optional[float]:
        """获取当前温度"""
        if not self.hwmon_path:
            return None
        path = os.path.join(self.hwmon_path, "temp1_input")
        val = self._read_sysfs(path)
        if val:
            return int(val) / 1000.0
        return None

    def run_glmark_load(self, duration: int) -> float:
        """运行glmark2负载并返回平均fps"""
        print(f"  运行glmark2负载 {duration}秒...")
        try:
            result = subprocess.run(
                ["glmark2", "--run-forever"],
                capture_output=True, text=True,
                timeout=duration
            )
            fps_values = []
            for line in result.stdout.split('\n'):
                if 'FPS' in line:
                    parts = line.split()
                    for i, p in enumerate(parts):
                        if p == 'FPS':
                            try:
                                fps_values.append(float(parts[i + 1]))
                            except (ValueError, IndexError):
                                pass
            if fps_values:
                return sum(fps_values) / len(fps_values)
            return 0.0
        except subprocess.TimeoutExpired:
            return 0.0
        except FileNotFoundError:
            print("  [WARN] glmark2未安装，使用stress-ng模拟负载")
            return self._run_stress_load(duration)

    def _run_stress_load(self, duration: int) -> float:
        """在glmark2不可用时使用stress-ng"""
        try:
            subprocess.run(
                ["stress-ng", "--gpu", "1", "--gpu-ops", str(duration * 100)],
                capture_output=True, timeout=duration + 5
            )
        except (subprocess.TimeoutExpired, FileNotFoundError):
            pass
        return 0.0

    def collect_snapshot(self) -> Dict:
        """采集当前状态快照"""
        return {
            "timestamp": datetime.now().isoformat(),
            "power_watts": self.get_current_power(),
            "temp_celsius": self.get_current_temp(),
            "ppt_setting": self._read_sysfs(
                os.path.join(self.hwmon_path, "power1_cap")) if self.hwmon_path else None
        }

    def run_test_sequence(self):
        """执行完整的测试序列"""
        os.makedirs(self.config.output_dir, exist_ok=True)

        print("=" * 60)
        print("AMDGPU 电源管理自动化测试")
        print("=" * 60)

        for ppt in self.config.ppt_watts:
            print(f"\n--- PPT: {ppt}W ---")
            self.set_ppt_limit(ppt)
            time.sleep(2)

            for sclk in self.config.sclk_mhz:
                print(f"  SCLK: {sclk} MHz")
                time.sleep(1)

                snapshot_before = self.collect_snapshot()
                avg_fps = self.run_glmark_load(self.config.workload_duration)
                snapshot_after = self.collect_snapshot()

                self.results.append({
                    "ppt_watts": ppt,
                    "sclk_mhz": sclk,
                    "avg_fps": avg_fps,
                    "power_before": snapshot_before["power_watts"],
                    "power_after": snapshot_after["power_watts"],
                    "temp_before": snapshot_before["temp_celsius"],
                    "temp_after": snapshot_after["temp_celsius"]
                })

                print(f"    FPS: {avg_fps:.1f}, "
                      f"功耗: {snapshot_after['power_watts']:.1f}W, "
                      f"温度: {snapshot_after['temp_celsius']:.1f}°C")

                time.sleep(self.config.cooldown_seconds)

        self._export_results()

    def _export_results(self):
        """导出测试结果为CSV"""
        csv_path = os.path.join(self.config.output_dir, "pm_test_results.csv")
        if not self.results:
            print("无测试结果可导出")
            return

        with open(csv_path, 'w', newline='') as f:
            writer = csv.DictWriter(f, fieldnames=self.results[0].keys())
            writer.writeheader()
            writer.writerows(self.results)

        print(f"\n结果已导出至: {csv_path}")

        json_path = os.path.join(self.config.output_dir, "pm_test_results.json")
        with open(json_path, 'w') as f:
            json.dump(self.results, f, indent=2)
        print(f"JSON结果已导出至: {json_path}")

        print("\n最佳能效配置:")
        best_eff = max(self.results,
                       key=lambda r: r["avg_fps"] / r["power_after"] if r["power_after"] else 0)
        print(f"  PPT={best_eff['ppt_watts']}W, "
              f"SCLK={best_eff['sclk_mhz']}MHz, "
              f"能效={best_eff['avg_fps'] / best_eff['power_after']:.2f} FPS/W")


if __name__ == "__main__":
    config = PMTestConfig(
        ppt_watts=[150, 200, 250],
        sclk_mhz=[1500, 1800, 2100],
        voltages_mv=[950, 1000, 1050],
        workload_duration=30,
        output_dir="./pm_test_results"
    )
    tester = AMDGPUPowerTester(config)
    tester.run_test_sequence()
```

**使用方法**：

```bash
    ppt_rows = [r for r in rows if r[\"ppt_watts\"] == ppt]
    avg_eff = sum(float(r['avg_fps']) / float(r['power_after']) for r in ppt_rows) / len(ppt_rows)
    avg_perf = sum(float(r['avg_fps']) for r in ppt_rows) / len(ppt_rows)
    print(f'PPT={ppt}W: 平均FPS={avg_perf:.1f}, 平均能效={avg_eff:.2f} FPS/W')
```

### J.2 跨平台测试脚本框架

```bash
#!/bin/bash
# cross_platform_pm_test.sh
# 跨平台GPU电源管理对比测试

detect_gpu() {
    if command -v rocm-smi &> /dev/null; then
        echo "AMD"
    elif command -v nvidia-smi &> /dev/null; then
        echo "NVIDIA"
    elif ls /sys/class/drm/card0/device/gt_* &> /dev/null; then
        echo "INTEL"
    else
        echo "UNKNOWN"
    fi
}

get_power_amd() {
    cat /sys/class/drm/card0/device/hwmon/hwmon*/power1_input 2>/dev/null | awk '{printf "%.1f", $1/1000000}'
}

get_power_nvidia() {
    nvidia-smi --query-gpu=power.draw --format=csv,noheader,nounits 2>/dev/null | head -1
}

get_freq_amd() {
    cat /sys/class/drm/card0/device/pp_dpm_sclk 2>/dev/null | tail -1 | awk '{print $1}'
}

get_temp() {
    if [ "$GPU_TYPE" = "AMD" ]; then
        cat /sys/class/drm/card0/device/hwmon/hwmon*/temp1_input 2>/dev/null | awk '{printf "%.1f", $1/1000}'
    elif [ "$GPU_TYPE" = "NVIDIA" ]; then
        nvidia-smi --query-gpu=temperature.gpu --format=csv,noheader,nounits 2>/dev/null
    fi
}

GPU_TYPE=$(detect_gpu)
echo "检测到GPU类型: $GPU_TYPE"

TEST_DURATION=${1:-60}
OUTPUT_FILE="cross_platform_${GPU_TYPE}_$(date +%Y%m%d_%H%M%S).csv"

echo "启动跨平台GPU性能测试 (${TEST_DURATION}秒)"
echo "时间,功耗(W),频率(MHz),温度(°C)" > "$OUTPUT_FILE"

END=$((SECONDS + TEST_DURATION))
while [ $SECONDS -lt $END ]; do
    TIMESTAMP=$(date +%H:%M:%S)
    
    if [ "$GPU_TYPE" = "AMD" ]; then
        POWER=$(get_power_amd)
        FREQ=$(get_freq_amd)
    elif [ "$GPU_TYPE" = "NVIDIA" ]; then
        POWER=$(get_power_nvidia)
        FREQ="N/A"
    fi
    
    TEMP=$(get_temp)
    
    echo "$TIMESTAMP,$POWER,$FREQ,$TEMP" >> "$OUTPUT_FILE"
    echo "$TIMESTAMP 功耗=${POWER}W 频率=${FREQ}MHz 温度=${TEMP}°C"
    
    sleep 1
done

echo "测试完成，结果已保存至: $OUTPUT_FILE"
echo ""
echo "=== 统计摘要 ==="
awk -F',' 'NR>1 {sum_p+=$2; sum_t+=$4; count++} END {printf "平均功耗: %.1fW\n平均温度: %.1f°C\n", sum_p/count, sum_t/count}' "$OUTPUT_FILE"
```

### K.2 答案与解析

**1. B** - MMRR是RDNA 3引入的新特性，允许通过PCIe BAR空间直接读取SMU内部寄存器，绕过传统的Mailbox通信机制，显著降低了延迟。

**2. C** - SMU固件同时监控温度和功耗，当两者都接近限制时，会计算各自的降频量并取更严格的那个执行。这是双窗口PPT算法与热节流协同工作的结果。

**3. B** - GFXOFF是RDNA架构引入的细粒度时钟门控技术，在GPU空闲时自动关闭GFX时钟域，恢复延迟极低（微秒级），对用户透明。

**4. B** - perf可以统计amdgpu_dpm_set_power_state tracepoint的事件频率，通过直方图或调用栈分析可以观察P-State转换的延迟分布。

**5. A** - 在多GPU系统中，每个GPU独立管理自己的电源状态，因为不同GPU可能承受不同的工作负载。但系统级的功耗预算（如整机功耗墙）可能需要协调各GPU的PPT设置。
