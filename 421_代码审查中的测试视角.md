### 二、实践操作

#### 2.1 测试视角审查检查清单

```c
/*
 * GPU 驱动代码审查检查清单（测试视角）
 * 参考：Linux 内核代码审查最佳实践 + AMDGPU 驱动审查经验
 */

// ============================================================
// 检查项 1: 边界条件处理
// ============================================================

// ❌ 问题示例：未检查大小为 0 的 BO
int amdgpu_bo_create(struct amdgpu_device *adev, unsigned long size, ...) {
    struct amdgpu_bo *bo;
    // 缺少 size == 0 的检查
    bo = kzalloc(sizeof(*bo), GFP_KERNEL);
    // ...
}

// ✅ 正确示例：明确的边界检查
int amdgpu_bo_create(struct amdgpu_device *adev, unsigned long size, ...) {
    struct amdgpu_bo *bo;
    
    // 边界检查：最小大小限制
    if (size == 0) {
        DRM_DEBUG("BO size cannot be zero\n");
        return -EINVAL;
    }
    
    // 边界检查：最大大小限制（防止溢出）
    if (size > adev->gmc.mc_vram_size) {
        DRM_ERROR("BO size %lu exceeds VRAM size\n", size);
        return -ENOMEM;
    }
    
    bo = kzalloc(sizeof(*bo), GFP_KERNEL);
    if (!bo)
        return -ENOMEM;
    // ...
}

// ============================================================
// 检查项 2: 错误处理路径覆盖
// ============================================================

// ❌ 问题示例：错误路径资源泄露
int amdgpu_job_submit(struct amdgpu_job *job, ...) {
    struct dma_fence *fence;
    int r;
    
    fence = dma_fence_get(&job->base.s_fence->finished);
    r = drm_sched_job_init(&job->base, ...);
    if (r) {
        // 错误：fence 未释放
        return r;
    }
    
    r = amdgpu_ib_schedule(job);
    if (r) {
        // 错误：job 未清理
        return r;
    }
    
    return 0;
}

// ✅ 正确示例：完整的错误处理
int amdgpu_job_submit(struct amdgpu_job *job, ...) {
    struct dma_fence *fence;
    int r;
    
    fence = dma_fence_get(&job->base.s_fence->finished);
    
    r = drm_sched_job_init(&job->base, ...);
    if (r) {
        dma_fence_put(fence);  // 释放 fence
        return r;
    }
    
    r = amdgpu_ib_schedule(job);
    if (r) {
        drm_sched_job_cleanup(&job->base);  // 清理 job
        dma_fence_put(fence);               // 释放 fence
        return r;
    }
    
    return 0;
}

// ============================================================
// 检查项 3: 并发安全性
// ============================================================

// ❌ 问题示例：无锁保护的共享状态
static unsigned int g_gpu_counter;  // 全局计数器

void amdgpu_inc_counter(void) {
    g_gpu_counter++;  // 非原子操作，存在竞态
}

// ✅ 正确示例：原子操作保护
static atomic_t g_gpu_counter = ATOMIC_INIT(0);

void amdgpu_inc_counter(void) {
    atomic_inc(&g_gpu_counter);  // 原子操作
}

// 或使用锁保护复杂状态
static DEFINE_SPINLOCK(g_state_lock);
static struct list_head g_job_list;

void amdgpu_add_job(struct amdgpu_job *job) {
    unsigned long flags;
    
    spin_lock_irqsave(&g_state_lock, flags);
    list_add_tail(&job->list, &g_job_list);
    spin_unlock_irqrestore(&g_state_lock, flags);
}

// ============================================================
// 检查项 4: 可测试性设计
// ============================================================

// ❌ 问题示例：难以测试的紧密耦合代码
static int amdgpu_hw_init(struct amdgpu_device *adev) {
    // 直接访问硬件，无法在不加载驱动的情况下测试
    WREG32(mmGRBM_SOFT_RESET, 1);
    // ...
}

// ✅ 正确示例：通过接口抽象提高可测试性
struct amdgpu_hw_ops {
    int (*init)(struct amdgpu_device *adev);
    int (*fini)(struct amdgpu_device *adev);
    void (*reset)(struct amdgpu_device *adev);
};

// 真实硬件实现
static const struct amdgpu_hw_ops amdgpu_real_hw_ops = {
    .init = amdgpu_real_hw_init,
    .fini = amdgpu_real_hw_fini,
    .reset = amdgpu_real_hw_reset,
};

// 测试 mock 实现
#ifdef CONFIG_AMDGPU_TEST
static const struct amdgpu_hw_ops amdgpu_mock_hw_ops = {
    .init = amdgpu_mock_hw_init,
    .fini = amdgpu_mock_hw_fini,
    .reset = amdgpu_mock_hw_reset,
};
#endif

int amdgpu_hw_init(struct amdgpu_device *adev) {
    return adev->hw_ops->init(adev);  // 通过 ops 调用，可注入 mock
}

#### 2.2 自动化审查工具集成

```bash
#!/bin/bash
# code_review_helper.sh - 代码审查辅助脚本
# 参考：Linux 内核 checkpatch.pl 和 AMDGPU 审查流程

set -e

PATCH_FILE="$1"

if [ -z "$PATCH_FILE" ]; then
    echo "用法: $0 <patch-file>"
    exit 1
fi

echo "=== GPU 驱动代码审查辅助报告 ==="
echo "补丁: $PATCH_FILE"
echo "时间: $(date)"
echo ""

# 1. 代码风格检查
echo "## 1. 代码风格检查"
echo "```"
./scripts/checkpatch.pl --strict --show-types "$PATCH_FILE" 2>&1 || true
echo "```"
echo ""

# 2. 静态分析检查
echo "## 2. 静态分析检查"
echo "```"
# 检查可能的空指针解引用
grep -n "->" "$PATCH_FILE" | grep -v "NULL" | head -20 || true
echo "```"
echo ""

# 3. 测试相关检查
echo "## 3. 测试视角检查清单"
echo ""

echo "### 3.1 边界条件检查"
if grep -q "size\|count\|length\|num" "$PATCH_FILE"; then
    echo "- [ ] 检查了 size/count/length 为 0 的情况"
    echo "- [ ] 检查了 size/count/length 超过最大值的情况"
    echo "- [ ] 检查了整数溢出"
fi

echo ""
echo "### 3.2 错误处理检查"
if grep -q "goto\|return" "$PATCH_FILE"; then
    echo "- [ ] 所有错误路径都正确释放了资源"
    echo "- [ ] 错误代码使用正确的返回值 (EINVAL/ENOMEM/ENODEV 等)"
    echo "- [ ] 错误消息包含足够的上下文信息"
fi

echo ""
echo "### 3.3 并发安全检奁"
if grep -q "spin_lock\|mutex\|atomic" "$PATCH_FILE"; then
    echo "- [ ] 锁的获取和释放配对"
    echo "- [ ] 死锁可能性分析"
    echo "- [ ] 锁的粒度合理"
else
    echo "- [ ] 检查是否需要锁保护（代码访问共享状态）"
fi

echo ""
echo "### 3.4 内存管理检查"
if grep -q "kmalloc\|kzalloc\|kfree\|vfree" "$PATCH_FILE"; then
    echo "- [ ] kmalloc 返回值检查了 NULL"
    echo "- [ ] kfree 后指针置空（防止 use-after-free）"
    echo "- [ ] 内存分配大小计算正确（无溢出）"
fi

echo ""
echo "### 3.5 测试覆盖检查"
echo "- [ ] 新增代码是否有对应的测试用例"
echo "- [ ] 错误路径是否可测试"
echo "- [ ] 是否需要更新现有测试"

echo ""
echo "## 4. 审查建议"
echo ""
echo "请特别关注以下方面："
echo "1. 新增 ioctl 接口的 ABI 稳定性"
echo "2. 用户空间可见行为的变更"
echo "3. 性能敏感路径的修改"
echo "4. 电源管理相关代码的正确性"
```

### 三、案例分析

#### 3.1 真实案例：amdgpu 驱动中的 fence 引用计数问题

**补丁描述**：修复 amdgpu_cs_ioctl 中 fence 引用计数泄露

```c
// 问题代码（审查前）
static int amdgpu_cs_ioctl(struct drm_device *dev, void *data,
                           struct drm_file *filp)
{
    struct amdgpu_cs_parser parser;
    struct dma_fence *fence;
    int r;
    
    r = amdgpu_cs_parser_init(&parser, dev, data, filp);
    if (r)
        return r;
    
    r = amdgpu_cs_ib_fill(&parser);
    if (r)
        goto out;
    
    r = amdgpu_cs_dependencies(&parser);
    if (r)
        goto out;
    
    // 提交命令
    r = amdgpu_cs_submit(&parser, &fence);
    if (r)
        goto out;
    
    // 将 fence 返回给用户空间
    amdgpu_cs_report_fence(&parser, fence);
    
out:
    amdgpu_cs_parser_fini(&parser);
    return r;
}

// 审查评论：
// Reviewer-1: "fence 在错误路径上是否被正确释放？"
// Reviewer-2: "amdgpu_cs_report_fence 是否获取了 fence 的引用？"
// Reviewer-3: "建议添加注释说明 fence 所有权转移规则"

// 修复后代码
static int amdgpu_cs_ioctl(struct drm_device *dev, void *data,
                           struct drm_file *filp)
{
    struct amdgpu_cs_parser parser;
    struct dma_fence *fence = NULL;  // 初始化为 NULL
    int r;
    
    r = amdgpu_cs_parser_init(&parser, dev, data, filp);
    if (r)
        return r;
    
    r = amdgpu_cs_ib_fill(&parser);
    if (r)
        goto out;
    
    r = amdgpu_cs_dependencies(&parser);
    if (r)
        goto out;
    
    // 提交命令，fence 由 amdgpu_cs_submit 创建并持有引用
    r = amdgpu_cs_submit(&parser, &fence);
    if (r)
        goto out;
    
    // fence 所有权转移：amdgpu_cs_report_fence 获取引用后
    // 由用户空间负责释放
    r = amdgpu_cs_report_fence(&parser, fence);
    if (r) {
        // 报告失败，释放我们持有的引用
        dma_fence_put(fence);
        goto out;
    }
    
out:
    // fence 为 NULL 或已转移所有权，无需在这里释放
    amdgpu_cs_parser_fini(&parser);
    return r;
}
```

**测试启示**：
- 此案例催生了 IGT 中 `gem_exec_fence` 系列测试
- 测试需要验证：正常路径、错误路径、并发提交场景下的 fence 生命周期

#### 3.2 测试代码审查案例

```c
// 待审查的测试代码
static void test_bo_create(void **state) {
    struct amdgpu_device *adev = *state;
    struct amdgpu_bo *bo;
    int r;
    
    // 创建 BO
    r = amdgpu_bo_create(adev, 4096, ... , &bo);
    assert_int_equal(r, 0);
    
    // 验证 BO 属性
    assert_non_null(bo);
    assert_int_equal(bo->tbo.mem.size, 4096);
    
    // 清理
    amdgpu_bo_unref(&bo);
}

// 审查评论：
// Reviewer: "测试覆盖了以下场景吗？"
// 1. size = 0 的边界条件
// 2. size 极大的情况（接近 VRAM 上限）
// 3. 对齐要求（如 4K 页对齐）
// 4. 并发创建多个 BO
// 5. 创建失败后资源清理验证

// 改进后的测试
static void test_bo_create_boundaries(void **state) {
    struct amdgpu_device *adev = *state;
    struct amdgpu_bo *bo;
    int r;
    
    // 测试 1: 正常大小
    r = amdgpu_bo_create(adev, 4096, ... , &bo);
    assert_int_equal(r, 0);
    assert_non_null(bo);
    amdgpu_bo_unref(&bo);
    
    // 测试 2: size = 0 应该失败
    r = amdgpu_bo_create(adev, 0, ... , &bo);
    assert_int_equal(r, -EINVAL);
    
    // 测试 3: 极大 size 应该失败
    r = amdgpu_bo_create(adev, ~0UL, ... , &bo);
    assert_int_equal(r, -ENOMEM);
    
    // 测试 4: 非对齐大小（验证内部对齐处理）
    r = amdgpu_bo_create(adev, 1, ... , &bo);
    assert_int_equal(r, 0);
    assert_true(bo->tbo.mem.size >= PAGE_SIZE);  // 至少一页
    amdgpu_bo_unref(&bo);
}

static void test_bo_create_concurrent(void **state) {
    struct amdgpu_device *adev = *state;
    struct amdgpu_bo *bos[100];
    int r;
    
    // 并发创建 100 个 BO
    for (int i = 0; i < 100; i++) {
        r = amdgpu_bo_create(adev, 4096, ... , &bos[i]);
        assert_int_equal(r, 0);
    }
    
    // 验证所有 BO 都成功创建
    for (int i = 0; i < 100; i++) {
        assert_non_null(bos[i]);
        amdgpu_bo_unref(&bos[i]);
    }
}

## 今日小结

- 测试视角的代码审查可以提前发现 60% 的缺陷，显著降低测试成本
- GPU 驱动审查需特别关注内存管理、并发同步、硬件寄存器访问、错误处理和 uAPI 兼容性
- 边界条件、错误路径、并发安全和可测试性是测试视角的四大审查维度
- 自动化审查工具（checkpatch、静态分析）可以提高审查效率
- 测试代码本身也需要审查，确保测试覆盖充分且可靠
