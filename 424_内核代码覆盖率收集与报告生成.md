# 内核代码覆盖率收集与报告生成

## 学习目标

- 了解覆盖率数据的持久化和增量收集策略
- 能够设计覆盖率驱动的测试改进流程
- 掌握覆盖率报告在团队中的分发和使用方法

### 二、实践操作

#### 2.1 完整覆盖率收集流程

```bash
#!/bin/bash
# full_coverage_workflow.sh - 完整覆盖率收集工作流
# 参考：内核 gcov 文档和 lcov 最佳实践

set -e

# 配置
KERNEL_SRC="${KERNEL_SRC:-/usr/src/linux}"
OUTPUT_DIR="${OUTPUT_DIR:-./coverage_output}"
JOBS="$(nproc)"

mkdir -p "$OUTPUT_DIR"

echo "=========================================="
echo "内核代码覆盖率收集工作流"
echo "=========================================="
echo "内核源码: $KERNEL_SRC"
echo "输出目录: $OUTPUT_DIR"
echo ""

# ═══════════════════════════════════════════════════════════════
# 步骤 1: 配置内核启用覆盖率
# ═══════════════════════════════════════════════════════════════

step1_configure_kernel() {
    echo "【步骤 1】配置内核启用覆盖率支持"
    cd "$KERNEL_SRC"
    
    # 备份原始配置
    cp .config .config.backup 2>/dev/null || true
    
    # 启用 gcov 相关配置
    cat >> .config << 'EOF'
# 调试文件系统
CONFIG_DEBUG_FS=y

# GCOV 覆盖率
CONFIG_GCOV_KERNEL=y
CONFIG_GCOV_FORMAT_AUTODETECT=y

# 可选：启用所有调试选项以获取更完整覆盖
CONFIG_DEBUG_KERNEL=y
CONFIG_DEBUG_INFO=y
CONFIG_DEBUG_INFO_DWARF5=y
EOF
    
    # 使用 oldconfig 应用新配置
    make olddefconfig
    
    # 验证配置
    echo "验证配置:"
    grep "CONFIG_GCOV" .config || true
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 步骤 2: 编译带覆盖率的内核
# ═══════════════════════════════════════════════════════════════

step2_compile_kernel() {
    echo "【步骤 2】编译带覆盖率的内核"
    cd "$KERNEL_SRC"
    
    # 清理旧的覆盖率数据
    make clean
    find . -name "*.gcno" -delete 2>/dev/null || true
    
    # 编译内核
    make -j"$JOBS"
    
    # 编译模块
    make -j"$JOBS" modules
    
    echo "编译完成"
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 步骤 3: 安装内核和模块
# ═══════════════════════════════════════════════════════════════

step3_install_kernel() {
    echo "【步骤 3】安装内核和模块"
    cd "$KERNEL_SRC"
    
    # 安装模块
    sudo make modules_install
    
    # 安装内核
    sudo make install
    
    # 更新引导
    sudo update-grub 2>/dev/null || sudo grub2-mkconfig -o /boot/grub2/grub.cfg
    
    echo "内核安装完成，请重启到新内核"
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 步骤 4: 收集覆盖率数据（运行时）
# ═══════════════════════════════════════════════════════════════

step4_collect_coverage() {
    echo "【步骤 4】收集覆盖率数据"
    
    # 检查 debugfs
    if ! mountpoint -q /sys/kernel/debug; then
        echo "挂载 debugfs..."
        sudo mount -t debugfs none /sys/kernel/debug
    fi
    
    # 检查 gcov 目录
    GCOV_DIR="/sys/kernel/debug/gcov"
    if [ ! -d "$GCOV_DIR" ]; then
        echo "错误: $GCOV_DIR 不存在"
        echo "请确保已重启到带覆盖率支持的内核"
        exit 1
    fi
    
    echo "gcov 数据位置: $GCOV_DIR"
    echo "数据大小:"
    sudo du -sh "$GCOV_DIR" 2>/dev/null || true
    echo ""
    
    # 创建收集目录
    COLLECT_DIR="$OUTPUT_DIR/gcov_data_$(date +%Y%m%d_%H%M%S)"
    mkdir -p "$COLLECT_DIR"
    
    # 复制 gcov 数据（需要 root 权限）
    echo "复制 gcov 数据..."
    sudo rsync -av "$GCOV_DIR/" "$COLLECT_DIR/" 2>/dev/null || \
    sudo cp -r "$GCOV_DIR"/* "$COLLECT_DIR/" 2>/dev/null || true
    
    # 修复权限
    sudo chown -R "$(whoami):$(whoami)" "$COLLECT_DIR"
    
    echo "数据已保存到: $COLLECT_DIR"
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 步骤 5: 生成覆盖率报告
# ═══════════════════════════════════════════════════════════════

step5_generate_report() {
    echo "【步骤 5】生成覆盖率报告"
    
    GCOV_DATA="$OUTPUT_DIR/gcov_data_"*
    LATEST_GCOV=$(ls -td $GCOV_DATA 2>/dev/null | head -1)
    
    if [ -z "$LATEST_GCOV" ] || [ ! -d "$LATEST_GCOV" ]; then
        echo "错误: 未找到 gcov 数据"
        exit 1
    fi
    
    echo "使用数据: $LATEST_GCOV"
    
    # 生成 lcov 信息文件
    INFO_FILE="$OUTPUT_DIR/coverage_$(date +%Y%m%d_%H%M%S).info"
    
    echo "生成 lcov 信息..."
    lcov --capture \
         --directory "$LATEST_GCOV" \
         --base-directory "$KERNEL_SRC" \
         --output-file "$INFO_FILE" \
         --rc lcov_branch_coverage=1
    
    # 过滤只保留 AMDGPU
    AMDGPU_INFO="$OUTPUT_DIR/amdgpu_coverage.info"
    echo "提取 AMDGPU 覆盖率..."
    lcov --extract "$INFO_FILE" \
         "*/drivers/gpu/drm/amd/*" \
         --output-file "$AMDGPU_INFO" \
         --rc lcov_branch_coverage=1
    
    # 移除测试代码自身
    CLEAN_INFO="$OUTPUT_DIR/amdgpu_coverage_clean.info"
    lcov --remove "$AMDGPU_INFO" \
         "*/tests/*" \
         "*/usr/include/*" \
         "*/tools/*" \
         --output-file "$CLEAN_INFO" \
         --rc lcov_branch_coverage=1
    
    # 生成 HTML
    HTML_DIR="$OUTPUT_DIR/html_report_$(date +%Y%m%d_%H%M%S)"
    echo "生成 HTML 报告..."
    genhtml "$CLEAN_INFO" \
            --output-directory "$HTML_DIR" \
            --branch-coverage \
            --show-details \
            --legend \
            --title "AMDGPU 驱动覆盖率报告 $(date)" \
            --prefix "$KERNEL_SRC"
    
    echo ""
    echo "报告已生成:"
    echo "  HTML: $HTML_DIR/index.html"
    echo "  INFO: $CLEAN_INFO"
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 步骤 6: 输出摘要
# ═══════════════════════════════════════════════════════════════

step6_summary() {
    echo "【步骤 6】覆盖率摘要"
    
    INFO_FILE="$OUTPUT_DIR/amdgpu_coverage_clean.info"
    if [ -f "$INFO_FILE" ]; then
        lcov --summary "$INFO_FILE" --rc lcov_branch_coverage=1
    fi
    echo ""
}

# ═══════════════════════════════════════════════════════════════
# 主流程
# ═══════════════════════════════════════════════════════════════

case "${1:-all}" in
    config)
        step1_configure_kernel
        ;;
    compile)
        step2_compile_kernel
        ;;
    install)
        step3_install_kernel
        ;;
    collect)
        step4_collect_coverage
        ;;
    report)
        step5_generate_report
        ;;
    summary)
        step6_summary
        ;;
    all)
        step1_configure_kernel
        step2_compile_kernel
        step4_collect_coverage
        step5_generate_report
        step6_summary
        ;;
    *)
        echo "用法: $0 {config|compile|install|collect|report|summary|all}"
        exit 1
        ;;
esac

echo "=========================================="
echo "覆盖率收集工作流完成"
echo "=========================================="
```

#### 2.2 增量覆盖率收集

```bash
#!/bin/bash
# incremental_coverage.sh - 增量覆盖率收集
# 用于对比两次测试之间的覆盖率变化

set -e

BASELINE_INFO="$1"
CURRENT_INFO="$2"
OUTPUT_DIR="${3:-./incremental_coverage}"

if [ -z "$BASELINE_INFO" ] || [ -z "$CURRENT_INFO" ]; then
    echo "用法: $0 <baseline.info> <current.info> [output_dir]"
    exit 1
fi

mkdir -p "$OUTPUT_DIR"

echo "=== 增量覆盖率分析 ==="
echo "基线: $BASELINE_INFO"
echo "当前: $CURRENT_INFO"
echo ""

# 1. 对比覆盖率变化
DIFF_INFO="$OUTPUT_DIR/coverage_diff.info"
lcov --diff "$BASELINE_INFO" "$CURRENT_INFO" \
     --output-file "$DIFF_INFO" \
     --rc lcov_branch_coverage=1 2>/dev/null || true

# 2. 生成差异报告
echo "## 覆盖率变化摘要"
echo ""
echo "### 基线覆盖率"
lcov --summary "$BASELINE_INFO" --rc lcov_branch_coverage=1 2>&1 | grep -E "lines|functions|branches"

echo ""
echo "### 当前覆盖率"
lcov --summary "$CURRENT_INFO" --rc lcov_branch_coverage=1 2>&1 | grep -E "lines|functions|branches"

echo ""
echo "### 新增覆盖"
lcov --extract "$CURRENT_INFO" --output-file "$OUTPUT_DIR/newly_covered.info" 2>/dev/null || true

# 3. 生成 HTML 差异报告
if [ -f "$DIFF_INFO" ]; then
    genhtml "$DIFF_INFO" \
            --output-directory "$OUTPUT_DIR/html_diff" \
            --branch-coverage \
            --legend \
            --title "覆盖率差异报告"
fi
```

## 今日小结

- 内核覆盖率收集需要编译时启用 CONFIG_GCOV_KERNEL 和运行时挂载 debugfs
- 覆盖率数据通过 .gcno（编译时）和 .gcda（运行时）文件生成
- lcov 是聚合覆盖率数据的标准工具，genhtml 生成可视化报告
- 增量覆盖率分析可以识别测试改进的效果
- 覆盖率报告应关注低覆盖模块和未覆盖的错误处理路径
