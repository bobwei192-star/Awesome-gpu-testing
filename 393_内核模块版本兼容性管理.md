### 概念原理

#### 内核版本兼容性基础

Linux 内核不保证稳定内核 API，模块必须处理不同内核版本的差异：

```
内核模块版本兼容性挑战:

┌────────────────────────────────────────────────────────────┐
│                   内核版本演进                               │
├────────────────────────────────────────────────────────────┤
│                                                            │
│  Linux 5.15         Linux 6.1           Linux 6.6          │
│  ┌──────────────┐   ┌──────────────┐   ┌──────────────┐   │
│  │ amdgpu 5.15  │   │ amdgpu 6.1  │   │ amdgpu 6.6  │   │
│  │              │   │              │   │              │   │
│  │ API:         │   │ API:         │   │ API:         │   │
│  │ - dma_buf    │ → │ - dma_buf    │ → │ - dma_buf    │   │
│  │   v1         │   │   v2         │   │   v3         │   │
│  │ - drm_kms    │   │ - drm_kms    │   │ - drm_kms    │   │
│  │   helper v1  │   │   helper v2  │   │   helper v3  │   │
│  └──────────────┘   └──────────────┘   └──────────────┘   │
│                                                            │
│  兼容性问题:                                                │
│  ┌──────────────────────────────────────────────────────┐  │
│  │ 1. 函数签名变化 (参数/返回值类型)                     │  │
│  │ 2. 结构体字段变化 (新增/删除/重命名)                  │  │
│  │ 3. 枚举/宏定义变化                                   │  │
│  │ 4. 语义变化 (行为不一致)                              │  │
│  │ 5. 依赖的内核子系统变化                               │  │
│  └──────────────────────────────────────────────────────┘  │
│                                                            │
└────────────────────────────────────────────────────────────┘
```

#### MODVERSIONS 机制

```c
// Linux 内核的模块版本校验 (MODVERSIONS / CONFIG_MODVERSIONS)

/*
 * MODVERSIONS 原理:
 * 每个导出符号都有一个 CRC 校验值
 * 加载模块时，内核验证符号的 CRC 是否匹配
 * 如果不匹配，拒绝加载模块
 */

// 内核符号版本控制
// 内核中每个 EXPORT_SYMBOL 都会生成 CRC
// 例如:
// EXPORT_SYMBOL(drm_gem_object_init)
// 
// 生成的 CRC 基于:
// - 函数签名 (参数类型和返回值)
// - 结构体定义 (字段布局)
// - 头文件变化

// 检查模块版本信息
$ modprobe --dump-modversions amdgpu.ko
# 输出示例:
# 0x1234abcd    drm_gem_object_init
# 0x5678ef01    drm_gem_object_lookup
# 0x9abcdef0    drm_gem_object_put

// 处理 MODVERSIONS 不匹配
// 错误示例:
// [  123.456] amdgpu: disagrees about version of symbol drm_gem_object_init
// [  123.456] amdgpu: Unknown symbol drm_gem_object_init (err -2)
```

#### MODULE_IMPORT_NS 命名空间

```c
// 内核 5.3+ 引入的模块命名空间机制

// 在模块中声明使用的命名空间
MODULE_IMPORT_NS(AMDGPU);
MODULE_IMPORT_NS(DRM);
MODULE_IMPORT_NS(DMA_BUF);

/*
 * 命名空间用途:
 * 1. 限制符号导出范围
 * 2. 明确模块依赖关系
 * 3. 避免命名冲突
 * 
 * 内核导出符号的命名空间:
 * EXPORT_SYMBOL_NS(symbol_name, AMDGPU);
 * 
 * 只有声明了 MODULE_IMPORT_NS(AMDGPU) 的模块才能使用该符号
 */

// amdgpu 模块声明
MODULE_IMPORT_NS(DMA_BUF);
MODULE_IMPORT_NS(DRM_GEM);
MODULE_IMPORT_NS(DRM_PRIME);
```

### 实践操作

#### 内核版本检测工具

```bash
#!/bin/bash
# 内核版本兼容性检查

echo "=============================================="
echo "  内核版本兼容性检查"
echo "=============================================="
echo ""

check_kernel_version() {
    echo "[1/6] 内核版本"
    echo "-------------"
    
    local version=$(uname -r)
    local major=$(echo ${version} | cut -d. -f1)
    local minor=$(echo ${version} | cut -d. -f2)
    local patch=$(echo ${version} | cut -d. -f3 | cut -d- -f1)
    
    echo "  当前内核: ${version}"
    echo "  主要版本: ${major}.${minor}.${patch}"
    
    # 版本检查
    local required_major=5
    local required_minor=4
    
    if [ "${major}" -gt "${required_major}" ] || \
       ([ "${major}" -eq "${required_major}" ] && [ "${minor}" -ge "${required_minor}" ]); then
        echo "  ✅ 内核版本满足 amdgpu 要求 (>= ${required_major}.${required_minor})"
    else
        echo "  ❌ 内核版本过低，amdgpu 需要 >= ${required_major}.${required_minor}"
    fi
}

check_module_compatibility() {
    echo ""
    echo "[2/6] 模块兼容性"
    echo "--------------"
    
    # 检查 amdgpu 模块版本
    if [ -d /sys/module/amdgpu ]; then
        echo "  ✅ amdgpu 已加载"
        
        # 查找 amdgpu.ko 位置
        local ko_path=$(modinfo amdgpu 2>/dev/null | grep "^filename" | awk '{print $2}')
        if [ -n "${ko_path}" ]; then
            echo "  模块路径: ${ko_path}"
        fi
        
        # 检查模块版本
        local module_version=$(modinfo amdgpu 2>/dev/null | grep "^version" | awk '{print $2}')
        if [ -n "${module_version}" ]; then
            echo "  模块版本: ${module_version}"
        fi
        
        # 检查模块参数
        echo "  支持的参数:"
        modinfo -p amdgpu 2>/dev/null | head -20 | while read line; do
            echo "    ${line}"
        done
    else
        echo "  ⚠️ amdgpu 未加载"
    fi
}

check_symbol_versions() {
    echo ""
    echo "[3/6] 符号版本一致性"
    echo "-------------------"
    
    if ! command -v modprobe &> /dev/null; then
        echo "  modprobe 不可用"
        return
    fi
    
    local ko_path=$(modinfo amdgpu 2>/dev/null | grep "^filename" | awk '{print $2}')
    if [ -z "${ko_path}" ]; then
        echo "  amdgpu.ko 未找到"
        return
    fi
    
    echo "  关键符号检查:"
    
    # 检查 DRM 符号
    for symbol in "drm_gem_object_init" "drm_gem_object_put" \
                  "drm_gem_prime_import" "drm_gem_prime_export"; do
        if modprobe --dump-modversions "${ko_path}" 2>/dev/null | grep -q "${symbol}"; then
            echo "  ✅ ${symbol}"
        else
            echo "  ❌ ${symbol} (未找到)"
        fi
    done
}

check_drm_compat() {
    echo ""
    echo "[4/6] DRM 子系统兼容性"
    echo "---------------------"
    
    # 检查 DRM 版本
    if [ -d /sys/module/drm ]; then
        echo "  ✅ DRM 模块已加载"
        
        # 检查 DRM 功能
        for feature in "prime" "atomic" "modeset"; do
            local drm_dir="/sys/module/drm/parameters"
            if [ -d "${drm_dir}" ]; then
                echo "  DRM ${feature}: 支持"
            fi
        done
    else
        echo "  ⚠️ DRM 未加载"
    fi
    
    # 检查是否支持 GEM
    if [ -c /dev/dri/card0 ]; then
        echo "  ✅ DRM 设备节点存在"
    fi
}

check_kconfig_options() {
    echo ""
    echo "[5/6] 内核配置选项"
    echo "-----------------"
    
    # 检查关键 Kconfig 选项
    local config_file="/boot/config-$(uname -r)"
    if [ ! -f "${config_file}" ]; then
        config_file="/proc/config.gz"
        if [ -f "${config_file}" ]; then
            # 使用 zcat 读取
            local options=(
                "CONFIG_DRM_AMDGPU"
                "CONFIG_DRM_AMDGPU_SI"
                "CONFIG_DRM_AMDGPU_CIK"
                "CONFIG_DRM_AMD_DC"
                "CONFIG_DRM_AMD_DC_DCN"
                "CONFIG_HSA_AMD"
                "CONFIG_DRM_AMDGPU_USERPTR"
            )
            
            for opt in "${options[@]}"; do
                if zcat "${config_file}" 2>/dev/null | grep -q "^${opt}=y"; then
                    echo "  ✅ ${opt}=y"
                elif zcat "${config_file}" 2>/dev/null | grep -q "^${opt}=m"; then
                    echo "  ⚠️ ${opt}=m (模块化)"
                else
                    echo "  ❌ ${opt}=n (未启用)"
                fi
            done
        else
            echo "  无法读取内核配置"
        fi
    elif [ -f "${config_file}" ]; then
        local options=(
            "CONFIG_DRM_AMDGPU"
            "CONFIG_DRM_AMDGPU_SI"
            "CONFIG_DRM_AMDGPU_CIK"
            "CONFIG_DRM_AMD_DC"
            "CONFIG_DRM_AMD_DC_DCN"
            "CONFIG_HSA_AMD"
            "CONFIG_DRM_AMDGPU_USERPTR"
        )
        
        for opt in "${options[@]}"; do
            if grep -q "^${opt}=y" "${config_file}" 2>/dev/null; then
                echo "  ✅ ${opt}=y"
            elif grep -q "^${opt}=m" "${config_file}" 2>/dev/null; then
                echo "  ⚠️ ${opt}=m (模块化)"
            else
                echo "  ❌ ${opt}=n (未启用)"
            fi
        done
    fi
}

check_backward_compat() {
    echo ""
    echo "[6/6] 向后兼容性"
    echo "--------------"
    
    # 检查旧 GPU 支持 (SI/CIK)
    if [ -f /sys/module/amdgpu/parameters/si_support ]; then
        local si=$(cat /sys/module/amdgpu/parameters/si_support)
        echo "  SI 支持 (GCN 1.0):  ${si}"
    fi
    
    if [ -f /sys/module/amdgpu/parameters/cik_support ]; then
        local cik=$(cat /sys/module/amdgpu/parameters/cik_support)
        echo "  CIK 支持 (GCN 2.0): ${cik}"
    fi
    
    # 检查固件兼容性
    local firmware_dir="/lib/firmware/amdgpu"
    if [ -d "${firmware_dir}" ]; then
        echo ""
        echo "  固件文件:"
        ls "${firmware_dir}" 2>/dev/null | head -20 | while read line; do
            echo "    ${line}"
        done
        echo "  ... 共 $(ls ${firmware_dir} 2>/dev/null | wc -l) 个文件"
    fi
    
    # 检查 dkms 状态
    if command -v dkms &> /dev/null; then
        echo ""
        echo "  DKMS 状态:"
        dkms status 2>/dev/null | grep -i amdgpu | while read line; do
            echo "    ${line}"
        done
    fi
}

# 主流程
check_kernel_version
check_module_compatibility
check_symbol_versions
check_drm_compat
check_kconfig_options
check_backward_compat

echo ""
echo "=============================================="
echo "  兼容性检查完成"
echo "=============================================="
```

// 头文件: amdgpu_compat.h
// 统一处理内核版本差异

#ifndef _AMDGPU_COMPAT_H_
#define _AMDGPU_COMPAT_H_

#include <linux/version.h>

// ============================================================
// 一致性: dma_fence 相关
// ============================================================

#if LINUX_VERSION_CODE < KERNEL_VERSION(5, 4, 0)
// 旧版: dma_fence_get_rcu 不存在
static inline struct dma_fence *
amdgpu_compat_dma_fence_get_rcu(struct dma_fence *fence)
{
    return dma_fence_get_rcu(fence);
}
#else
// 新版: dma_fence_get_rcu 直接可用
#define amdgpu_compat_dma_fence_get_rcu dma_fence_get_rcu
#endif

// ============================================================
// 一致性: drm_gem_object 引用计数
// ============================================================

#if LINUX_VERSION_CODE >= KERNEL_VERSION(5, 12, 0)
// 新版: drm_gem_object_get/put
#define amdgpu_gem_object_get(x) drm_gem_object_get(x)
#define amdgpu_gem_object_put(x) drm_gem_object_put(x)
#else
// 旧版: drm_gem_object_reference/unreference
#define amdgpu_gem_object_get(x) drm_gem_object_reference(x)
#define amdgpu_gem_object_put(x) drm_gem_object_unreference(x)
#endif

// ============================================================
// 一致性: DDR5/G6 内存类型检测
// ============================================================

#if LINUX_VERSION_CODE >= KERNEL_VERSION(6, 3, 0)
#define amdgpu_compat_pci_find_capability(pdev, cap) \
    pci_find_capability(pdev, cap)
#else
static inline int
amdgpu_compat_pci_find_capability(struct pci_dev *pdev, int cap)
{
    return pci_find_capability(pdev, cap);
}
#endif

// ============================================================
// 一致性: drm_crtc_state 结构体
// ============================================================

#if LINUX_VERSION_CODE >= KERNEL_VERSION(6, 1, 0)
// drm_crtc_state 新增了 no_vblank 字段
#define amdgpu_compat_crtc_no_vblank(state) ((state)->no_vblank)
#else
#define amdgpu_compat_crtc_no_vblank(state) (false)
#endif

// ============================================================
// 一致性: pcie_aspm 相关
// ============================================================

#if LINUX_VERSION_CODE >= KERNEL_VERSION(5, 14, 0)
static inline int
amdgpu_compat_pcie_set_aspm(struct pci_dev *pdev, bool enable)
{
    return pcie_aspm_set(pdev, enable);
}
#else
static inline int
amdgpu_compat_pcie_set_aspm(struct pci_dev *pdev, bool enable)
{
    // 旧版不支持 ASPM 控制
    return 0;
}
#endif

// ============================================================
// 一致性: devm_ 资源管理
// ============================================================

#if LINUX_VERSION_CODE >= KERNEL_VERSION(5, 11, 0)
// devm_drm_dev_alloc 可用
#define amdgpu_compat_devm_drm_alloc(dev, type, member) \
    devm_drm_dev_alloc(dev, type, member)
#else
// 回退到手动分配
static inline void *
amdgpu_compat_devm_drm_alloc(struct device *dev,
                               size_t size, size_t offset)
{
    void *ptr = devm_kzalloc(dev, size, GFP_KERNEL);
    if (ptr)
        *(struct drm_device **)((char *)ptr + offset) = NULL;
    return ptr;
}
#endif

#endif /* _AMDGPU_COMPAT_H_ */

#### 跨版本模块构建工具

```bash
#!/bin/bash
# 多内核版本模块构建工具

set -e

echo "=============================================="
echo "  跨版本 amdgpu 模块构建"
echo "=============================================="
echo ""

# 配置
SOURCE_DIR="/usr/src/amdgpu"
BUILD_DIR="/tmp/amdgpu-build"
KERNEL_VERSIONS=(
    "5.15.0"
    "6.1.0"
    "6.6.0"
)

build_for_kernel() {
    local kernel_version=$1
    local kernel_dir="/lib/modules/${kernel_version}/build"
    
    echo "构建内核 ${kernel_version}..."
    
    if [ ! -d "${kernel_dir}" ]; then
        echo "  ❌ 内核源码目录不存在: ${kernel_dir}"
        return 1
    fi
    
    # 创建构建目录
    local build_dir="${BUILD_DIR}/${kernel_version}"
    mkdir -p "${build_dir}"
    
    # 设置内核构建环境
    export KERNELDIR="${kernel_dir}"
    export ARCH=$(uname -m)
    
    # 执行构建
    cd "${SOURCE_DIR}"
    
    # 清理旧构建
    make -C "${KERNELDIR}" M="${SOURCE_DIR}" clean 2>/dev/null || true
    
    # 构建模块
    make -C "${KERNELDIR}" M="${SOURCE_DIR}" modules 2>&1 | \
        while read line; do
            if echo "${line}" | grep -q "error\|warning\|ERROR\|Error"; then
                echo "  ⚠️  ${line}"
            fi
        done
    
    # 检查构建结果
    if [ -f "${SOURCE_DIR}/amdgpu.ko" ]; then
        echo "  ✅ 构建成功"
        
        # 复制到构建目录
        mkdir -p "${build_dir}/kernel/drivers/gpu/drm/amd/amdgpu"
        find "${SOURCE_DIR}" -name "*.ko" -exec cp {} \
            "${build_dir}/kernel/drivers/gpu/drm/amd/amdgpu/" \;
        
        # 复制固件
        if [ -d "${SOURCE_DIR}/firmware" ]; then
            cp -r "${SOURCE_DIR}/firmware" "${build_dir}/"
        fi
        
        echo "  📦 模块已保存到 ${build_dir}"
    else
        echo "  ❌ 构建失败"
        return 1
    fi
}

check_build_deps() {
    echo "检查构建依赖..."
    echo "-------------------"
    
    # 检查必需的构建工具
    local deps=("make" "gcc" "ld" "modprobe" "depmod")
    
    for dep in "${deps[@]}"; do
        if command -v "${dep}" &> /dev/null; then
            echo "  ✅ ${dep}"
        else
            echo "  ❌ ${dep} (未安装)"
        fi
    done
    
    # 检查内核头文件
    for ver in "${KERNEL_VERSIONS[@]}"; do
        local header_dir="/lib/modules/${ver}/build"
        if [ -d "${header_dir}" ]; then
            echo "  ✅ 内核 ${ver} 头文件"
        else
            echo "  ⚠️  内核 ${ver} 头文件缺失"
        fi
    done
    
    echo ""
}

verify_module() {
    local ko_file=$1
    local kernel_version=$2
    
    echo "  验证模块: ${ko_file}"
    echo "    针对内核: ${kernel_version}"
    
    # 检查模块格式
    if modprobe --dump-modversions "${ko_file}" &>/dev/null; then
        echo "    ✅ 符号版本表有效"
    else
        echo "    ❌ 符号版本表无效"
        return 1
    fi
    
    # 检查模块依赖
    local deps=$(modprobe --dump-dependencies "${ko_file}" 2>/dev/null)
    if [ -n "${deps}" ]; then
        echo "    依赖: ${deps}"
    fi
    
    # 检查模块版本
    local version=$(modprobe --first-time "${ko_file}" 2>&1 || true)
    echo "    版本信息: $(strings "${ko_file}" | grep "^ver" 2>/dev/null || echo 'N/A')"
    
    return 0
}

# 主流程
case "$1" in
    build-all)
        check_build_deps
        
        for ver in "${KERNEL_VERSIONS[@]}"; do
            build_for_kernel "${ver}"
            echo ""
        done
        ;;
    
    build)
        if [ -z "$2" ]; then
            echo "用法: $0 build <kernel-version>"
            exit 1
        fi
        check_build_deps
        build_for_kernel "$2"
        ;;
    
    verify)
        if [ -z "$2" ]; then
            echo "用法: $0 verify <module.ko> [kernel-version]"
            exit 1
        fi
        verify_module "$2" "$3"
        ;;
    
    test-load)
        if [ -z "$2" ]; then
            echo "用法: $0 test-load <kernel-version>"
            exit 1
        fi
        
        local build_dir="${BUILD_DIR}/${2}"
        if [ -d "${build_dir}" ]; then
            echo "测试加载模块 (内核 ${2})..."
            
            # 尝试插入模块
            insmod "${build_dir}/kernel/drivers/gpu/drm/amd/amdgpu/amdgpu.ko" 2>&1 || \
                echo "  ⚠️ 加载失败 (可能已在运行)"
        else
            echo "  ❌ 未找到构建目录: ${build_dir}"
        fi
        ;;
    
    *)
        echo "用法: $0 {build-all|build|verify|test-load}"
        echo ""
        echo "  build-all     - 为所有支持的内核版本构建"
        echo "  build <ver>   - 为指定内核版本构建"
        echo "  verify <ko>   - 验证模块兼容性"
        echo "  test-load     - 测试模块加载"
        echo ""
        echo "示例:"
        echo "  $0 build-all"
        echo "  $0 build 6.6.0"
        echo "  $0 verify amdgpu.ko 6.6.0"
        ;;
esac
```

### 案例分析

#### 案例一：内核升级后 amdgpu 模块加载失败

**问题现象：**
```
从内核 5.15 升级到 6.1 后，amdgpu 模块无法加载:
[    3.456] amdgpu: disagrees about version of symbol drm_gem_object_init
[    3.456] amdgpu: Unknown symbol drm_gem_object_init (err -2)
```

**诊断过程：**
```bash
# 1. 检查内核版本
uname -r
# 6.1.0

# 2. 检查模块版本
modinfo amdgpu | grep version
# version: 5.15.0  ← 旧版本模块!

# 3. 检查符号版本
modprobe --dump-modversions /lib/modules/6.1.0/kernel/drivers/gpu/drm/amd/amdgpu/amdgpu.ko
# 确定 amdgpu 模块的内核版本

# 4. 确认问题
# 内核升级后，模块没有重新编译
# 5.15 编译的模块不能在 6.1 内核上运行
```

**解决方案：**
```bash
# 方案 1: 重建 initramfs
sudo update-initramfs -u -k 6.1.0

# 方案 2: 重新安装 amdgpu 驱动
# (使用 DKMS 自动重建)
sudo dkms install amdgpu/6.1.0

# 方案 3: 手动重建
cd /usr/src/amdgpu
make -C /lib/modules/6.1.0/build M=$PWD modules
make -C /lib/modules/6.1.0/build M=$PWD modules_install
depmod -a 6.1.0

# 验证
modinfo amdgpu | grep version
# version: 6.1.0  ← 版本正确

# 加载测试
modprobe amdgpu
lsmod | grep amdgpu
# amdgpu 已加载
```

#### 案例二：内核 API 变更导致编译失败

**问题现象：**
```
在内核 6.6 编译 amdgpu 报错:
drivers/gpu/drm/amd/amdgpu/amdgpu_device.c:1234:23: 
error: 'struct drm_gem_object' has no member named 'vma_node'
 1234 |     obj->vma_node.vm_start = addr;
      |                      ^~~~~~~
```

**诊断过程：**
```bash
# 内核 6.6 移除了 drm_gem_object.vma_node
# 改为通过 drm_gem_object_get_vma_node() 访问

# 查看内核变更
grep -r "vma_node" /lib/modules/6.6.0/build/include/drm/drm_gem.h
# 确认 API 变更
```

**解决方案：**
```c
// 添加条件编译兼容性代码

// amdgpu_compat.h
#if LINUX_VERSION_CODE >= KERNEL_VERSION(6, 6, 0)
// 新 API: 使用 drm_gem_object_get_vma_node()
static inline struct drm_vma_offset_node *
amdgpu_gem_get_vma_node(struct drm_gem_object *obj)
{
    return drm_gem_object_get_vma_node(obj);
}
#else
// 旧 API: 直接访问结构体成员
static inline struct drm_vma_offset_node *
amdgpu_gem_get_vma_node(struct drm_gem_object *obj)
{
    return &obj->vma_node;
}
#endif

// 使用方式:
// struct drm_vma_offset_node *node = amdgpu_gem_get_vma_node(obj);
// node->vm_start = addr;
```

### 今日小结

内核模块版本兼容性管理是 GPU 驱动开发的重要环节：

1. **内核 API 不稳定性**: Linux 内核不保证稳定 API，每个版本都可能变化
2. **MODVERSIONS 机制**: 通过 CRC 校验符号版本，防止模块加载到错误内核
3. **条件编译**: 使用 LINUX_VERSION_CODE 宏处理版本差异
4. **兼容性头文件**: 封装版本差异，统一接口
5. **构建策略**: 每个内核版本单独编译模块
6. **测试验证**: 模块加载测试、符号版本检查、功能验证
7. **常见问题**: 内核升级后模块不匹配、API 变更导致编译失败
