## 知识详解

### 一、概念原理

#### 1.1 测试设计原则

```
┌─────────────────────────────────────────────────────────────────────┐
│              IGT 测试设计原则                                        │
│                                                                     │
│  1. 单一职责                                                         │
│     ├── 每个子测试验证一个具体功能点                                 │
│     ├── 避免一个子测试验证多个不相关的功能                           │
│     └── 子测试名称应清晰描述测试内容                                 │
│                                                                     │
│  2. 独立性                                                           │
│     ├── 子测试之间不应有依赖关系                                     │
│     ├── 每个子测试应能独立运行                                       │
│     └── 失败不应影响其他子测试                                       │
│                                                                     │
│  3. 可重复性                                                         │
│     ├── 相同输入应产生相同结果                                       │
│     ├── 避免依赖外部状态（时间、随机数等）                           │
│     └── 使用固定种子如果需要随机数                                   │
│                                                                     │
│  4. 完备性                                                           │
│     ├── 覆盖正常路径和错误路径                                       │
│     ├── 测试边界条件                                                 │
│     └── 验证错误处理                                                 │
│                                                                     │
│  5. 可维护性                                                         │
│     ├── 代码清晰易读                                                 │
│     ├── 添加注释说明测试目的                                         │
│     └── 使用有意义的变量名                                           │
│                                                                     │
└─────────────────────────────────────────────────────────────────────┘
```

### 二、实践操作

#### 2.1 完整测试示例

```c
/*
 * tests/amdgpu/amd_buffer.c
 * AMDGPU 缓冲区操作测试
 *
 * 测试目标：验证 AMDGPU GEM 缓冲区的创建、映射和销毁
 */

#include "igt.h"
#include <amdgpu.h>
#include <amdgpu_drm.h>

IGT_TEST_DESCRIPTION("AMDGPU GEM buffer object operations");

/* 测试状态 */
static int drm_fd = -1;
static amdgpu_device_handle device = NULL;
static uint32_t major_version;
static uint32_t minor_version;

/*
 * 辅助函数：创建 BO
 */
static amdgpu_bo_handle create_bo(uint64_t size, uint32_t domain)
{
    amdgpu_bo_handle bo;
    amdgpu_bo_alloc_req req;
    amdgpu_bo_alloc_result res;
    int ret;
    
    memset(&req, 0, sizeof(req));
    req.alloc_size = size;
    req.phys_alignment = 4096;
    req.preferred_heap = domain;
    
    ret = amdgpu_bo_alloc(device, &req, &bo);
    igt_assert_eq(ret, 0);
    
    /* 验证分配结果 */
    memset(&res, 0, sizeof(res));
    ret = amdgpu_bo_query_info(bo, &res);
    igt_assert_eq(ret, 0);
    igt_assert_eq(res.alloc_size, size);
    
    return bo;
}

/*
 * 辅助函数：销毁 BO
 */
static void destroy_bo(amdgpu_bo_handle bo)
{
    int ret = amdgpu_bo_free(bo);
    igt_assert_eq(ret, 0);
}

/*
 * 测试夹具：初始化
 */
igt_fixture {
    /* 打开 AMDGPU 设备 */
    drm_fd = drm_open_driver(DRIVER_AMDGPU);
    igt_require_f(drm_fd >= 0,
                  "AMDGPU driver not available\n");
    
    /* 初始化 AMDGPU 设备 */
    int ret = amdgpu_device_initialize(drm_fd, &major_version,
                                       &minor_version, &device);
    igt_require_f(ret == 0,
                  "Failed to initialize AMDGPU device: %d\n", ret);
    
    igt_info("AMDGPU device initialized (version %d.%d)\n",
             major_version, minor_version);
}

/*
 * 子测试 1：基本 BO 创建和销毁
 */
igt_subtest("bo-create-destroy") {
    const uint64_t sizes[] = {4096, 65536, 1024 * 1024, 16 * 1024 * 1024};
    
    for (int i = 0; i < ARRAY_SIZE(sizes); i++) {
        amdgpu_bo_handle bo;
        
        igt_info("Testing BO size: %lu bytes\n", sizes[i]);
        
        /* 在 GTT 域创建 */
        bo = create_bo(sizes[i], AMDGPU_GEM_DOMAIN_GTT);
        destroy_bo(bo);
        
        /* 在 VRAM 域创建 */
        bo = create_bo(sizes[i], AMDGPU_GEM_DOMAIN_VRAM);
        destroy_bo(bo);
    }
}

/*
 * 子测试 2：BO 内存映射
 */
igt_subtest("bo-mmap") {
    amdgpu_bo_handle bo;
    void *ptr;
    int ret;
    
    /* 创建 BO */
    bo = create_bo(4096, AMDGPU_GEM_DOMAIN_GTT);
    
    /* 映射 BO */
    ret = amdgpu_bo_cpu_map(bo, &ptr);
    igt_assert_eq(ret, 0);
    igt_assert_neq(ptr, NULL);
    
    /* 写入数据 */
    memset(ptr, 0xAB, 4096);
    
    /* 验证数据 */
    for (int i = 0; i < 4096; i++) {
        igt_assert_f(((uint8_t *)ptr)[i] == 0xAB,
                     "Data mismatch at offset %d\n", i);
    }
    
    /* 解除映射 */
    ret = amdgpu_bo_cpu_unmap(bo);
    igt_assert_eq(ret, 0);
    
    /* 清理 */
    destroy_bo(bo);
}

/*
 * 子测试 3：无效参数测试
 */
igt_subtest("bo-invalid") {
    amdgpu_bo_handle bo;
    amdgpu_bo_alloc_req req;
    int ret;
    
    /* 测试零大小 */
    memset(&req, 0, sizeof(req));
    req.alloc_size = 0;
    req.preferred_heap = AMDGPU_GEM_DOMAIN_GTT;
    
    ret = amdgpu_bo_alloc(device, &req, &bo);
    igt_assert_f(ret != 0, "Zero-size allocation should fail\n");
    
    /* 测试超大大小 */
    memset(&req, 0, sizeof(req));
    req.alloc_size = ~0ULL;  /* 最大值 */
    req.preferred_heap = AMDGPU_GEM_DOMAIN_GTT;
    
    ret = amdgpu_bo_alloc(device, &req, &bo);
    igt_assert_f(ret != 0, "Oversized allocation should fail\n");
}

/*
 * 子测试 4：并发 BO 操作
 */
igt_subtest("bo-concurrent") {
    const int num_bos = 100;
    amdgpu_bo_handle bos[num_bos];
    
    /* 并发创建 */
    for (int i = 0; i < num_bos; i++) {
        bos[i] = create_bo(4096, AMDGPU_GEM_DOMAIN_GTT);
    }
    
    igt_info("Created %d BOs concurrently\n", num_bos);
    
    /* 并发销毁 */
    for (int i = 0; i < num_bos; i++) {
        destroy_bo(bos[i]);
    }
    
    igt_info("Destroyed %d BOs concurrently\n", num_bos);
}

/*
 * 测试夹具：清理
 */
igt_fixture {
    if (device) {
        amdgpu_device_deinitialize(device);
        device = NULL;
    }
    
    if (drm_fd >= 0) {
        close(drm_fd);
        drm_fd = -1;
    }
    
    igt_info("AMDGPU device cleaned up\n");
}
```

#### 2.2 编译和运行

```bash
#!/bin/bash
# build_and_run_custom_test.sh

set -e

IGT_DIR="${1:-$HOME/igt-gpu-tools}"
TEST_NAME="amd_buffer"

echo "=== 编译和运行自定义 IGT 测试 ==="
echo ""

# 1. 复制测试到 IGT 目录
echo "【1】复制测试文件"
cp "${TEST_NAME}.c" "$IGT_DIR/tests/amdgpu/"

# 2. 更新 meson.build
echo "【2】更新构建配置"
cd "$IGT_DIR"

# 检查是否已添加
if ! grep -q "$TEST_NAME" tests/amdgpu/meson.build; then
    echo "请手动在 tests/amdgpu/meson.build 中添加:"
    echo "  test_progs += [ '$TEST_NAME' ]"
    read -p "按 Enter 继续..."
fi

# 3. 编译
echo "【3】编译"
ninja -C build

# 4. 运行
echo "【4】运行测试"
TEST_BIN="$IGT_DIR/build/tests/amdgpu/$TEST_NAME"

if [ -f "$TEST_BIN" ]; then
    echo "列出子测试:"
    "$TEST_BIN" --list-subtests
    
    echo ""
    echo "运行所有子测试:"
    sudo "$TEST_BIN" 2>&1 | tee "${TEST_NAME}.log"
else
    echo "错误: 编译失败，未找到测试程序"
    exit 1
fi

echo ""
echo "=== 完成 ==="
```

### 三、案例分析

#### 3.1 测试代码审查

```
审查清单：

□ 测试描述
  - 是否有 IGT_TEST_DESCRIPTION？
  - 描述是否清晰说明测试目的？

□ 设备管理
  - 是否正确打开和关闭 DRM 设备？
  - 是否使用 igt_require 检查设备可用性？

□ 子测试设计
  - 每个子测试是否独立？
  - 子测试名称是否清晰？
  - 是否覆盖正常和错误路径？

□ 断言使用
  - 是否使用适当的断言类型？
  - 断言消息是否清晰？

□ 资源管理
  - 是否正确释放所有资源？
  - 是否有内存泄露？

□ 错误处理
  - 是否测试错误路径？
  - 错误消息是否清晰？
```

## 今日小结

- 编写 IGT 测试需要遵循单一职责、独立性、可重复性、完备性和可维护性原则
- 测试结构包括 IGT_TEST_DESCRIPTION、igt_fixture、igt_subtest 和 igt_main
- 辅助函数可以提高代码复用性和可读性
- 测试应覆盖正常路径、错误路径和边界条件
- 资源管理（打开/关闭设备、创建/销毁 BO）需要特别注意
