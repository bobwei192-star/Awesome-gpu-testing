## 2. 单元测试（Unit Testing）

### 2.1 单元测试定义与目标

单元测试是对软件最小可测试单元（函数、方法、模块）的验证活动。在 GPU 驱动的上下文中，一个"单元"可以是：

- 单个 C 函数（如 `amdgpu_bo_create()`）
- 内部辅助函数（如 `amdgpu_vm_bo_update_mapping()`）
- 数据结构操作（如 `list_add_tail()` 在特定上下文的应用）
- 宏和内联函数（如 `WREG32_FIELD()`）

**单元测试的核心目标**：

1. **验证逻辑正确性**：确保函数在隔离环境中按预期工作
2. **捕获回归缺陷**：代码变更后快速发现已有功能破坏
3. **提供使用示例**：测试代码作为函数的实际使用文档
4. **支持重构**：确保重构不改变外部行为
5. **降低调试成本**：在最小范围内定位问题

**参考来源**：Kent Beck, "Test Driven Development: By Example", 2002, Addison-Wesley

### 2.2 AMDGPU 驱动的单元测试框架：KUnit

Linux 内核自 5.5 版本开始集成了 KUnit 测试框架，为内核代码提供单元测试支持。AMDGPU 驱动团队已经逐步开始使用 KUnit 进行单元测试。

#### 2.2.1 KUnit 基本结构

```c
// SPDX-License-Identifier: GPL-2.0
/*
 * KUnit 测试示例：amdgpu_bo_kunit.c
 * 测试 amdgpu_bo 核心数据结构操作
 */

#include <linux/module.h>
#include <kunit/test.h>

/* 测试辅助结构 */
struct amdgpu_bo_test_ctx {
    struct amdgpu_device *adev;
    struct amdgpu_bo *bo;
};

/* 测试准备：每个测试用例执行前调用 */
static int amdgpu_bo_test_init(struct kunit *test)
{
    struct amdgpu_bo_test_ctx *ctx;

    ctx = kunit_kzalloc(test, sizeof(*ctx), GFP_KERNEL);
    KUNIT_ASSERT_NOT_ERR_OR_NULL(test, ctx);

    /* 初始化模拟设备（通过 KUnit mock 框架） */
    ctx->adev = kunit_kzalloc(test, sizeof(*ctx->adev), GFP_KERNEL);
    KUNIT_ASSERT_NOT_ERR_OR_NULL(test, ctx->adev);

    test->priv = ctx;
    return 0;
}

/* 测试清理：每个测试用例执行后调用 */
static void amdgpu_bo_test_exit(struct kunit *test)
{
    struct amdgpu_bo_test_ctx *ctx = test->priv;

    if (ctx->bo) {
        amdgpu_bo_unref(&ctx->bo);
    }
    /* kunit_kzalloc 自动释放 */
}

/*
 * 测试用例 1：基本 BO 创建
 * 验证使用默认参数创建 BO 的正确性
 */
static void amdgpu_bo_create_default_test(struct kunit *test)
{
    struct amdgpu_bo_test_ctx *ctx = test->priv;
    struct amdgpu_bo *bo;
    int r;

    /* 分配 4KB BO，默认对齐到 PAGE_SIZE */
    r = amdgpu_bo_create(ctx->adev, 4096, PAGE_SIZE, false,
                          AMDGPU_GEM_DOMAIN_VRAM,
                          AMDGPU_GEM_CREATE_CPU_ACCESS_REQUIRED,
                          NULL, NULL, &bo);

    KUNIT_ASSERT_EQ(test, r, 0);
    KUNIT_ASSERT_NOT_ERR_OR_NULL(test, bo);

    /* 验证 BO 属性 */
    KUNIT_EXPECT_EQ(test, bo->tbo.base.size, 4096);
    KUNIT_EXPECT_EQ(test, bo->preferred_domain, AMDGPU_GEM_DOMAIN_VRAM);
    KUNIT_EXPECT_TRUE(test, bo->flags & AMDGPU_GEM_CREATE_CPU_ACCESS_REQUIRED);

    ctx->bo = bo;
}

/*
 * 测试用例 2：大小为 0 的 BO 创建
 * 验证边界条件处理
 */
static void amdgpu_bo_create_zero_size_test(struct kunit *test)
{
    struct amdgpu_bo_test_ctx *ctx = test->priv;
    struct amdgpu_bo *bo;
    int r;

    r = amdgpu_bo_create(ctx->adev, 0, PAGE_SIZE, false,
                          AMDGPU_GEM_DOMAIN_VRAM, 0, NULL, NULL, &bo);

    KUNIT_EXPECT_EQ(test, r, -EINVAL);
    KUNIT_EXPECT_NULL(test, bo);
}

/*
 * 测试用例 3：BO 引用计数
 * 验证引用计数的正确增减
 */
static void amdgpu_bo_refcount_test(struct kunit *test)
{
    struct amdgpu_bo_test_ctx *ctx = test->priv;
    struct amdgpu_bo *bo;
    struct amdgpu_bo *bo_ref;
    int r;

    r = amdgpu_bo_create(ctx->adev, 4096, PAGE_SIZE, false,
                          AMDGPU_GEM_DOMAIN_VRAM, 0, NULL, NULL, &bo);
    KUNIT_ASSERT_EQ(test, r, 0);

    /* 获取引用 */
    bo_ref = amdgpu_bo_ref(bo);
    KUNIT_EXPECT_PTR_EQ(test, bo_ref, bo);
    KUNIT_EXPECT_EQ(test, kref_read(&bo->tbo.base.refcount), 2);

    /* 释放引用 */
    amdgpu_bo_unref(&bo_ref);
    KUNIT_EXPECT_NULL(test, bo_ref);
    KUNIT_EXPECT_EQ(test, kref_read(&bo->tbo.base.refcount), 1);

    ctx->bo = bo;
}

/* 注册测试套件 */
static struct kunit_case amdgpu_bo_test_cases[] = {
    KUNIT_CASE(amdgpu_bo_create_default_test),
    KUNIT_CASE(amdgpu_bo_create_zero_size_test),
    KUNIT_CASE(amdgpu_bo_refcount_test),
    {}
};

static struct kunit_suite amdgpu_bo_test_suite = {
    .name = "amdgpu-bo",
    .init = amdgpu_bo_test_init,
    .exit = amdgpu_bo_test_exit,
    .test_cases = amdgpu_bo_test_cases,
};

kunit_test_suite(amdgpu_bo_test_suite);

MODULE_LICENSE("GPL");
MODULE_AUTHOR("AMD GPU Driver Team");
MODULE_DESCRIPTION("KUnit tests for AMDGPU BO operations");
```

### 2.3 KUnit Mock 框架的使用

对于 GPU 驱动，许多函数依赖于硬件寄存器和中断等不可模拟的资源。KUnit 的 Mock 框架提供了解耦机制：

```c
/*
 * 使用 KUnit Mock 模拟硬件寄存器访问
 */

/* 定义 mock 函数指针 */
static int (*mock_wreg32)(struct amdgpu_device *adev, uint32_t reg, uint32_t val);

/* 原始的 amdgpu_wreg32 包装 */
int amdgpu_wreg32_mockable(struct amdgpu_device *adev, uint32_t reg, uint32_t val)
{
    if (mock_wreg32)
        return mock_wreg32(adev, reg, val);
    return amdgpu_wreg32_impl(adev, reg, val);
}

/* 测试用例：验证 GFX_OFF 切换 */
static void amdgpu_gfx_off_switch_test(struct kunit *test)
{
    struct amdgpu_device *adev;
    bool gfx_off_called = false;

    /* 设置 mock：验证写入 GFX_OFF 控制寄存器 */
    mock_wreg32 = kunit_mock_function(test, (typeof(mock_wreg32))
        ({ if (reg == mmRLC_GPM_STAT)
               gfx_off_called = true;
           return 0; }));

    /* 调用 GFX_OFF 切换 */
    amdgpu_gfx_off_control(adev, true);

    /* 验证 GFX_OFF 被调起 */
    KUNIT_EXPECT_TRUE(test, gfx_off_called);

    /* 清理 mock */
    mock_wreg32 = NULL;
}
```

**参考来源**：Linux Kernel KUnit Mock Documentation, `Documentation/dev-tools/kunit/mock.rst`

### 2.4 单元测试的最佳实践

#### 2.4.1 FIRST 原则

```
F - Fast (快速): 单个测试应在毫秒内完成
I - Isolated (隔离): 测试不依赖其他测试的顺序
R - Repeatable (可重复): 任意顺序多次执行结果一致
S - Self-validating (自验证): 测试自动判断 pass/fail
T - Timely (及时): 测试应与代码同时编写
```

## 3. 集成测试（Integration Testing）

### 3.1 集成测试定义与目标

集成测试验证多个单元/模块之间的交互是否正确。在 GPU 驱动中，集成测试关注：

- **模块间接口**：不同 IP Block 的初始化和通信
- **内核子系统交互**：DRM 核心与 AMDGPU 驱动的接口
- **用户-内核交互**：ioctl 路径的完整性
- **硬件交互**：寄存器写入/读取的正确流程
- **数据协议一致性**：SMU 消息格式、fence 协议等

**集成测试的关键目标**：

1. **接口合规性**：验证模块间的契约是否得到遵守
2. **数据流完整性**：确保数据在模块间正确传递和转换
3. **错误传播**：验证错误在模块间正确传递和处理
4. **时序协调**：验证模块间的同步和时序约束
5. **资源管理**：验证跨模块的资源分配和释放

**参考来源**：IEEE Std 1012-2016, "IEEE Standard for System and Software Verification and Validation"

### 3.2 集成测试策略

#### 3.2.1 Big Bang 集成

所有模块一次性集成后测试。

```
[Module A] ----\
[Module B] -----+----> [Big Bang Test]
[Module C] ----/
[Module D] ----/

优点：简单直接，适合小项目
缺点：问题定位困难，不适合 GPU 驱动复杂度
```

#### 3.2.2 自顶向下集成

从顶层模块开始，逐步加入下层模块，使用桩（stub）模拟未集成的下层。

```
测试顺序：
Step 1: [Top Module] + [Stub A] + [Stub B]
Step 2: [Top Module] + [Real A] + [Stub B]
Step 3: [Top Module] + [Real A] + [Real B]

GPU驱动中的应用：
Step 1: DRM ioctl + Stub AMDGPU
Step 2: DRM ioctl + amdgpu_cs + Stub Ring
Step 3: DRM ioctl + amdgpu_cs + Real Ring
```

#### 3.2.3 自底向上集成（推荐用于 GPU 驱动）

从底层模块开始，逐步加入上层模块，使用驱动器（driver）模拟上层调用。

```
测试顺序：
Step 1: [Driver] + [Ring Buffer]
Step 2: [Driver] + [Ring Buffer] + [Scheduler]
Step 3: [Driver] + [Ring Buffer] + [Scheduler] + [Command Submission]

优点：底层功能先行验证，符合 GPU 驱动开发习惯
```

#### 3.2.4 三明治集成

结合自顶向下和自底向上，从两端向中间集成。

```
          [用户空间 ioctl]   (自顶向下)
                |
          [Command Submission]
                |                    中间层
          [Ring Buffer]
                |
          [硬件寄存器访问]     (自底向上)

适用于：GPU 驱动的分层架构
```

### 3.3 AMDGPU 集成测试实现

#### 3.3.1 ioctl 路径集成测试

```c
/*
 * DRM ioctl 路径集成测试
 * 测试 amdgpu_cs_ioctl 的完整处理路径
 */

#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <drm/amdgpu_drm.h>
#include <stdlib.h>
#include <string.h>
#include <stdio.h>

/*
 * 测试场景 1: 基本的 NOP 命令提交
 * 验证空操作的完整提交和完成流程
 */
static int test_cs_nop_basic(void)
{
    int fd, r;
    struct drm_amdgpu_cs_submit submit;
    struct drm_amdgpu_cs_chunk chunks[2];
    struct drm_amdgpu_cs_chunk_ib ib_chunk;
    struct drm_amdgpu_cs_chunk_fence fence_chunk;
    struct drm_amdgpu_cs_chunk_sem sem_chunk;
    uint32_t ib_data[256];
    struct drm_amdgpu_fence fence;

    fd = open("/dev/dri/card0", O_RDWR);
    if (fd < 0) {
        fprintf(stderr, "Failed to open DRI device: %m\n");
        return -1;
    }

    /* 准备 IB 数据：简单的 NOP 包 */
    memset(ib_data, 0, sizeof(ib_data));
    ib_data[0] = 0xc0001000;  /* PACKET3(PACKET3_NOP, 16) */
    ib_data[1] = 0;           /* NOP 填充数据 */
    ib_data[2] = 0;
    ib_data[3] = 0;

    /* 配置 IB chunk */
    memset(&ib_chunk, 0, sizeof(ib_chunk));
    ib_chunk._pad = 0;
    ib_chunk.va_start = (uint64_t)(uintptr_t)ib_data;
    ib_chunk.ib_bytes = 16; /* 4 dwords */
    ib_chunk.ip_type = AMDGPU_HW_IP_GFX;
    ib_chunk.ip_instance = 0;
    ib_chunk.ring = 0;

    chunks[0].chunk_id = AMDGPU_CHUNK_ID_IB;
    chunks[0].length_dw = sizeof(ib_chunk) / 4;
    chunks[0].chunk_data = (uint64_t)(uintptr_t)&ib_chunk;

    /* 配置 fence chunk */
    memset(&fence_chunk, 0, sizeof(fence_chunk));
    chunks[1].chunk_id = AMDGPU_CHUNK_ID_SCHEDULED_FENCE;
    chunks[1].length_dw = sizeof(fence_chunk) / 4;
    chunks[1].chunk_data = (uint64_t)(uintptr_t)&fence_chunk;

    /* 提交 */
    memset(&submit, 0, sizeof(submit));
    submit.context = 0;
    submit.num_chunks = 2;
    submit.chunks = (uint64_t)(uintptr_t)chunks;

    r = ioctl(fd, DRM_IOCTL_AMDGPU_CS, &submit);
    if (r != 0) {
        fprintf(stderr, "CS submit failed: %m\n");
        close(fd);
        return -1;
    }

    /* 等待 fence 完成 */
    fence.handle = 0;
    fence.flags = AMDGPU_FENCE_FLAG_EXE_SUB;
    fence.seq = submit.seq_no;

    r = ioctl(fd, DRM_IOCTL_AMDGPU_WAIT_FENCE, &fence);
    if (r != 0) {
        fprintf(stderr, "Fence wait failed: %m\n");
        close(fd);
        return -1;
    }

    printf("Basic NOP CS: seq_no=%u\n", submit.seq_no);
    close(fd);
    return 0;
}

/*
 * 测试场景 2: 依赖链测试
 * 提交两个命令，第二个等待第一个完成
 */
static int test_cs_dependency_chain(void)
{
    int fd, r;
    struct drm_amdgpu_cs_submit submit[2];
    struct drm_amdgpu_cs_chunk chunks[2][3];
    struct drm_amdgpu_cs_chunk_ib ib_chunk[2];
    struct drm_amdgpu_cs_chunk_fence fence_chunk[2];
    struct drm_amdgpu_cs_chunk_dep dep_chunk;
    uint32_t ib_data[2][64];
    struct drm_amdgpu_fence fence;

    fd = open("/dev/dri/card0", O_RDWR);
    if (fd < 0) return -1;

    /* 提交第一个命令 */
    memset(&submit[0], 0, sizeof(submit[0]));
```

```c
/*
 * 性能基准测试：计算 GFX Ring 命令提交延迟
 */

#include <time.h>
#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>

static uint64_t get_nanos(void)
{
    struct timespec ts;
    clock_gettime(CLOCK_MONOTONIC, &ts);
    return (uint64_t)ts.tv_sec * 1000000000ULL + ts.tv_nsec;
}

/*
 * 测量一次 NOP 提交的延迟
 * 从 ioctl 调用开始到 fence 信号完成
 */
static double measure_cs_latency(int fd, int iterations)
{
    uint64_t start, end;
    double total_ns = 0;
    struct drm_amdgpu_cs_submit submit;
    struct drm_amdgpu_cs_chunk chunks[2];
    struct drm_amdgpu_cs_chunk_ib ib_chunk;
    struct drm_amdgpu_cs_chunk_fence fence_chunk;
    uint32_t ib_data[4];
    struct drm_amdgpu_fence fence;

    memset(&ib_data, 0, sizeof(ib_data));
    ib_data[0] = 0xc0001000;

    for (int i = 0; i < iterations; i++) {
        memset(&ib_chunk, 0, sizeof(ib_chunk));
        ib_chunk.va_start = (uint64_t)(uintptr_t)ib_data;
        ib_chunk.ib_bytes = 16;
        ib_chunk.ip_type = AMDGPU_HW_IP_GFX;
        ib_chunk.ring = 0;

        chunks[0].chunk_id = AMDGPU_CHUNK_ID_IB;
        chunks[0].length_dw = sizeof(ib_chunk) / 4;
        chunks[0].chunk_data = (uint64_t)(uintptr_t)&ib_chunk;

        chunks[1].chunk_id = AMDGPU_CHUNK_ID_SCHEDULED_FENCE;
        chunks[1].length_dw = sizeof(fence_chunk) / 4;
        chunks[1].chunk_data = (uint64_t)(uintptr_t)&fence_chunk;

        submit.num_chunks = 2;
        submit.chunks = (uint64_t)(uintptr_t)chunks;

        start = get_nanos();
        if (ioctl(fd, DRM_IOCTL_AMDGPU_CS, &submit) != 0)
            continue;

        fence.flags = AMDGPU_FENCE_FLAG_EXE_SUB;
        fence.seq = submit.seq_no;
        ioctl(fd, DRM_IOCTL_AMDGPU_WAIT_FENCE, &fence);
        end = get_nanos();

        total_ns += (double)(end - start);
    }

    return total_ns / iterations;
}

int main(void)
{
    int fd = open("/dev/dri/card0", O_RDWR);
    if (fd < 0) {
        perror("open");
        return 1;
    }

    double lat_10 = measure_cs_latency(fd, 10);
    double lat_100 = measure_cs_latency(fd, 100);
    double lat_1000 = measure_cs_latency(fd, 1000);

    printf("CS Latency Analysis:\n");
    printf("  10 iterations:  avg %.2f us\n", lat_10 / 1000.0);
    printf("  100 iterations: avg %.2f us\n", lat_100 / 1000.0);
    printf("  1000 iterations: avg %.2f us\n", lat_1000 / 1000.0);

    close(fd);
    return 0;
}
```

```c
/* 测试类型选择矩阵 */
const char *select_test_type(
    const char *test_target,
    bool has_hardware,
    bool needs_full_stack)
{
    /* 纯逻辑/算法测试 -> 单元测试 */
    if (strcmp(test_target, "buffer_management") == 0 ||

        strcmp(test_target, "fence_signaling") == 0)
        return "unit_test";

    /* 模块交互 -> 集成测试 */
    if (strcmp(test_target, "ioctl_path") == 0 ||
        strcmp(test_target, "ip_block_init") == 0 ||
        strcmp(test_target, "interrupt_handling") == 0)
        return "integration_test";

    /* 全系统 -> 系统测试 */
    if (strcmp(test_target, "24h_stability") == 0 ||
        strcmp(test_target, "power_management") == 0 ||
        strcmp(test_target, "performance_benchmark") == 0)

    if (strcmp(test_target, "game_compatibility") == 0 ||
        strcmp(test_target, "api_conformance") == 0)
        return "acceptance_test";

}
```

### 7.4 集成测试实战：多引擎同步

以下是一个真实的AMDGPU多引擎同步集成测试，验证GFX和SDMA引擎之间的协作：

```c
/*
 * 多引擎同步集成测试
 * 测试场景: GFX引擎计算 -> SDMA引擎DMA传输 -> GFX引擎消费
 * 
 * 测试验证:
 *   1. GFX到SDMA的fence依赖链
 *   2. SDMA传输的正确性
 *   3. SDMA到GFX的fence依赖链
 *   4. 最终结果验证
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <assert.h>
#include <time.h>

struct amdgpu_bo {
    int fd;
    uint32_t handle;
    uint64_t addr;
    uint32_t size;
    void *cpu_map;
};

/*
 * 创建GPU BO并映射到CPU地址空间
 */
static struct amdgpu_bo *create_bo(int fd, uint32_t size)
{
    struct amdgpu_bo *bo = calloc(1, sizeof(*bo));
    struct drm_amdgpu_gem_create create = {
        .bo_size = size,
        .alignment = 4096,
        .domain_flags = AMDGPU_GEM_DOMAIN_VRAM |
                       AMDGPU_GEM_CREATE_CPU_ACCESS_REQUIRED,
    };
    
    bo->fd = fd;
    bo->size = size;
    
    int r = ioctl(fd, DRM_IOCTL_AMDGPU_GEM_CREATE, &create);
    assert(r == 0);
    
    bo->handle = create.handle;
    
    /* MMAP */
    struct drm_amdgpu_gem_mmap mmap_req = {
        .handle = bo->handle,
        .offset = 0,
        .size = bo->size,
    };
    r = ioctl(fd, DRM_IOCTL_AMDGPU_GEM_MMAP, &mmap_req);
    assert(r == 0);
    
    bo->cpu_map = mmap(NULL, bo->size, PROT_READ | PROT_WRITE,
                       MAP_SHARED, fd, mmap_req.addr_ptr);
    assert(bo->cpu_map != MAP_FAILED);
    
    return bo;
}

/*
 * 多引擎同步测试主体
 * 
 * 工作流程:
 *   1. 在VRAM中分配源和目标BO
 *   2. GFX引擎写入源数据
 *   3. GFX引擎发出fence(依赖点A)
 *   4. SDMA引擎等待依赖点A,然后执行DMA传输
 *   5. SDMA引擎发出fence(依赖点B)
 *   6. GFX引擎等待依赖点B,验证目标数据
 */
static int test_multi_engine_sync(void)
{
    int fd = open("/dev/dri/card0", O_RDWR);
    assert(fd >= 0);
    
    const uint32_t DATA_SIZE = 1024 * 1024; /* 1MB */
    const uint32_t PATTERN = 0xDEADBEEF;
    
    /* 分配源和目标BO */
    struct amdgpu_bo *src_bo = create_bo(fd, DATA_SIZE);
    struct amdgpu_bo *dst_bo = create_bo(fd, DATA_SIZE);
    
    /* Step 1: 通过CPU填充源数据 */
    uint32_t *src_data = (uint32_t *)src_bo->cpu_map;
    for (int i = 0; i < DATA_SIZE / 4; i++) {
        src_data[i] = PATTERN ^ i;
    }
    
    /* Step 2: 通过GFX引擎在GPU端验证数据
     * 在实际测试中，这里需要提交一个compute shader
     * 来验证GPU可以正确读取源数据
     */
    struct drm_amdgpu_cs_submit gfx_submit = {0};
    /* ... 构建GFX command submission ... */
    
    /* Step 3: 创建GFX fence作为SDMA的wait点 */
    uint32_t gfx_fence_seq;
    /* ... 获取GFX fence序列号 ... */
    
    /* Step 4: 提交SDMA DMA传输命令
     * SDMA等待GFX fence完成后开始传输
     */
    struct drm_amdgpu_cs_submit sdma_submit = {0};
    /* ... 构建SDMA DMA command submission ... */
    
    /* Step 5: 等待SDMA传输完成 */
    struct drm_amdgpu_fence sdma_fence = {
        .flags = AMDGPU_FENCE_FLAG_EXE_SUB,
        .seq = sdma_submit.seq_no,
    };
    ioctl(fd, DRM_IOCTL_AMDGPU_WAIT_FENCE, &sdma_fence);
    
    /* Step 6: 验证目标数据 */
    uint32_t *dst_data = (uint32_t *)dst_bo->cpu_map;
    int mismatches = 0;
    for (int i = 0; i < DATA_SIZE / 4; i++) {
        uint32_t expected = PATTERN ^ i;
        if (dst_data[i] != expected) {
            mismatches++;
            if (mismatches <= 5) {
                fprintf(stderr, "MISMATCH at [%d]: got 0x%08x, "
                        "expected 0x%08x\n",
                        i, dst_data[i], expected);
            }
        }
    }
    
    printf("\n多引擎同步测试结果:\n");
    printf("  数据大小: %u bytes\n", DATA_SIZE);
    printf("  检查元素: %u\n", DATA_SIZE / 4);
    printf("  不匹配数: %d\n", mismatches);
    
    if (mismatches == 0) {
        printf("  结论: PASS - GFX->SDMA->GFX同步正确\n");
    } else {
        printf("  结论: FAIL - 数据一致性被破坏\n");
    }
    
    /* 清理 */
    munmap(src_bo->cpu_map, src_bo->size);
    munmap(dst_bo->cpu_map, dst_bo->size);
    close(fd);
    
    return mismatches == 0 ? 0 : 1;
}

int main(void)
{
    printf("AMDGPU 多引擎同步集成测试\n");
    printf("==========================\n");
    return test_multi_engine_sync();
}
```

### 7.5 系统测试：长时间稳定性监控

以下是一个生产环境中使用的GPU驱动长时间稳定性测试方案：

```python
#!/usr/bin/env python3
"""
AMDGPU 长时间稳定性测试监控框架
运行时间: 24-72小时
监控指标: GPU温度、频率、功耗、内存使用、错误计数
"""

import subprocess
import time
import csv
import sys
import os
import signal
from datetime import datetime
from collections import deque

class GPUStabilityMonitor:
    def __init__(self, duration_hours=24, interval_seconds=10):
        self.duration = duration_hours * 3600
        self.interval = interval_seconds
        self.start_time = None
        self.metrics_history = deque(maxlen=10000)
        self.errors = []
        self.running = False
        
        # GPU sysfs路径
        self.gpu_path = "/sys/class/drm/card0/device"
        
    def read_gpu_metric(self, metric_file):
        """读取GPU sysfs指标"""
        try:
            path = f"{self.gpu_path}/{metric_file}"
            with open(path, 'r') as f:
                return f.read().strip()
        except (FileNotFoundError, IOError):
            return "N/A"
    
    def collect_metrics(self):
        """收集当前GPU状态"""
        metrics = {
            'timestamp': datetime.now().isoformat(),
            'elapsed_hours': 0,
            'gpu_temp': self.read_gpu_metric("hwmon/hwmon0/temp1_input"),
            'gpu_busy': self.read_gpu_metric("gpu_busy_percent"),
            'mem_busy': self.read_gpu_metric("mem_busy_percent"),
            'vram_usage': self.read_gpu_metric("mem_info_vram_total"),
            'vram_free': self.read_gpu_metric("mem_info_vram_free"),
            'power_level': self.read_gpu_metric("power_dpm_force_performance_level"),
            'pcie_speed': self.read_gpu_metric("current_link_speed"),
            'pcie_width': self.read_gpu_metric("current_link_width"),
        }
        
        if self.start_time:
            elapsed = (datetime.now() - self.start_time).total_seconds()
            metrics['elapsed_hours'] = round(elapsed / 3600, 2)
        
        return metrics
    
    def check_errors(self, metrics):
        """检查异常状态"""
        errors = []
        
        # 温度检查
        try:
            temp_millideg = int(metrics['gpu_temp'])
            temp_celsius = temp_millideg / 1000
            if temp_celsius > 95:
                errors.append(f"高温警告: {temp_celsius}°C")
            elif temp_celsius > 105:
                errors.append(f"临界温度: {temp_celsius}°C - 需要立即处理")
        except ValueError:
            pass
        
        # GPU使用率检查 - 如果长时间100%可能过热
        try:
            busy = int(metrics['gpu_busy'])
            if busy > 95:
                errors.append(f"GPU满载: {busy}%")
        except ValueError:
            pass
        
        # VRAM使用检查
        try:
            vram_total = int(metrics['vram_usage'])
            vram_free = int(metrics['vram_free'])
            vram_used_pct = (vram_total - vram_free) / vram_total * 100
            if vram_used_pct > 95:
                errors.append(f"VRAM使用率高: {vram_used_pct:.1f}%")
        except (ValueError, ZeroDivisionError):
            pass
        
        return errors
    
    def run_workload(self):
        """运行GPU负载测试"""
        workloads = [
            # 轻量3D负载
            "glxgears -geometry 800x600 > /dev/null 2>&1 &",
            # Vulkan计算负载  
            "vulkaninfo --summary > /dev/null 2>&1 &",
            # 视频解码负载
            "ffplay -loglevel quiet -t 60 test_4k.h264 -nodisp &",
        ]
        
        for wl in workloads:
            try:
                subprocess.Popen(wl, shell=True)
            except Exception as e:
                self.errors.append(f"负载启动失败: {wl}: {e}")
        
        # 定期切换负载类型
        time.sleep(300)  # 5分钟后更换负载
    
    def cleanup_workloads(self):
        """清理测试负载进程"""
        for proc in ["glxgears", "vulkaninfo", "ffplay"]:
            try:
                subprocess.run(f"pkill {proc}", shell=True)
            except:
                pass
    
    def run(self):
        """主监控循环"""
        self.start_time = datetime.now()
        self.running = True
        workload_switch_time = time.time()
        
        # CSV日志文件
        log_file = f"gpu_stability_{self.start_time.strftime('%Y%m%d_%H%M%S')}.csv"
        
        print(f"GPU稳定性测试开始 - {self.start_time}")
        print(f"测试时长: {self.duration/3600:.0f}小时")
        print(f"采样间隔: {self.interval}秒")
        print(f"日志文件: {log_file}")
        print("=" * 60)
        
        with open(log_file, 'w', newline='') as csvfile:
            fieldnames = [
                'timestamp', 'elapsed_hours', 'gpu_temp',
                'gpu_busy', 'mem_busy', 'vram_usage',
                'vram_free', 'power_level', 'pcie_speed',
                'pcie_width'
            ]
            writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
            writer.writeheader()
            
            # 启动初始负载
            self.run_workload()
            
            while self.running:
                current_time = time.time()
                elapsed = current_time - time.mktime(
                    self.start_time.timetuple())
                
                if elapsed > self.duration:
                    print(f"\n测试完成 - 达到预定时间 {self.duration/3600:.0f}小时")
                    break
                
                metrics = self.collect_metrics()
                writer.writerow(metrics)
                self.metrics_history.append(metrics)
                
                # 错误检查
                errors = self.check_errors(metrics)
                if errors:
                    self.errors.extend(errors)
                    for err in errors:
                        print(f"[{metrics['timestamp']}] 错误: {err}")
                
                # 每5分钟切换负载
                if current_time - workload_switch_time > 300:
                    self.cleanup_workloads()
                    time.sleep(1)
                    self.run_workload()
                    workload_switch_time = current_time
                
                # 打印状态
                if len(self.metrics_history) % 10 == 0:
                    temp = metrics['gpu_temp']
                    busy = metrics['gpu_busy']
                    hours = metrics['elapsed_hours']
                    print(f"[{hours:.1f}h] 温度:{temp}°C "
                          f"负载:{busy}% 错误:{len(self.errors)}")
                
                time.sleep(self.interval)
        
        self.cleanup_workloads()
        self.print_summary()
    
    def print_summary(self):
        """打印测试摘要"""
        print("\n" + "=" * 60)
        print("稳定性测试摘要")
        print("=" * 60)
        print(f"测试开始: {self.start_time}")
        print(f"测试结束: {datetime.now()}")
        print(f"总运行时间: {self.metrics_history[-1]['elapsed_hours']:.2f}h")
        print(f"采样点数: {len(self.metrics_history)}")
        print(f"错误数: {len(self.errors)}")
        
        if self.errors:
            print("\n错误列表:")
            for err in self.errors[:20]:
                print(f"  - {err}")
        
        if len(self.errors) == 0:
            print("\n结论: PASS - 系统稳定运行无错误")
        else:
            print(f"\n结论: FAIL - 发现 {len(self.errors)} 个错误")

if __name__ == "__main__":
    monitor = GPUStabilityMonitor(duration_hours=24, interval_seconds=10)
    try:
        monitor.run()
    except KeyboardInterrupt:
        print("\n\n测试被用户中断")
        monitor.cleanup_workloads()
        monitor.print_summary()
```

```c
/*
 * 测试类型自动选择函数
 */
enum test_type_decision select_test_type(
    bool has_hardware,
    bool single_component,
    bool component_interaction,
    bool end_to_end,
    bool api_compliance)
{
    /* 无硬件依赖 + 单组件 -> 单元测试 */
    if (!has_hardware && single_component && !component_interaction)
        return TEST_TYPE_UNIT;
    
    /* 有硬件依赖 + 组件交互但不涉及完整流程 -> 集成测试 */
    if (has_hardware && component_interaction && !end_to_end)
        return TEST_TYPE_INTEGRATION;
    
    /* 端到端验证但不涉及API合规 -> 系统测试 */
    if (end_to_end && !api_compliance)
        return TEST_TYPE_SYSTEM;
    
    /* API合规性验证 -> 验收测试 */
    if (api_compliance)
        return TEST_TYPE_ACCEPTANCE;
    
    /* 默认返回集成测试 */
    return TEST_TYPE_INTEGRATION;
}
```

#### 问题2: 集成测试如何管理GPU设备依赖？

```c
/*
 * GPU设备管理抽象层
 * 使集成测试可以灵活管理硬件资源
 */
struct gpu_device_manager {
    int (*open_device)(int card_index);
    void (*close_device)(int fd);
    int (*reset_device)(int fd);
    int (*get_device_info)(int fd, struct gpu_info *info);
    bool (*is_device_available)(int card_index);
    
    /* 多GPU支持 */
    int gpu_count;
    int fds[MAX_GPUS];
};

/*
 * 设备发现和初始化
 */
static struct gpu_device_manager *gpu_devmgr_create(void)
{
    struct gpu_device_manager *mgr = calloc(1, sizeof(*mgr));
    
    /* 扫描所有dri设备 */
    for (int i = 0; i < MAX_GPUS; i++) {
        char path[64];
        snprintf(path, sizeof(path), "/dev/dri/card%d", i);
        
        if (access(path, F_OK) == 0) {
            int fd = open(path, O_RDWR);
            if (fd >= 0) {
                mgr->fds[mgr->gpu_count++] = fd;
                printf("发现GPU设备: %s (fd=%d)\n", path, fd);
            }
        }
    }
    
    if (mgr->gpu_count == 0) {
        fprintf(stderr, "警告: 未发现任何GPU设备\n");
        mgr->gpu_count = 0;
    }
    
    return mgr;
}
```
